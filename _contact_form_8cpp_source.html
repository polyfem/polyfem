73 void ContactForm::update_barrier_stiffness(const Eigen::VectorXd &x, const Eigen::MatrixXd &grad_energy)
74 {
75 if (!use_adaptive_barrier_stiffness())
76 return;
77
78 const Eigen::MatrixXd displaced_surface = compute_displaced_surface(x);
79
80 // The adative stiffness is designed for the non-convergent formulation,
81 // so we need to compute the gradient of the non-convergent barrier.
82 // After we can map it to a good value for the convergent formulation.
83 ipc::CollisionConstraints nonconvergent_constraints;
84 nonconvergent_constraints.set_use_convergent_formulation(false);
85 nonconvergent_constraints.build(
86 collision_mesh_, displaced_surface, dhat_, dmin_, broad_phase_method_);
87 Eigen::VectorXd grad_barrier = nonconvergent_constraints.compute_potential_gradient(
88 collision_mesh_, displaced_surface, dhat_);
89 grad_barrier = collision_mesh_.to_full_dof(grad_barrier);
90
91 barrier_stiffness_ = ipc::initial_barrier_stiffness(
92 ipc::world_bbox_diagonal_length(displaced_surface), dhat_, avg_mass_,
93 grad_energy, grad_barrier, max_barrier_stiffness_);
94
95 if (use_convergent_formulation())
96 {
97 double scaling_factor = 0;
98 if (!nonconvergent_constraints.empty())
99 {
100 const double nonconvergent_potential = nonconvergent_constraints.compute_potential(
101 collision_mesh_, displaced_surface, dhat_);
102
103 update_constraint_set(displaced_surface);
104 const double convergent_potential = constraint_set_.compute_potential(
105 collision_mesh_, displaced_surface, dhat_);
106
107 scaling_factor = nonconvergent_potential / convergent_potential;
108 }
109 else
110 {
111 // Hardcoded difference between the non-convergent and convergent barrier
112 scaling_factor = dhat_ * std::pow(dhat_ + 2 * dmin_, 2);
113 }
114 barrier_stiffness_ *= scaling_factor;
115 max_barrier_stiffness_ *= scaling_factor;
116 }
117
118 // Remove the acceleration scaling from the barrier stiffness because it will be applied later.
119 barrier_stiffness_ /= weight_;
120
121 logger().debug("adaptive barrier form stiffness {}", barrier_stiffness());
122 }

145 Eigen::VectorXd ContactForm::value_per_element_unweighted(const Eigen::VectorXd &x) const
146 {
147 const Eigen::MatrixXd V = compute_displaced_surface(x);
148 assert(V.rows() == collision_mesh_.num_vertices());
149
150 const size_t num_vertices = collision_mesh_.num_vertices();
151
152 if (constraint_set_.empty())
153 {
154 return Eigen::VectorXd::Zero(collision_mesh_.full_num_vertices());
155 }
156
157 const Eigen::MatrixXi &E = collision_mesh_.edges();
158 const Eigen::MatrixXi &F = collision_mesh_.faces();
159
160 auto storage = utils::create_thread_storage<Eigen::VectorXd>(Eigen::VectorXd::Zero(num_vertices));
161
162 utils::maybe_parallel_for(constraint_set_.size(), [&](int start, int end, int thread_id) {
163 Eigen::VectorXd &local_storage = utils::get_local_thread_storage(storage, thread_id);
164
165 for (size_t i = start; i < end; i++)
166 {
167 // Quadrature weight is premultiplied by compute_potential
168 const double potential = constraint_set_[i].compute_potential(V, E, F, dhat_);
169
170 const int n_v = constraint_set_[i].num_vertices();
171 const std::array<long, 4> vis = constraint_set_[i].vertex_ids(E, F);
172 for (int j = 0; j < n_v; j++)
173 {
174 assert(0 <= vis[j] && vis[j] < num_vertices);
175 local_storage[vis[j]] += potential / n_v;
176 }
177 }
178 });
179
180 Eigen::VectorXd out = Eigen::VectorXd::Zero(num_vertices);
181 for (const auto &local_potential : storage)
182 {
183 out += local_potential;
184 }
185
186 Eigen::VectorXd out_full = Eigen::VectorXd::Zero(collision_mesh_.full_num_vertices());
187 for (int i = 0; i < out.size(); i++)
188 out_full[collision_mesh_.to_full_vertex_id(i)] = out[i];
189
190 assert(std::abs(value_unweighted(x) - out_full.sum()) < std::max(1e-10 * out_full.sum(), 1e-10));
191
192 return out_full;
193 }

213 double ContactForm::max_step_size(const Eigen::VectorXd &x0, const Eigen::VectorXd &x1) const
214 {
215 // Extract surface only
216 const Eigen::MatrixXd V0 = compute_displaced_surface(x0);
217 const Eigen::MatrixXd V1 = compute_displaced_surface(x1);
218
219 if (save_ccd_debug_meshes)
220 {
221 const Eigen::MatrixXi E = collision_mesh_.dim() == 2 ? Eigen::MatrixXi() : collision_mesh_.edges();
222 const Eigen::MatrixXi &F = collision_mesh_.faces();
223 igl::writePLY(resolve_output_path("debug_ccd_0.ply"), V0, F, E);
224 igl::writePLY(resolve_output_path("debug_ccd_1.ply"), V1, F, E);
225 }
226
227 double max_step;
228 if (use_cached_candidates_ && broad_phase_method_ != ipc::BroadPhaseMethod::SWEEP_AND_TINIEST_QUEUE_GPU)
229 max_step = candidates_.compute_collision_free_stepsize(
230 collision_mesh_, V0, V1, dmin_, ccd_tolerance_, ccd_max_iterations_);
231 else
232 max_step = ipc::compute_collision_free_stepsize(
233 collision_mesh_, V0, V1, broad_phase_method_, ccd_tolerance_, ccd_max_iterations_);
234
235 if (save_ccd_debug_meshes && ipc::has_intersections(collision_mesh_, (V1 - V0) * max_step + V0))
236 {
237 log_and_throw_error("Taking max_step results in intersections (max_step={})", max_step);
238 }
239
240#ifndef NDEBUG
241 // This will check for static intersections as a failsafe. Not needed if we use our conservative CCD.
242 Eigen::MatrixXd V_toi = (V1 - V0) * max_step + V0;
243
244 while (ipc::has_intersections(collision_mesh_, V_toi))
245 {
246 logger().error("Taking max_step results in intersections (max_step={:g})", max_step);
247 max_step /= 2.0;
248
249 const double Linf = (V_toi - V0).lpNorm<Eigen::Infinity>();
250 if (max_step <= 0 || Linf == 0)
251 log_and_throw_error("Unable to find an intersection free step size (max_step={:g} L={:g})", max_step, Linf);
252
253 V_toi = (V1 - V0) * max_step + V0;
254 }
255#endif
256
257 return max_step;
258 }

311 bool ContactForm::is_step_collision_free(const Eigen::VectorXd &x0, const Eigen::VectorXd &x1) const
312 {
313 const auto displaced0 = compute_displaced_surface(x0);
314 const auto displaced1 = compute_displaced_surface(x1);
315
316 // Skip CCD if the displacement is zero.
317 if ((displaced1 - displaced0).lpNorm<Eigen::Infinity>() == 0.0)
318 {
319 // Assumes initially intersection-free
320 return true;
321 }
322
323 bool is_valid;
324 if (use_cached_candidates_)
325 is_valid = candidates_.is_step_collision_free(
326 collision_mesh_, displaced0, displaced1, dmin_,
327 ccd_tolerance_, ccd_max_iterations_);
328 else
329 is_valid = ipc::is_step_collision_free(
330 collision_mesh_, displaced0, displaced1, broad_phase_method_,
331 dmin_, ccd_tolerance_, ccd_max_iterations_);
332
333 return is_valid;
