250 void utils::BoundarySampler::sample_parametric_quad_face(int index, int n_samples, Eigen::MatrixXd &uv, Eigen::MatrixXd &samples)
251 {
252 auto endpoints = hex_local_node_coordinates_from_face(index);
253 const Eigen::VectorXd t = Eigen::VectorXd::LinSpaced(n_samples, 0, 1);
254 samples.resize(n_samples * n_samples, endpoints.cols());
255 Eigen::MatrixXd left(n_samples, endpoints.cols());
256 Eigen::MatrixXd right(n_samples, endpoints.cols());
257
258 uv.resize(n_samples * n_samples, endpoints.cols());
259 uv.setZero();
260
261 for (int c = 0; c < 3; ++c)
262 {
263 left.col(c) = (1.0 - t.array()).matrix() * endpoints(0, c) + t * endpoints(1, c);
264 right.col(c) = (1.0 - t.array()).matrix() * endpoints(3, c) + t * endpoints(2, c);
265 }
266 for (int c = 0; c < 3; ++c)
267 {
268 Eigen::MatrixXd x = (1.0 - t.array()).matrix() * left.col(c).transpose() + t * right.col(c).transpose();
269 assert(x.size() == n_samples * n_samples);
270
271 samples.col(c) = Eigen::Map<const Eigen::VectorXd>(x.data(), x.size());
272 }

275 void utils::BoundarySampler::sample_parametric_tri_face(int index, int n_samples, Eigen::MatrixXd &uv, Eigen::MatrixXd &samples)
276 {
277 auto endpoints = tet_local_node_coordinates_from_face(index);
278 const Eigen::VectorXd t = Eigen::VectorXd::LinSpaced(n_samples, 0, 1);
279 samples.resize(n_samples * n_samples, endpoints.cols());
280
281 uv.resize(n_samples * n_samples, 3);
282
283 int counter = 0;
284 for (int u = 0; u < n_samples; ++u)
285 {
286 for (int v = 0; v < n_samples; ++v)
287 {
288 if (t(u) + t(v) > 1)
289 continue;
290
291 uv(counter, 0) = t(u);
292 uv(counter, 1) = t(v);
293 uv(counter, 2) = 1 - t(u) - t(v);
294
295 for (int c = 0; c < 3; ++c)
296 {
297 samples(counter, c) = t(u) * endpoints(0, c) + t(v) * endpoints(1, c) + (1 - t(u) - t(v)) * endpoints(2, c);
298 }
299 ++counter;
300 }
301 }
302 samples.conservativeResize(counter, 3);
303 uv.conservativeResize(counter, 3);

462 bool utils::BoundarySampler::boundary_quadrature(const LocalBoundary &local_boundary, const int order, const Mesh &mesh, const bool skip_computation, Eigen::MatrixXd &uv, Eigen::MatrixXd &points, Eigen::MatrixXd &normals, Eigen::VectorXd &weights, Eigen::VectorXi &global_primitive_ids)
463 {
464 uv.resize(0, 0);
465 points.resize(0, 0);
466 normals.resize(0, 0);
467 weights.resize(0);
468 global_primitive_ids.resize(0);
469
470 for (int i = 0; i < local_boundary.size(); ++i)
471 {
472 const int gid = local_boundary.global_primitive_id(i);
473 Eigen::MatrixXd tmp_p, tmp_uv, tmp_n;
474 Eigen::VectorXd tmp_w;
475 switch (local_boundary.type())
476 {
477 case BoundaryType::TriLine:
478 quadrature_for_tri_edge(local_boundary[i], order, gid, mesh, tmp_uv, tmp_p, tmp_w);
479 normal_for_tri_edge(local_boundary[i], tmp_n);
480 break;
481 case BoundaryType::QuadLine:
482 quadrature_for_quad_edge(local_boundary[i], order, gid, mesh, tmp_uv, tmp_p, tmp_w);
483 normal_for_quad_edge(local_boundary[i], tmp_n);
484 break;
485 case BoundaryType::Quad:
486 quadrature_for_quad_face(local_boundary[i], order, gid, mesh, tmp_uv, tmp_p, tmp_w);
487 normal_for_quad_face(local_boundary[i], tmp_n);
488 break;
489 case BoundaryType::Tri:
490 quadrature_for_tri_face(local_boundary[i], order, gid, mesh, tmp_uv, tmp_p, tmp_w);
491 normal_for_tri_face(local_boundary[i], tmp_n);
492 break;
493 case BoundaryType::Polygon:
494 quadrature_for_polygon_edge(local_boundary.element_id(), local_boundary.global_primitive_id(i), order, mesh, tmp_uv, tmp_p, tmp_w);
495 normal_for_polygon_edge(local_boundary.element_id(), local_boundary.global_primitive_id(i), mesh, tmp_n);
496 break;
497 case BoundaryType::Invalid:
498 assert(false);
499 break;
500 default:
501 assert(false);
502 }
503
504 uv.conservativeResize(uv.rows() + tmp_uv.rows(), tmp_uv.cols());
505 uv.bottomRows(tmp_uv.rows()) = tmp_uv;
506
507 points.conservativeResize(points.rows() + tmp_p.rows(), tmp_p.cols());
508 points.bottomRows(tmp_p.rows()) = tmp_p;
509
510 normals.conservativeResize(normals.rows() + tmp_p.rows(), tmp_p.cols());
511 for (int k = normals.rows() - tmp_p.rows(); k < normals.rows(); ++k)
512 normals.row(k) = tmp_n;
513
514 weights.conservativeResize(weights.rows() + tmp_w.rows(), tmp_w.cols());
515 weights.bottomRows(tmp_w.rows()) = tmp_w;
516
517 global_primitive_ids.conservativeResize(global_primitive_ids.rows() + tmp_p.rows());
518 global_primitive_ids.bottomRows(tmp_p.rows()).setConstant(gid);
519 }
520
521 assert(uv.rows() == global_primitive_ids.size());
522 assert(points.rows() == global_primitive_ids.size());
523 assert(normals.rows() == global_primitive_ids.size());
524 assert(weights.size() == global_primitive_ids.size());
525
526 return true;
527 }

530 {

559 assert(false);
560 }
561
562 uv.conservativeResize(uv.rows() + tmp_uv.rows(), tmp_uv.cols());
563 uv.bottomRows(tmp_uv.rows()) = tmp_uv;
564

566 samples.bottomRows(tmp.rows()) = tmp;
567
568 global_primitive_ids.conservativeResize(global_primitive_ids.rows() + tmp.rows());
569 global_primitive_ids.bottomRows(tmp.rows()).setConstant(local_boundary.global_primitive_id(i));
570 }
