368 void VTUWriter::add_field(const std::string &name, const Eigen::MatrixXd &data)
369 {
370 using std::abs;
371
372 Eigen::MatrixXd tmp;
373 tmp.resizeLike(data);
374
375 for (long i = 0; i < data.size(); ++i)
376 tmp(i) = abs(data(i)) < 1e-16 ? 0 : data(i);
377
378 if (tmp.cols() == 1)
379 add_scalar_field(name, tmp);
380 else
381 add_vector_field(name, tmp);
