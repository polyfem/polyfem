34 void replace_rows_by_identity(StiffnessMatrix &reduced_mat, const StiffnessMatrix &mat, const std::vector<int> &rows)
35 {
36 reduced_mat.resize(mat.rows(), mat.cols());
37
38 std::vector<bool> mask(mat.rows(), false);
39 for (int i : rows)
40 mask[i] = true;
41
42 std::vector<Eigen::Triplet<double>> coeffs;
43 for (int k = 0; k < mat.outerSize(); ++k)
44 {
45 for (StiffnessMatrix::InnerIterator it(mat, k); it; ++it)
46 {
47 if (mask[it.row()])
48 {
49 if (it.row() == it.col())
50 coeffs.emplace_back(it.row(), it.col(), 1.0);
51 }
52 else
53 coeffs.emplace_back(it.row(), it.col(), it.value());
54 }
55 }
56 reduced_mat.setFromTriplets(coeffs.begin(), coeffs.end());
57 }

59 void compute_force_jacobian(State &state, const Eigen::MatrixXd &sol, const Eigen::MatrixXd &disp_grad, StiffnessMatrix &hessian)
60 {
61 auto &s = state;
62
63 if (s.problem->is_time_dependent())
64 {
65 if (s.assembler->is_linear() && !s.is_contact_enabled())
66 log_and_throw_adjoint_error("Differentiable transient linear solve is not supported!");
67
68 StiffnessMatrix tmp_hess;
69 s.solve_data.nl_problem->set_project_to_psd(false);
70 s.solve_data.nl_problem->FullNLProblem::solution_changed(sol);
71 s.solve_data.nl_problem->FullNLProblem::hessian(sol, tmp_hess);
72 hessian.setZero();
73 replace_rows_by_identity(hessian, tmp_hess, s.boundary_nodes);
74 }
75 else // static formulation
76 {
77 if (s.assembler->is_linear() && !s.is_contact_enabled() && !s.is_homogenization())
78 {
79 hessian.setZero();
80 StiffnessMatrix stiffness;
81 s.build_stiffness_mat(stiffness);
82 replace_rows_by_identity(hessian, stiffness, s.boundary_nodes);
83 }
84 else
85 {
86 s.solve_data.nl_problem->set_project_to_psd(false);
87 if (s.is_homogenization())
88 {
89 Eigen::VectorXd reduced;
90 std::shared_ptr<solver::NLHomoProblem> homo_problem = std::dynamic_pointer_cast<solver::NLHomoProblem>(s.solve_data.nl_problem);
91 reduced = homo_problem->full_to_reduced(sol, disp_grad);
92 s.solve_data.nl_problem->solution_changed(reduced);
93 s.solve_data.nl_problem->hessian(reduced, hessian);
94 }
95 else
96 {
97 StiffnessMatrix tmp_hess;
98 s.solve_data.nl_problem->FullNLProblem::solution_changed(sol);
99 s.solve_data.nl_problem->FullNLProblem::hessian(sol, tmp_hess);
100 hessian.setZero();
101 replace_rows_by_identity(hessian, tmp_hess, s.boundary_nodes);
102 }
103 }
104 }

202 void DiffCache::cache_transient(
203 int step,
204 State &state,
205 const Eigen::MatrixXd &sol,
206 const Eigen::MatrixXd *disp_grad,
207 const Eigen::MatrixXd *pressure)
208 {
209 auto &s = state;
210 if (pressure)
211 {
212 log_and_throw_adjoint_error("Navier stoke problem is not supported in adjoint optimization.");
213 }
214
215 Eigen::MatrixXd disp_grad_final;
216 if (disp_grad)
217 {
218 disp_grad_final = *disp_grad;
219 }
220 else
221 {
222 int mesh_dim = s.mesh->dimension();
223 disp_grad_final = Eigen::MatrixXd::Zero(mesh_dim, mesh_dim);
224 }
225
226 StiffnessMatrix gradu_h(sol.size(), sol.size());
227 if (step == 0)
228 {
229 init(s.mesh->dimension(), s.ndof(), s.problem->is_time_dependent() ? s.args["time"]["time_steps"].get<int>() : 0);
230 }
231
232 ipc::NormalCollisions cur_collision_set;
233 ipc::SmoothCollisions cur_smooth_collision_set;
234 ipc::TangentialCollisions cur_friction_set;
235 ipc::NormalCollisions cur_normal_adhesion_set;
236 ipc::TangentialCollisions cur_tangential_adhesion_set;
237
238 if (s.optimization_enabled == solver::CacheLevel::Derivatives)
239 {
240 if (!s.problem->is_time_dependent() || step > 0)
241 compute_force_jacobian(s, sol, disp_grad_final, gradu_h);
242
243 if (s.solve_data.contact_form)
244 {
245 if (const auto barrier_contact = dynamic_cast<const solver::BarrierContactForm *>(s.solve_data.contact_form.get()))
246 cur_collision_set = barrier_contact->collision_set();
247 else if (const auto smooth_contact = dynamic_cast<const solver::SmoothContactForm *>(s.solve_data.contact_form.get()))
248 cur_smooth_collision_set = smooth_contact->collision_set();
249 }
250 cur_friction_set = s.solve_data.friction_form ? s.solve_data.friction_form->friction_collision_set() : ipc::TangentialCollisions();
251 cur_normal_adhesion_set = s.solve_data.normal_adhesion_form ? s.solve_data.normal_adhesion_form->collision_set() : ipc::NormalCollisions();
252 cur_tangential_adhesion_set = s.solve_data.tangential_adhesion_form ? s.solve_data.tangential_adhesion_form->tangential_collision_set() : ipc::TangentialCollisions();
253 }
254
255 if (s.problem->is_time_dependent())
256 {
257 if (s.args["time"]["quasistatic"].get<bool>())
258 {
259 cache_quantities_quasistatic(step, sol, gradu_h, cur_collision_set, cur_smooth_collision_set, cur_normal_adhesion_set, disp_grad_final);
260 }
261 else
262 {
263 Eigen::MatrixXd vel, acc;
264 if (step == 0)
265 {
266 if (dynamic_cast<time_integrator::BDF *>(s.solve_data.time_integrator.get()))
267 {
268 const auto bdf_integrator = dynamic_cast<time_integrator::BDF *>(s.solve_data.time_integrator.get());
269 vel = bdf_integrator->weighted_sum_v_prevs();
270 }
271 else if (dynamic_cast<time_integrator::ImplicitEuler *>(s.solve_data.time_integrator.get()))
272 {
273 const auto euler_integrator = dynamic_cast<time_integrator::ImplicitEuler *>(s.solve_data.time_integrator.get());
274 vel = euler_integrator->v_prev();
275 }
276 else
277 log_and_throw_error("Differentiable code doesn't support this time integrator!");
278
279 acc.setZero(s.ndof(), 1);
280 }
281 else
282 {
283 vel = s.solve_data.time_integrator->compute_velocity(sol);
284 acc = s.solve_data.time_integrator->compute_acceleration(vel);
285 }
286
287 cache_quantities_transient(step, s.solve_data.time_integrator->steps(), sol, vel, acc, gradu_h, cur_collision_set, cur_smooth_collision_set, cur_friction_set);
288 }
289 }
290 else
291 {
292 cache_quantities_static(sol, gradu_h, cur_collision_set, cur_smooth_collision_set, cur_friction_set, cur_normal_adhesion_set, cur_tangential_adhesion_set, disp_grad_final);
293 }
