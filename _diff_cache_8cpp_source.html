42 void replace_rows_by_identity(StiffnessMatrix &reduced_mat, const StiffnessMatrix &mat, const std::vector<int> &rows)
43 {
44 reduced_mat.resize(mat.rows(), mat.cols());
45
46 std::vector<bool> mask(mat.rows(), false);
47 for (int i : rows)
48 mask[i] = true;
49
50 std::vector<Eigen::Triplet<double>> coeffs;
51 for (int k = 0; k < mat.outerSize(); ++k)
52 {
53 for (StiffnessMatrix::InnerIterator it(mat, k); it; ++it)
54 {
55 if (mask[it.row()])
56 {
57 if (it.row() == it.col())
58 coeffs.emplace_back(it.row(), it.col(), 1.0);
59 }
60 else
61 coeffs.emplace_back(it.row(), it.col(), it.value());
62 }
63 }
64 reduced_mat.setFromTriplets(coeffs.begin(), coeffs.end());
65 }

67 void compute_force_jacobian(State &state, const Eigen::MatrixXd &sol, const Eigen::MatrixXd &disp_grad, StiffnessMatrix &hessian)
68 {
69 auto &s = state;
70
71 if (s.problem->is_time_dependent())
72 {
73 if (s.assembler->is_linear() && !s.is_contact_enabled())
74 log_and_throw_adjoint_error("Differentiable transient linear solve is not supported!");
75
76 StiffnessMatrix tmp_hess;
77 s.solve_data.nl_problem->set_project_to_psd(false);
78 s.solve_data.nl_problem->FullNLProblem::solution_changed(sol);
79 s.solve_data.nl_problem->FullNLProblem::hessian(sol, tmp_hess);
80 hessian.setZero();
81 replace_rows_by_identity(hessian, tmp_hess, s.boundary_nodes);
82 }
83 else // static formulation
84 {
85 if (s.assembler->is_linear() && !s.is_contact_enabled() && !s.is_homogenization())
86 {
87 hessian.setZero();
88 StiffnessMatrix stiffness;
89 s.build_stiffness_mat(stiffness);
90 replace_rows_by_identity(hessian, stiffness, s.boundary_nodes);
91 }
92 else
93 {
94 s.solve_data.nl_problem->set_project_to_psd(false);
95 if (s.is_homogenization())
96 {
97 Eigen::VectorXd reduced;
98 std::shared_ptr<solver::NLHomoProblem> homo_problem = std::dynamic_pointer_cast<solver::NLHomoProblem>(s.solve_data.nl_problem);
99 reduced = homo_problem->full_to_reduced(sol, disp_grad);
100 s.solve_data.nl_problem->solution_changed(reduced);
101 s.solve_data.nl_problem->hessian(reduced, hessian);
102 }
103 else
104 {
105 StiffnessMatrix tmp_hess;
106 s.solve_data.nl_problem->FullNLProblem::solution_changed(sol);
107 s.solve_data.nl_problem->FullNLProblem::hessian(sol, tmp_hess);
108 hessian.setZero();
109 replace_rows_by_identity(hessian, tmp_hess, s.boundary_nodes);
110 }
111 }
112 }

115 StiffnessMatrix compute_basis_nodes_to_gbasis_nodes(const State &state)
116 {
117 auto &gbases = state.geom_bases();
118 auto &bases = state.bases;
119
120 std::map<std::array<int, 2>, double> pairs;
121 for (int e = 0; e < gbases.size(); e++)
122 {
123 auto &gbs = gbases[e].bases;
124 auto &bs = bases[e].bases;
125 assert(!bs.empty());
126
127 Eigen::MatrixXd local_pts;
128 int order = bs.front().order();
129 if (state.mesh->is_volume())
130 {
131 if (state.mesh->is_simplex(e))
132 {
133 autogen::p_nodes_3d(order, local_pts);
134 }
135 else
136 {
137 autogen::q_nodes_3d(order, local_pts);
138 }
139 }
140 else
141 {
142 if (state.mesh->is_simplex(e))
143 {
144 autogen::p_nodes_2d(order, local_pts);
145 }
146 else
147 {
148 autogen::q_nodes_2d(order, local_pts);
149 }
150 }
151
152 assembler::ElementAssemblyValues vals;
153 vals.compute(e, state.mesh->is_volume(), local_pts, gbases[e], gbases[e]);
154
155 for (int i = 0; i < bs.size(); i++)
156 {
157 for (int j = 0; j < gbs.size(); j++)
158 {
159 if (std::abs(vals.basis_values[j].val(i)) > 1e-7)
160 {
161 std::array<int, 2> index = {{gbs[j].global()[0].index, bs[i].global()[0].index}};
162 pairs.insert({index, vals.basis_values[j].val(i)});
163 }
164 }
165 }
166 }
167
168 int dim = state.mesh->dimension();
169 std::vector<Eigen::Triplet<double>> coeffs;
170 coeffs.reserve(pairs.size() * dim);
171 for (const auto &iter : pairs)
172 {
173 for (int d = 0; d < dim; d++)
174 {
175 coeffs.emplace_back(iter.first[0] * dim + d, iter.first[1] * dim + d, iter.second);
176 }
177 }
178
179 StiffnessMatrix mapping;
180 mapping.resize(state.n_geom_bases * dim, state.n_bases * dim);
181 mapping.setFromTriplets(coeffs.begin(), coeffs.end());
182 return mapping;
183 }

288 void DiffCache::cache_transient(
289 int step,
290 State &state,
291 const Eigen::MatrixXd &sol,
292 const Eigen::MatrixXd *disp_grad,
293 const Eigen::MatrixXd *pressure)
294 {
295 auto &s = state;
296 if (pressure)
297 {
298 log_and_throw_adjoint_error("Navier stoke problem is not supported in adjoint optimization.");
299 }
300
301 if (step == 0)
302 {
303 basis_nodes_to_gbasis_nodes_ = compute_basis_nodes_to_gbasis_nodes(s);
304 }
305
306 Eigen::MatrixXd disp_grad_final;
307 if (disp_grad)
308 {
309 disp_grad_final = *disp_grad;
310 }
311 else
312 {
313 int mesh_dim = s.mesh->dimension();
314 disp_grad_final = Eigen::MatrixXd::Zero(mesh_dim, mesh_dim);
315 }
316
317 StiffnessMatrix gradu_h(sol.size(), sol.size());
318 if (step == 0)
319 {
320 init(s.mesh->dimension(), s.ndof(), s.problem->is_time_dependent() ? s.args["time"]["time_steps"].get<int>() : 0);
321 }
322
323 ipc::NormalCollisions cur_collision_set;
324 ipc::SmoothCollisions cur_smooth_collision_set;
325 ipc::TangentialCollisions cur_friction_set;
326 ipc::NormalCollisions cur_normal_adhesion_set;
327 ipc::TangentialCollisions cur_tangential_adhesion_set;
328
329 if (s.optimization_enabled == solver::CacheLevel::Derivatives)
330 {
331 if (!s.problem->is_time_dependent() || step > 0)
332 compute_force_jacobian(s, sol, disp_grad_final, gradu_h);
333
334 if (s.solve_data.contact_form)
335 {
336 if (const auto barrier_contact = dynamic_cast<const solver::BarrierContactForm *>(s.solve_data.contact_form.get()))
337 cur_collision_set = barrier_contact->collision_set();
338 else if (const auto smooth_contact = dynamic_cast<const solver::SmoothContactForm *>(s.solve_data.contact_form.get()))
339 cur_smooth_collision_set = smooth_contact->collision_set();
340 }
341 cur_friction_set = s.solve_data.friction_form ? s.solve_data.friction_form->friction_collision_set() : ipc::TangentialCollisions();
342 cur_normal_adhesion_set = s.solve_data.normal_adhesion_form ? s.solve_data.normal_adhesion_form->collision_set() : ipc::NormalCollisions();
343 cur_tangential_adhesion_set = s.solve_data.tangential_adhesion_form ? s.solve_data.tangential_adhesion_form->tangential_collision_set() : ipc::TangentialCollisions();
344 }
345
346 if (s.problem->is_time_dependent())
347 {
348 if (s.args["time"]["quasistatic"].get<bool>())
349 {
350 cache_quantities_quasistatic(step, sol, gradu_h, cur_collision_set, cur_smooth_collision_set, cur_normal_adhesion_set, disp_grad_final);
351 }
352 else
353 {
354 Eigen::MatrixXd vel, acc;
355 if (step == 0)
356 {
357 if (dynamic_cast<time_integrator::BDF *>(s.solve_data.time_integrator.get()))
358 {
359 const auto bdf_integrator = dynamic_cast<time_integrator::BDF *>(s.solve_data.time_integrator.get());
360 vel = bdf_integrator->weighted_sum_v_prevs();
361 }
362 else if (dynamic_cast<time_integrator::ImplicitEuler *>(s.solve_data.time_integrator.get()))
363 {
364 const auto euler_integrator = dynamic_cast<time_integrator::ImplicitEuler *>(s.solve_data.time_integrator.get());
365 vel = euler_integrator->v_prev();
366 }
367 else
368 log_and_throw_error("Differentiable code doesn't support this time integrator!");
369
370 acc.setZero(s.ndof(), 1);
371 }
372 else
373 {
374 vel = s.solve_data.time_integrator->compute_velocity(sol);
375 acc = s.solve_data.time_integrator->compute_acceleration(vel);
376 }
377
378 cache_quantities_transient(step, s.solve_data.time_integrator->steps(), sol, vel, acc, gradu_h, cur_collision_set, cur_smooth_collision_set, cur_friction_set);
379 }
380 }
381 else
382 {
383 cache_quantities_static(sol, gradu_h, cur_collision_set, cur_smooth_collision_set, cur_friction_set, cur_normal_adhesion_set, cur_tangential_adhesion_set, disp_grad_final);
384 }
