161 void NCMesh3D::compute_boundary_ids(const double eps)
162 {
163 boundary_ids_.resize(n_faces());
164 std::fill(boundary_ids_.begin(), boundary_ids_.end(), -1);
165
166 RowVectorNd minV, maxV;
167 bounding_box(minV, maxV);
168
169 for (int f = 0; f < n_faces(); ++f)
170 {
171 if (!is_boundary_face(f))
172 continue;
173
174 const auto p = face_barycenter(f);
175
176 if (fabs(p(0) - minV(0)) < eps)
177 boundary_ids_[f] = 1;
178 else if (fabs(p(1) - minV(1)) < eps)
179 boundary_ids_[f] = 2;
180 else if (fabs(p(2) - minV(2)) < eps)
181 boundary_ids_[f] = 5;
182
183 else if (fabs(p(0) - maxV(0)) < eps)
184 boundary_ids_[f] = 3;
185 else if (fabs(p(1) - maxV(1)) < eps)
186 boundary_ids_[f] = 4;
187 else if (fabs(p(2) - maxV(2)) < eps)
188 boundary_ids_[f] = 6;
189 else
190 boundary_ids_[f] = 7;
191
192 faces[valid_to_all_face(f)].boundary_id = boundary_ids_[f];

294 void NCMesh3D::triangulate_faces(Eigen::MatrixXi &tris, Eigen::MatrixXd &pts, std::vector<int> &ranges) const
295 {
296 ranges.clear();
297
298 std::vector<Eigen::MatrixXi> local_tris(n_cells());
299 std::vector<Eigen::MatrixXd> local_pts(n_cells());
300 Eigen::MatrixXi tets;
301
302 int total_tris = 0;
303 int total_pts = 0;
304
305 ranges.push_back(0);
306
307 Eigen::MatrixXd face_barys;
308 face_barycenters(face_barys);
309
310 Eigen::MatrixXd cell_barys;
311 cell_barycenters(cell_barys);
312
313 for (std::size_t e = 0; e < n_cells(); ++e)
314 {
315 const int n_vertices = n_cell_vertices(e);
316 const int n_faces = n_cell_faces(e);
317
318 Eigen::MatrixXd local_pt(n_vertices + n_faces, 3);
319
320 std::map<int, int> global_to_local;
321
322 for (int i = 0; i < n_vertices; ++i)
323 {
324 const int global_index = cell_vertex(e, i);
325 local_pt.row(i) = point(global_index);
326 global_to_local[global_index] = i;
327 }
328
329 int n_local_faces = 0;
330 for (int i = 0; i < n_faces; ++i)
331 {
332 const int f_id = cell_face(e, i);
333 n_local_faces += n_face_vertices(f_id);
334
335 local_pt.row(n_vertices + i) = face_barys.row(f_id);
336 }
337
338 Eigen::MatrixXi local_faces(n_local_faces, 3);
339
340 int face_index = 0;
341 for (int i = 0; i < n_faces; ++i)
342 {
343 const int f_id = cell_face(e, i);
344
345 const Eigen::RowVector3d e0 = (point(face_vertex(f_id, 0)) - local_pt.row(n_vertices + i));
346 const Eigen::RowVector3d e1 = (point(face_vertex(f_id, 1)) - local_pt.row(n_vertices + i));
347 const Eigen::RowVector3d normal = e0.cross(e1);
348 // const Eigen::RowVector3d check_dir = (node_from_element(e)-p);
349 const Eigen::RowVector3d check_dir = (cell_barys.row(e) - point(face_vertex(f_id, 1)));
350
351 const bool reverse_order = normal.dot(check_dir) > 0;
352
353 for (int j = 0; j < n_face_vertices(f_id); ++j)
354 {
355 const int jp = (j + 1) % n_face_vertices(f_id);
356 if (reverse_order)
357 {
358 local_faces(face_index, 0) = global_to_local[face_vertex(f_id, jp)];
359 local_faces(face_index, 1) = global_to_local[face_vertex(f_id, j)];
360 }
361 else
362 {
363 local_faces(face_index, 0) = global_to_local[face_vertex(f_id, j)];
364 local_faces(face_index, 1) = global_to_local[face_vertex(f_id, jp)];
365 }
366 local_faces(face_index, 2) = n_vertices + i;
367
368 ++face_index;
369 }
370 }
371
372 local_pts[e] = local_pt;
373 local_tris[e] = local_faces;
374
375 total_tris += local_tris[e].rows();
376 total_pts += local_pts[e].rows();
377
378 ranges.push_back(total_tris);
379
380 assert(local_pts[e].rows() == local_pt.rows());
381 }
382
383 tris.resize(total_tris, 3);
384 pts.resize(total_pts, 3);
385
386 int tri_index = 0;
387 int pts_index = 0;
388 for (std::size_t i = 0; i < local_tris.size(); ++i)
389 {
390 tris.block(tri_index, 0, local_tris[i].rows(), local_tris[i].cols()) = local_tris[i].array() + pts_index;
391 tri_index += local_tris[i].rows();
392
393 pts.block(pts_index, 0, local_pts[i].rows(), local_pts[i].cols()) = local_pts[i];
394 pts_index += local_pts[i].rows();
395 }
396 }

468 Navigation3D::Index NCMesh3D::get_index_from_element_face(int hi, int v0, int v1, int v2) const
469 {
470 Navigation3D::Index idx;
471
472 idx.element = hi;
473 idx.vertex = v0;
474 idx.face = all_to_valid_face(find_face(valid_to_all_vertex(v0), valid_to_all_vertex(v1), valid_to_all_vertex(v2)));
475
476 for (int d = 0; d < n_face_vertices(idx.face); d++)
477 if (face_vertex(idx.face, d) == idx.vertex)
478 idx.face_corner = d;
479
480 int v0_ = v0, v1_ = v1;
481
482 // let v0_ < v1_
483 if (v0_ > v1_)
484 std::swap(v0_, v1_);
485
486 for (int i = 0; i < 4; i++)
487 {
488 const int fid = cell_face(idx.element, i);
489 if (fid != idx.face)

622 void NCMesh3D::refine_element(int id_full)
623 {
624 assert(elements[id_full].is_valid());
625 if (elements[id_full].is_not_valid())
626 throw std::runtime_error("Cannot refine an invalid element!");
627
628 const auto v = elements[id_full].vertices;
629 elements[id_full].is_refined = true;
630 n_elements--;
631
632 for (int f = 0; f < elements[id_full].faces.size(); f++)
633 faces[elements[id_full].faces(f)].remove_element(id_full);
634
635 for (int e = 0; e < elements[id_full].edges.size(); e++)
636 edges[elements[id_full].edges(e)].remove_element(id_full);
637
638 for (int i = 0; i < v.size(); i++)
639 vertices[v(i)].remove_element(id_full);
640
641 if (elements[id_full].children(0) >= 0)
642 {
643 for (int c = 0; c < elements[id_full].children.size(); c++)
644 {
645 const int child_id = elements[id_full].children(c);
646 auto &elem = elements[child_id];
647 elem.is_ghost = false;
648 n_elements++;
649
650 for (int f = 0; f < elem.faces.size(); f++)
651 faces[elem.faces(f)].add_element(child_id);
652
653 for (int e = 0; e < elem.edges.size(); e++)
654 edges[elem.edges(e)].add_element(child_id);
655
656 for (int v = 0; v < elem.vertices.size(); v++)
657 vertices[elem.vertices(v)].add_element(child_id);
658 }
659 }
660 else
661 {
662 // create mid-points if not exist
663 const int v1 = v[0];
664 const int v2 = v[1];
665 const int v3 = v[2];
666 const int v4 = v[3];
667 const int v5 = get_vertex(Eigen::Vector2i(v1, v2));
668 const int v8 = get_vertex(Eigen::Vector2i(v3, v2));
669 const int v6 = get_vertex(Eigen::Vector2i(v1, v3));
670 const int v7 = get_vertex(Eigen::Vector2i(v1, v4));
671 const int v9 = get_vertex(Eigen::Vector2i(v4, v2));
672 const int v10 = get_vertex(Eigen::Vector2i(v3, v4));
