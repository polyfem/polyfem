171 void State::compute_mesh_stats()
172 {
173 if (!mesh)
174 {
175 logger().error("Load the mesh first!");
176 return;
177 }
178
179 bases.clear();
180 pressure_bases.clear();
181 geom_bases.clear();
182 boundary_nodes.clear();

187
188 stiffness.resize(0, 0);

190 sol.resize(0, 0);
191 pressure.resize(0, 0);
192
193 n_bases = 0;
194 n_pressure_bases = 0;
195
196 simplex_count = 0;
197 regular_count = 0;
198 regular_boundary_count = 0;
199 simple_singular_count = 0;
200 multi_singular_count = 0;
201 boundary_count = 0;
202 non_regular_boundary_count = 0;
203 non_regular_count = 0;
204 undefined_count = 0;
205 multi_singular_boundary_count = 0;
206
207 const auto &els_tag = mesh->elements_tag();
208
209 mesh->prepare_mesh();
210
211 for (size_t i = 0; i < els_tag.size(); ++i)
212 {
213 const ElementType type = els_tag[i];
214
215 switch (type)
216 {
217 case ElementType::Simplex:
218 simplex_count++;
219 break;
220 case ElementType::RegularInteriorCube:
221 regular_count++;
222 break;
223 case ElementType::RegularBoundaryCube:
224 regular_boundary_count++;
225 break;
226 case ElementType::SimpleSingularInteriorCube:
227 simple_singular_count++;
228 break;
229 case ElementType::MultiSingularInteriorCube:
230 multi_singular_count++;
231 break;
232 case ElementType::SimpleSingularBoundaryCube:
233 boundary_count++;
234 break;
235 case ElementType::InterfaceCube:
236 case ElementType::MultiSingularBoundaryCube:
237 multi_singular_boundary_count++;
238 break;
239 case ElementType::BoundaryPolytope:
240 non_regular_boundary_count++;
241 break;
242 case ElementType::InteriorPolytope:
243 non_regular_count++;
244 break;
245 case ElementType::Undefined:
246 undefined_count++;
247 break;
248 }
249 }
250
251 logger().info("simplex_count: \t{}", simplex_count);
252 logger().info("regular_count: \t{}", regular_count);
253 logger().info("regular_boundary_count: \t{}", regular_boundary_count);
254 logger().info("simple_singular_count: \t{}", simple_singular_count);
255 logger().info("multi_singular_count: \t{}", multi_singular_count);
256 logger().info("boundary_count: \t{}", boundary_count);
257 logger().info("multi_singular_boundary_count: \t{}", multi_singular_boundary_count);
258 logger().info("non_regular_count: \t{}", non_regular_count);
259 logger().info("non_regular_boundary_count: \t{}", non_regular_boundary_count);
260 logger().info("undefined_count: \t{}", undefined_count);
261 logger().info("total count:\t {}", mesh->n_elements());
262 }
