109 void AssemblerUtils::merge_mixed_matrices(
110 const int n_bases, const int n_pressure_bases, const int problem_dim, const bool add_average,
111 const StiffnessMatrix &velocity_stiffness, const StiffnessMatrix &mixed_stiffness, const StiffnessMatrix &pressure_stiffness,
112 StiffnessMatrix &stiffness)
113 {
114 assert(velocity_stiffness.rows() == velocity_stiffness.cols());
115 assert(velocity_stiffness.rows() == n_bases * problem_dim);
116
117 assert(mixed_stiffness.size() == 0 || mixed_stiffness.rows() == n_bases * problem_dim);
118 assert(mixed_stiffness.size() == 0 || mixed_stiffness.cols() == n_pressure_bases);
119
120 assert(pressure_stiffness.size() == 0 || pressure_stiffness.rows() == n_pressure_bases);
121 assert(pressure_stiffness.size() == 0 || pressure_stiffness.cols() == n_pressure_bases);
122
123 const int avg_offset = add_average ? 1 : 0;
124
125 std::vector<Eigen::Triplet<double>> blocks;
126 blocks.reserve(velocity_stiffness.nonZeros() + 2 * mixed_stiffness.nonZeros() + pressure_stiffness.nonZeros() + 2 * avg_offset * velocity_stiffness.rows());
127
128 for (int k = 0; k < velocity_stiffness.outerSize(); ++k)
129 {
130 for (StiffnessMatrix::InnerIterator it(velocity_stiffness, k); it; ++it)
131 {
132 blocks.emplace_back(it.row(), it.col(), it.value());
133 }
134 }
135
136 for (int k = 0; k < mixed_stiffness.outerSize(); ++k)
137 {
138 for (StiffnessMatrix::InnerIterator it(mixed_stiffness, k); it; ++it)
139 {
140 blocks.emplace_back(it.row(), n_bases * problem_dim + it.col(), it.value());
141 blocks.emplace_back(it.col() + n_bases * problem_dim, it.row(), it.value());
142 }
143 }
144
145 for (int k = 0; k < pressure_stiffness.outerSize(); ++k)
146 {
147 for (StiffnessMatrix::InnerIterator it(pressure_stiffness, k); it; ++it)
148 {
149 blocks.emplace_back(n_bases * problem_dim + it.row(), n_bases * problem_dim + it.col(), it.value());
150 }
151 }
152
153 if (add_average)
154 {
155 const double val = 1.0 / n_pressure_bases;
156 for (int i = 0; i < n_pressure_bases; ++i)
157 {
158 blocks.emplace_back(n_bases * problem_dim + i, n_bases * problem_dim + n_pressure_bases, val);
159 blocks.emplace_back(n_bases * problem_dim + n_pressure_bases, n_bases * problem_dim + i, val);
160 }
161 }
162
163 stiffness.resize(n_bases * problem_dim + n_pressure_bases + avg_offset, n_bases * problem_dim + n_pressure_bases + avg_offset);
164 stiffness.setFromTriplets(blocks.begin(), blocks.end());
165 stiffness.makeCompressed();

170 // Eigen::saveMarket(mixed_stiffness, "mixed_stiffness.txt");
171 // Eigen::saveMarket(pressure_stiffness, "pressure_stiffness.txt");
172 }

174 int AssemblerUtils::quadrature_order(const std::string &assembler, const int basis_degree, const BasisType &b_type, const int dim)
175 {
176 if (assembler == "Mass")
177 {
178 if (b_type == BasisType::SIMPLEX_LAGRANGE || b_type == BasisType::CUBE_LAGRANGE)
179 return std::max(basis_degree * 2, 1);
180 else
181 return basis_degree * 2 + 1;
182 }
183 else if (assembler == "NavierStokes")
184 {
185 if (b_type == BasisType::SIMPLEX_LAGRANGE)
186 return std::max((basis_degree - 1) + basis_degree, 1);
187 else if (b_type == BasisType::CUBE_LAGRANGE)
188 return std::max(basis_degree * 2, 1);
189 else
190 return basis_degree * 2 + 1;
191 }
192 else
193 {
194 if (b_type == BasisType::SIMPLEX_LAGRANGE)
195 return std::max((basis_degree - 1) * 2, 1);
196 else if (b_type == BasisType::CUBE_LAGRANGE)
197 return std::max(basis_degree * 2, 1);
198 else
199 return (basis_degree - 1) * 2 + 1;
200 }
201 }
