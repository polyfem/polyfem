38 Eigen::Matrix3d to_rotation_matrix(const json &jr, std::string mode)
39 {
40 std::transform(mode.begin(), mode.end(), mode.begin(), ::tolower);
41
42 Eigen::VectorXd r;
43 if (jr.is_number())
44 {
45 r.setZero(3);
46 assert(mode.size() == 1); // must be either "x", "y", or "z"
47 int i = mode[0] - 'x';
48 assert(i >= 0 && i < 3);
49 r[i] = jr.get<double>();
50 }
51 else
52 {
53 assert(jr.is_array());
54 r = jr;
55 }
56
57 if (mode == "axis_angle")
58 {
59 assert(r.size() == 4);
60 double angle = deg2rad(r[0]); // NOTE: assumes input angle is in degrees
61 Eigen::Vector3d axis = r.tail<3>().normalized();
62 return Eigen::AngleAxisd(angle, axis).toRotationMatrix();
63 }
64
65 if (mode == "quaternion")
66 {
67 assert(r.size() == 4);
68 Eigen::Vector4d q = r.normalized();
69 return Eigen::Quaterniond(q).toRotationMatrix();
70 }
71
72 // The following expect the input is given in degrees
73 r = deg2rad(r);
74
75 if (mode == "rotation_vector")
76 {
77 assert(r.size() == 3);
78 double angle = r.norm();
79 if (angle != 0)
80 {
81 return Eigen::AngleAxisd(angle, r / angle).toRotationMatrix();
82 }
83 else
84 {
85 return Eigen::Matrix3d::Identity();
86 }
87 }
88
89 Eigen::Matrix3d R = Eigen::Matrix3d::Identity();
90
91 for (int i = 0; i < mode.size(); i++)
92 {
93 int j = mode[i] - 'x';
94 assert(j >= 0 && j < 3);
95 Eigen::Vector3d axis = Eigen::Vector3d::Zero();
96 axis[j] = 1;
97 R = Eigen::AngleAxisd(r[j], axis).toRotationMatrix() * R;
98 }
