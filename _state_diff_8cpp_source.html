42 void replace_rows_by_identity(StiffnessMatrix &reduced_mat, const StiffnessMatrix &mat, const std::vector<int> &rows)
43 {
44 reduced_mat.resize(mat.rows(), mat.cols());
45
46 std::vector<bool> mask(mat.rows(), false);
47 for (int i : rows)
48 mask[i] = true;
49
50 std::vector<Eigen::Triplet<double>> coeffs;
51 for (int k = 0; k < mat.outerSize(); ++k)
52 {
53 for (StiffnessMatrix::InnerIterator it(mat, k); it; ++it)
54 {
55 if (mask[it.row()])
56 {
57 if (it.row() == it.col())
58 coeffs.emplace_back(it.row(), it.col(), 1.0);
59 }
60 else
61 coeffs.emplace_back(it.row(), it.col(), it.value());
62 }
63 }
64 reduced_mat.setFromTriplets(coeffs.begin(), coeffs.end());
65 }

320 {

322

324 adjoint.setZero(ndof(), adjoint_rhs.cols());

386 Eigen::MatrixXd State::solve_transient_adjoint(const Eigen::MatrixXd &adjoint_rhs) const
387 {
388 const double dt = args["time"]["dt"];
389 const int time_steps = args["time"]["time_steps"];
390
391 int bdf_order = 1;
392 if (args["time"]["integrator"].is_string())
393 bdf_order = 1;
394 else if (args["time"]["integrator"]["type"] == "ImplicitEuler")
395 bdf_order = 1;
396 else if (args["time"]["integrator"]["type"] == "BDF")
397 bdf_order = args["time"]["integrator"]["steps"].get<int>();
398 else
399 log_and_throw_adjoint_error("Integrator type not supported for differentiability.");
400
401 assert(adjoint_rhs.cols() == time_steps + 1);
402
403 const int cols_per_adjoint = time_steps + 1;
404 Eigen::MatrixXd adjoints;
405 adjoints.setZero(ndof(), cols_per_adjoint * 2);
406
407 // set dirichlet rows of mass to identity
408 StiffnessMatrix reduced_mass;
409 replace_rows_by_identity(reduced_mass, mass, boundary_nodes);
410
411 Eigen::MatrixXd sum_alpha_p, sum_alpha_nu;
412 for (int i = time_steps; i >= 0; --i)
413 {
414 {
415 sum_alpha_p.setZero(ndof(), 1);
416 sum_alpha_nu.setZero(ndof(), 1);
417
418 const int num = std::min(bdf_order, time_steps - i);
419
420 Eigen::VectorXd bdf_coeffs(num);
421 for (int j = 0; j < bdf_order && i + j < time_steps; ++j)
422 bdf_coeffs(j) = -time_integrator::BDF::alphas(std::min(bdf_order - 1, i + j))[j];
423
424 sum_alpha_p = adjoints.middleCols(i + 1, num) * bdf_coeffs;
425 sum_alpha_nu = adjoints.middleCols(cols_per_adjoint + i + 1, num) * bdf_coeffs;
426 }
427
428 Eigen::VectorXd rhs_ = -reduced_mass.transpose() * sum_alpha_nu - adjoint_rhs.col(i);
429 for (int j = 1; j <= bdf_order; j++)
430 {
431 if (i + j > time_steps)
432 break;
433
434 StiffnessMatrix gradu_h_prev;
435 compute_force_jacobian_prev(i + j, i, gradu_h_prev);
436 Eigen::VectorXd tmp = adjoints.col(i + j) * (time_integrator::BDF::betas(diff_cached.bdf_order(i + j) - 1) * dt);
437 tmp(boundary_nodes).setZero();
438 rhs_ += -gradu_h_prev.transpose() * tmp;
439 }
440
441 if (i > 0)
442 {
443 double beta_dt = time_integrator::BDF::betas(diff_cached.bdf_order(i) - 1) * dt;
444
445 rhs_ += (1. / beta_dt) * (diff_cached.gradu_h(i) - reduced_mass).transpose() * sum_alpha_p;
446
447 {
448 StiffnessMatrix A = diff_cached.gradu_h(i).transpose();
449 Eigen::VectorXd b_ = rhs_;
450 b_(boundary_nodes).setZero();
451
452 auto solver = polysolve::linear::Solver::create(args["solver"]["adjoint_linear"], adjoint_logger());
453
454 Eigen::VectorXd x;
455 dirichlet_solve(*solver, A, b_, boundary_nodes, x, A.rows(), "", false, false, false);
456 adjoints.col(i + cols_per_adjoint) = x;
457 }
458
459 // TODO: generalize to BDFn
460 Eigen::VectorXd tmp = rhs_(boundary_nodes);
461 if (i + 1 < cols_per_adjoint)
462 tmp += -2. / beta_dt * adjoints(boundary_nodes, i + 1);
463 if (i + 2 < cols_per_adjoint)
464 tmp += (1. / beta_dt) * adjoints(boundary_nodes, i + 2);
465
466 tmp -= (diff_cached.gradu_h(i).transpose() * adjoints.col(i + cols_per_adjoint))(boundary_nodes);
467 adjoints(boundary_nodes, i + cols_per_adjoint) = tmp;
468 adjoints.col(i) = beta_dt * adjoints.col(i + cols_per_adjoint) - sum_alpha_p;
469 }
470 else
471 {
472 adjoints.col(i) = -reduced_mass.transpose() * sum_alpha_p;
473 adjoints.col(i + cols_per_adjoint) = rhs_; // adjoint_nu[0] actually stores adjoint_mu[0]
474 }
475 }
476 return adjoints;
477 }
