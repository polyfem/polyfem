94 void Remesher::project_quantities()
95 {
96 POLYFEM_REMESHER_SCOPED_TIMER("Project quantities");
97
98 using namespace polyfem::assembler;
99 using namespace polyfem::basis;
100 using namespace polyfem::utils;
101
102 const std::unique_ptr<Mesh> from_mesh = Mesh::create(
103 global_projection_cache.rest_positions, global_projection_cache.elements);
104 std::vector<ElementBases> from_bases;
105 std::vector<LocalBoundary> _;
106 Eigen::VectorXi from_vertex_to_basis;
107 int n_from_basis = build_bases(*from_mesh, state.formulation(), from_bases, _, from_vertex_to_basis);
108 assert(from_vertex_to_basis.size() == global_projection_cache.rest_positions.rows());
109
110 // Old values of independent variables
111 Eigen::MatrixXd from_projection_quantities = reorder_matrix(
112 global_projection_cache.projection_quantities,
113 from_vertex_to_basis, n_from_basis, dim());
114 append_rows(from_projection_quantities, obstacle_quantities());
115 n_from_basis += obstacle().n_vertices();
116 assert(dim() * n_from_basis == from_projection_quantities.rows());
117
118 // --------------------------------------------------------------------
119
120 Eigen::MatrixXd rest_positions = this->rest_positions();
121 Eigen::MatrixXi elements = this->elements();
122
123 const std::unique_ptr<Mesh> to_mesh = Mesh::create(rest_positions, elements);
124 std::vector<ElementBases> to_bases;
125 Eigen::VectorXi to_vertex_to_basis;
126 int n_to_basis = build_bases(*to_mesh, state.formulation(), to_bases, _, to_vertex_to_basis);
127 assert(to_vertex_to_basis.size() == rest_positions.rows());
128
129 rest_positions = reorder_matrix(rest_positions, to_vertex_to_basis, n_to_basis);
130 elements = map_index_matrix(elements, to_vertex_to_basis);
131
132 // Interpolated values of independent variables
133 Eigen::MatrixXd to_projection_quantities = reorder_matrix(
134 projection_quantities(), to_vertex_to_basis, n_to_basis, dim());
135 append_rows(to_projection_quantities, obstacle_quantities());
136 n_to_basis += obstacle().n_vertices();
137 assert(dim() * n_to_basis == to_projection_quantities.rows());
138
139 // --------------------------------------------------------------------
140
141 // solve M x = A y for x where M is the mass matrix and A is the cross mass matrix.
142 Eigen::SparseMatrix<double> M, A;
143 {
144 MassMatrixAssembler assembler;
145 Density no_density; // Density of one (i.e., no scaling of mass matrix)
146 AssemblyValsCache cache;
147
148 assembler.assemble(
149 is_volume(), dim(),
150 n_to_basis, no_density, to_bases, to_bases,
151 cache, M);
152 assert(M.rows() == to_projection_quantities.rows());
153
154 // if (lump_mass_matrix)
155 // M = lump_matrix(M);
156
157 assembler.assemble_cross(
158 is_volume(), dim(),
159 n_from_basis, from_bases, from_bases,
160 n_to_basis, to_bases, to_bases,
161 cache, A);
162 assert(A.rows() == to_projection_quantities.rows());
163 assert(A.cols() == from_projection_quantities.rows());
164 }
165
166 // --------------------------------------------------------------------
167
168 ipc::CollisionMesh collision_mesh;
169 {
170 Eigen::MatrixXi boundary_edges, boundary_faces;
171 if (dim() == 2)
172 {
173 igl::boundary_facets(elements, boundary_edges);
174 }
175 else
176 {
177 igl::boundary_facets(elements, boundary_faces);
178 igl::edges(boundary_faces, boundary_edges);
179 }
180
181 if (obstacle().n_edges() > 0)
182 utils::append_rows(boundary_edges, obstacle().e().array() + rest_positions.rows());
183 if (obstacle().n_faces() > 0)
184 utils::append_rows(boundary_faces, obstacle().f().array() + rest_positions.rows());
185
186 utils::append_rows(rest_positions, obstacle().v());
187 assert(rest_positions.size() == to_projection_quantities.rows());
188
189 collision_mesh = ipc::CollisionMesh::build_from_full_mesh(
190 rest_positions, boundary_edges, boundary_faces);
191 }
192
193 // --------------------------------------------------------------------
194
195 Eigen::MatrixXd projected_quantities(to_projection_quantities.rows(), n_quantities());
196 const int n_constrained_quantaties = n_quantities() / 3;
197 const int n_unconstrained_quantaties = n_quantities() - n_constrained_quantaties;
198
199 const std::vector<int> boundary_nodes = this->boundary_nodes(to_vertex_to_basis);
200 for (int i = 0; i < n_constrained_quantaties; ++i)
201 {
202 projected_quantities.col(i) = constrained_L2_projection(
203 state.make_nl_solver<polyfem::solver::NLProblem>(),
204 // L2 projection form
205 M, A, /*y=*/from_projection_quantities.col(i),
206 // Inversion-free form
207 rest_positions, elements, dim(),
208 // Contact form
209 collision_mesh, state.args["contact"]["dhat"],
210 state.solve_data.contact_form
211 ? state.solve_data.contact_form->barrier_stiffness()
212 : 1.0,
213 state.args["contact"]["use_convergent_formulation"],
214 state.args["solver"]["contact"]["CCD"]["broad_phase"],
215 state.args["solver"]["contact"]["CCD"]["tolerance"],
216 state.args["solver"]["contact"]["CCD"]["max_iterations"],
217 // Augmented lagrangian form
218 boundary_nodes, obstacle().ndof(), to_projection_quantities.col(i),
219 // Initial guess
220 to_projection_quantities.col(i));
221 }
222
223 // NOTE: no need for to_projection_quantities.rightCols(n_unconstrained_quantaties)
224 projected_quantities.rightCols(n_unconstrained_quantaties) = unconstrained_L2_projection(
225 M, A, from_projection_quantities.rightCols(n_unconstrained_quantaties));
226
227 // --------------------------------------------------------------------
228
229 assert(projected_quantities.rows() == dim() * n_to_basis);
230 set_projection_quantities(unreorder_matrix(
231 projected_quantities, to_vertex_to_basis, to_vertex_to_basis.size(), dim()));
232 }

294 void Remesher::write_mesh(const std::string &path) const
295 {
296 assert(utils::StringUtils::endswith(path, ".vtu"));
297
298 paraviewo::VTUWriter writer;
299
300 Eigen::MatrixXd rest_positions = this->rest_positions();
301 Eigen::MatrixXd displacements = this->displacements();
302
303 const Eigen::MatrixXd projection_quantities = this->projection_quantities();
304 std::vector<Eigen::MatrixXd> unflattened_projection_quantities;
305 for (const auto &q : projection_quantities.colwise())
306 unflattened_projection_quantities.push_back(utils::unflatten(q, dim()));
307
308 Eigen::MatrixXi elements = this->elements();
309 std::vector<std::vector<int>> all_elements(elements.rows(), std::vector<int>(elements.cols()));
310 for (int i = 0; i < elements.rows(); i++)
311 for (int j = 0; j < elements.cols(); j++)
312 all_elements[i][j] = elements(i, j);
313
314 const int offset = rest_positions.rows();
315 if (obstacle().n_vertices() > 0)
316 {
317 utils::append_rows(rest_positions, obstacle().v());
318 utils::append_rows(displacements, obstacle_displacements());
319 for (int i = 0; i < unflattened_projection_quantities.size(); ++i)
320 utils::append_rows(
321 unflattened_projection_quantities[i],
322 utils::unflatten(obstacle_quantities().col(i), dim()));
323
324 if (obstacle().n_edges() > 0)
325 {
326 const Eigen::MatrixXi obstacle_edges = obstacle().e().array() + offset;
327 all_elements.resize(all_elements.size() + obstacle_edges.rows());
328 for (int i = 0; i < obstacle().n_edges(); i++)
329 {
330 all_elements[i + elements.rows()] = std::vector<int>(obstacle_edges.cols());
331 for (int j = 0; j < obstacle_edges.cols(); j++)
332 all_elements[i + elements.rows()][j] = obstacle_edges(i, j);
333 }
334 }
335
336 if (obstacle().n_faces() > 0)
337 {
338 const Eigen::MatrixXi obstacle_faces = obstacle().f().array() + offset;
339 all_elements.resize(all_elements.size() + obstacle_faces.rows());
340 for (int i = 0; i < obstacle().n_edges(); i++)
341 {
342 all_elements[i + elements.rows()] = std::vector<int>(obstacle_faces.cols());
343 for (int j = 0; j < obstacle_faces.cols(); j++)
344 all_elements[i + elements.rows()][j] = obstacle_faces(i, j);
345 }
346 }
347 }
348
349 for (int i = 0; i < unflattened_projection_quantities.size(); ++i)
350 writer.add_field(fmt::format("projection_quantity_{:d}", i), unflattened_projection_quantities[i]);
351 writer.add_field("displacement", displacements);
352 writer.write_mesh(path, rest_positions, all_elements, /*is_simplicial=*/true, /*has_poly=*/false);
353 }

355 Eigen::MatrixXd Remesher::combine_time_integrator_quantities(
356 const std::shared_ptr<time_integrator::ImplicitTimeIntegrator> &time_integrator)
357 {
358 if (time_integrator == nullptr)
359 return Eigen::MatrixXd();
360
361 // not including current displacement as this will be handled as positions
362 Eigen::MatrixXd projection_quantities(
363 time_integrator->x_prev().size(), 3 * time_integrator->steps());
364 int i = 0;
365 for (const Eigen::VectorXd &x : time_integrator->x_prevs())
366 projection_quantities.col(i++) = x;
367 for (const Eigen::VectorXd &v : time_integrator->v_prevs())
368 projection_quantities.col(i++) = v;
369 for (const Eigen::VectorXd &a : time_integrator->a_prevs())
370 projection_quantities.col(i++) = a;
371 assert(i == projection_quantities.cols());
372
373 return projection_quantities;
374 }
