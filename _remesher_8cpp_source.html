124 std::vector<ElementBases> to_bases;
125 Eigen::VectorXi to_vertex_to_basis;

127 assert(to_vertex_to_basis.size() == rest_positions.rows());

129 rest_positions = reorder_matrix(rest_positions, to_vertex_to_basis, n_to_basis);

132 // Interpolated values of independent variables

134 projection_quantities(), to_vertex_to_basis, n_to_basis, dim());

141 // solve M x = A y for x where M is the mass matrix and A is the cross mass matrix.

144 MassMatrixAssembler assembler;
145 assembler::Mass mass_matrix_assembler;
146 AssemblyValsCache cache;

148 mass_matrix_assembler.assemble(

151 cache, 0, M, true);

154 // if (lump_mass_matrix)
155 // M = lump_matrix(M);

157 assembler.assemble_cross(
158 is_volume(), dim(),
159 n_from_basis, from_bases, from_bases,
160 n_to_basis, to_bases, to_bases,
161 cache, A);
162 assert(A.rows() == to_projection_quantities.rows());
163 assert(A.cols() == from_projection_quantities.rows());
164 }
165
166 // --------------------------------------------------------------------
167
168 ipc::CollisionMesh collision_mesh;
169 {
170 Eigen::MatrixXi boundary_edges, boundary_faces;
171 if (dim() == 2)
172 {
173 igl::boundary_facets(elements, boundary_edges);
174 }
175 else
176 {
177 igl::boundary_facets(elements, boundary_faces);
178 igl::edges(boundary_faces, boundary_edges);
179 }
180
181 if (obstacle().n_edges() > 0)
182 utils::append_rows(boundary_edges, obstacle().e().array() + rest_positions.rows());
183 if (obstacle().n_faces() > 0)
184 utils::append_rows(boundary_faces, obstacle().f().array() + rest_positions.rows());
185
186 utils::append_rows(rest_positions, obstacle().v());
187 assert(rest_positions.size() == to_projection_quantities.rows());

189 collision_mesh = ipc::CollisionMesh::build_from_full_mesh(
190 rest_positions, boundary_edges, boundary_faces);
191 }

193 // --------------------------------------------------------------------
194
195 Eigen::MatrixXd projected_quantities(to_projection_quantities.rows(), n_quantities());
196 const int n_constrained_quantaties = n_quantities() / 3;
197 const int n_unconstrained_quantaties = n_quantities() - n_constrained_quantaties;
198
199 const std::vector<int> boundary_nodes = this->boundary_nodes(to_vertex_to_basis);
200 for (int i = 0; i < n_constrained_quantaties; ++i)
201 {
202 projected_quantities.col(i) = constrained_L2_projection(
203 state.make_nl_solver(/*for_al=*/false),
204 // L2 projection form
205 M, A, /*y=*/from_projection_quantities.col(i),
206 // Inversion-free form
207 rest_positions, elements, dim(),
208 // Contact form
209 collision_mesh, state.args["contact"]["dhat"],
210 state.solve_data.contact_form
211 ? state.solve_data.contact_form->barrier_stiffness()
212 : 1.0,
213 state.args["contact"]["use_convergent_formulation"] ? bool(state.args["contact"]["use_area_weighting"]) : false,
214 state.args["contact"]["use_convergent_formulation"] ? bool(state.args["contact"]["use_improved_max_operator"]) : false,
215 state.args["contact"]["use_convergent_formulation"] ? bool(state.args["contact"]["use_physical_barrier"]) : false,
216 state.args["solver"]["contact"]["CCD"]["broad_phase"],
217 state.args["solver"]["contact"]["CCD"]["tolerance"],
218 state.args["solver"]["contact"]["CCD"]["max_iterations"],
219 // Augmented lagrangian form
220 boundary_nodes, obstacle().ndof(), to_projection_quantities.col(i),
221 // Initial guess
222 to_projection_quantities.col(i));
223 }
224
225 // Set entry for obstacle to identity
226 // ┌ ┐ ┌ ┐ ┌ ┐
227 // │M 0│ │x_fem│ |y_fem|
228 // │ │ │ │ = | | ⟹ x_obs = y_obs
229 // │0 I│ │x_obs│ |y_obs|
230 // └ ┘ └ ┘ └ ┘
231
232 for (int i = 0; i < dim() * obstacle().n_vertices(); ++i)
233 {
234 const int row = M.rows() - i - 1; // fill from bottom
235 M.coeffRef(row, row) = 1.0;
236 }
237
238 for (int i = 0; i < dim() * obstacle().n_vertices(); ++i)
239 {
240 const int row = A.rows() - i - 1; // fill from bottom
241 const int col = A.cols() - i - 1; // fill from bottom
242 A.coeffRef(row, col) = 1.0;
243 }
244
245 // NOTE: no need for to_projection_quantities.rightCols(n_unconstrained_quantaties)
246 projected_quantities.rightCols(n_unconstrained_quantaties) = unconstrained_L2_projection(
247 M, A, from_projection_quantities.rightCols(n_unconstrained_quantaties));
248
249 // --------------------------------------------------------------------
250
251 assert(projected_quantities.rows() == dim() * n_to_basis);
252 set_projection_quantities(unreorder_matrix(
253 projected_quantities, to_vertex_to_basis, to_vertex_to_basis.size(), dim()));

316 void Remesher::write_mesh(const std::string &path) const
317 {
318 assert(utils::StringUtils::endswith(path, ".vtu"));
319
320 paraviewo::VTUWriter writer;
321
322 Eigen::MatrixXd rest_positions = this->rest_positions();
323 Eigen::MatrixXd displacements = this->displacements();
324
325 const Eigen::MatrixXd projection_quantities = this->projection_quantities();
326 std::vector<Eigen::MatrixXd> unflattened_projection_quantities;
327 for (const auto &q : projection_quantities.colwise())
328 unflattened_projection_quantities.push_back(utils::unflatten(q, dim()));
329
330 Eigen::MatrixXi elements = this->elements();
331 std::vector<std::vector<int>> all_elements(elements.rows(), std::vector<int>(elements.cols()));
332 for (int i = 0; i < elements.rows(); i++)
333 for (int j = 0; j < elements.cols(); j++)
334 all_elements[i][j] = elements(i, j);
335
336 const int offset = rest_positions.rows();
337 if (obstacle().n_vertices() > 0)
338 {
339 utils::append_rows(rest_positions, obstacle().v());
340 utils::append_rows(displacements, obstacle_displacements());
341 for (int i = 0; i < unflattened_projection_quantities.size(); ++i)
342 utils::append_rows(
343 unflattened_projection_quantities[i],
344 utils::unflatten(obstacle_quantities().col(i), dim()));
345
346 if (obstacle().n_edges() > 0)
347 {
348 const Eigen::MatrixXi obstacle_edges = obstacle().e().array() + offset;
349 all_elements.resize(all_elements.size() + obstacle_edges.rows());
350 for (int i = 0; i < obstacle().n_edges(); i++)
351 {
352 all_elements[i + elements.rows()] = std::vector<int>(obstacle_edges.cols());
353 for (int j = 0; j < obstacle_edges.cols(); j++)
354 all_elements[i + elements.rows()][j] = obstacle_edges(i, j);
355 }
356 }
357
358 if (obstacle().n_faces() > 0)
359 {
360 const Eigen::MatrixXi obstacle_faces = obstacle().f().array() + offset;
361 all_elements.resize(all_elements.size() + obstacle_faces.rows());
362 for (int i = 0; i < obstacle().n_edges(); i++)
363 {
364 all_elements[i + elements.rows()] = std::vector<int>(obstacle_faces.cols());
365 for (int j = 0; j < obstacle_faces.cols(); j++)
366 all_elements[i + elements.rows()][j] = obstacle_faces(i, j);
367 }
368 }
369 }
370
371 for (int i = 0; i < unflattened_projection_quantities.size(); ++i)
372 writer.add_field(fmt::format("projection_quantity_{:d}", i), unflattened_projection_quantities[i]);
373 writer.add_field("displacement", displacements);
374 writer.write_mesh(path, rest_positions, all_elements, /*is_simplicial=*/true, /*has_poly=*/false);
375 }

377 Eigen::MatrixXd Remesher::combine_time_integrator_quantities(
378 const std::shared_ptr<time_integrator::ImplicitTimeIntegrator> &time_integrator)
379 {
380 if (time_integrator == nullptr)
381 return Eigen::MatrixXd();
382
383 // not including current displacement as this will be handled as positions
384 Eigen::MatrixXd projection_quantities(
385 time_integrator->x_prev().size(), 3 * time_integrator->steps());
386 int i = 0;
387 for (const Eigen::VectorXd &x : time_integrator->x_prevs())
388 projection_quantities.col(i++) = x;
389 for (const Eigen::VectorXd &v : time_integrator->v_prevs())
390 projection_quantities.col(i++) = v;
391 for (const Eigen::VectorXd &a : time_integrator->a_prevs())
392 projection_quantities.col(i++) = a;
393 assert(i == projection_quantities.cols());
394
395 return projection_quantities;
396 }
