111void RBFWithLinear::bases_grads(const int axis, const Eigen::MatrixXd &samples, Eigen::MatrixXd &val) const
112{
113 const int num_kernels = centers_.rows();
114 const int dim = centers_.cols();
115
116 // Compute ∇xA
117 Eigen::MatrixXd A_prime(samples.rows(), num_kernels + 1 + dim);
118 A_prime.setZero();
119
120 for (int j = 0; j < num_kernels; ++j)
121 {
122 A_prime.col(j) = (samples.rowwise() - centers_.row(j)).rowwise().norm().unaryExpr([this](double x) { return kernel_prime(is_volume(), x) / x; });
123 A_prime.col(j) = (samples.col(axis).array() - centers_(j, axis)) * A_prime.col(j).array();
124 }
125 // Linear terms
126 A_prime.middleCols(num_kernels + 1 + axis, 1).setOnes();
127
128 // Apply weights
129 val = A_prime * weights_;
130}

134void RBFWithLinear::compute_kernels_matrix(const Eigen::MatrixXd &samples, Eigen::MatrixXd &A) const
135{
136 // Compute A
137 const int num_kernels = centers_.rows();
138 const int dim = centers_.cols();
139
140 A.resize(samples.rows(), num_kernels + 1 + dim);
141 for (int j = 0; j < num_kernels; ++j)
142 {
143 A.col(j) = (samples.rowwise() - centers_.row(j)).rowwise().norm().unaryExpr([this](double x) { return kernel(is_volume(), x); });
144 }
145 A.col(num_kernels).setOnes(); // constant term
146 A.rightCols(dim) = samples; // linear terms
147}

151void RBFWithLinear::compute_constraints_matrix(
152 const int num_bases,
153 const Quadrature &quadr,
154 const Eigen::MatrixXd &local_basis_integral,
155 Eigen::MatrixXd &L,
156 Eigen::MatrixXd &t) const
157{
158 const int num_kernels = centers_.rows();
159 const int dim = centers_.cols();
160
161 // Compute KI
162 Eigen::MatrixXd KI(num_kernels, dim);
163 for (int j = 0; j < num_kernels; ++j)
164 {
165 // ∫∇x(φj)(p) = Σ_q (xq - xk) * 1/r * h'(r) * wq
166 // - xq is the x coordinate of the q-th quadrature point
167 // - wq is the q-th quadrature weight
168 // - r is the distance from pq to the kernel center
169 // - h is the RBFWithLinear RBF kernel (scalar function)
170 const Eigen::MatrixXd drdp = quadr.points.rowwise() - centers_.row(j);
171 const Eigen::VectorXd r = drdp.rowwise().norm();
172 KI.row(j) = (drdp.array().colwise() * (quadr.weights.array() * r.unaryExpr([this](double x) { return kernel_prime(is_volume(), x); }).array() / r.array())).colwise().sum();
173 }
174 KI /= quadr.weights.sum();
175
176 // Compute L
177 L.resize(num_kernels + dim + 1, num_kernels + 1);
178 L.setZero();
179 L.diagonal().setOnes();
180 L.block(num_kernels + 1, 0, dim, num_kernels) = -KI.transpose();
181
182 // Compute t
183 t.resize(num_kernels + 1 + dim, num_bases);
184 t.setZero();
185 t.bottomRows(dim) = local_basis_integral.transpose().topRows(dim) / quadr.weights.sum();
186}

190void RBFWithLinear::compute_weights(const Eigen::MatrixXd &samples,
191 const Eigen::MatrixXd &local_basis_integral, const Quadrature &quadr,
192 Eigen::MatrixXd &rhs, bool with_constraints)
193{
194 logger().trace("#kernel centers: {}", centers_.rows());
195 logger().trace("#collocation points: {}", samples.rows());
196 logger().trace("#quadrature points: {}", quadr.weights.size());
197 logger().trace("#non-vanishing bases: {}", rhs.cols());
198
199 if (!with_constraints)
200 {
201 // Compute A
202 Eigen::MatrixXd A;
203 compute_kernels_matrix(samples, A);
204
205 // Solve the system
206 const int num_kernels = centers_.rows();
207 logger().trace("-- Solving system of size {}x{}", num_kernels, num_kernels);
208 weights_ = (A.transpose() * A).ldlt().solve(A.transpose() * rhs);
209 logger().trace("-- Solved!");
210
211 return;
212 }
213
214 // For each basis function f that is nonzero on the element E, we want to
215 // solve the least square system A w = rhs, where:
216 // ┏ ┓
217 // ┃ φj(pi) ... 1 xi yi ┃
218 // A = ┃ ┊ ┊ ┊ ┊ ┃ ∊ ℝ^{#S x (#K+1+dim)}
219 // ┃ ┊ ┊ ┊ ┊ ┃
220 // ┗ ┛
221 // ┏ ┓^⊤
222 // w = ┃ wj ... a00 a10 a01 ┃ ∊ ℝ^{#K+1+dim}
223 // ┗ ┛
224 // - A is the RBF kernels evaluated over the collocation points (#S)
225 // - b is the expected value of the basis sampled on the boundary (#S)
226 // - w is the weight of the kernels defining the basis
227 // - pi = (xi, yi) is the i-th collocation point
228 //
229 // Moreover, we want to impose a constraint on the gradients of the kernels
230 // so that the integral of the gradients over the polytope must be equal to
231 // the value specified in the argument `local_basis_integral` (#K)
232 //
233 // Let `lb` be the precomputed expected value of ∫f over the rest of the mesh.
234 // We write down the constraint as:
235 //
236 // ∫_{p ∊ E} Σ_j wj ∇x(φj)(p) + ∇x(a^⊤·p + c) dp = lb (1)
237 // (1) ⇔ ∫_{p ∊ E} Σ_j wj ∇x(φj)(p) + ax dp = lb
238 // ⇔ lb - Σ_j wj ∫_{p ∊ E} ∇x(φj)(p) dp = ax Vol(E)
239 //
240 // We now have a relationship w = Lv + t, where the weights (and esp. the
241 // linear terms in the weight vector w), are expressed as an affine
242 // combination of unknowns v = [wj ... c] ∊ ℝ^{#K+1} and a translation t
243 //
244 // After solving the new least square system A L v = rhs - A t, we can retrieve
245 // w = L v
246
247 //
248 // ┏ ┓^⊤
249 // t = ┃ 0 ┈ ┈ 0 0 lbx lby ┃ / Vol(E) ∊ ℝ^{#K+1+dim}
250 // ┗ ┛
251 //
252 // ┏ ┓
253 // ┃ 1 ┃
254 // ┃ 1 ┃
255 // ┃ · ┃
256 // L = ┃ · ┃ ∊ ℝ^{ (#K+1+dim) x (#K+1}) }
257 // ┃ 1 ┃
258 // ┃ Lx_j ┈ 0 ┃
259 // ┃ Ly_j ┈ 0 ┃
260 // ┗ ┛
261 // Where Lx_j = -∫∇xφj / Vol(E) = -∫_{p ∊ E} ∇x(φj)(p) / Vol(E) is integrated numerically
262 //
263
264 const int num_bases = rhs.cols();
265
266 // Compute A
267 Eigen::MatrixXd A;
268 compute_kernels_matrix(samples, A);
269
270 // Compute L and t
271 // Note that t is stored into `weights_` for memory efficiency reasons
272 Eigen::MatrixXd L;
273 compute_constraints_matrix(num_bases, quadr, local_basis_integral, L, weights_);
274
275 // Compute b = rhs - A t
276 rhs -= A * weights_;
277
278 // Solve the system
279 logger().trace("-- Solving system of size {}x{}", L.cols(), L.cols());
280 weights_ += L * (L.transpose() * A.transpose() * A * L).ldlt().solve(L.transpose() * A.transpose() * rhs);
281 logger().trace("-- Solved!");
282}
