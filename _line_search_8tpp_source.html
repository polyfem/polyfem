51 template <typename ProblemType>
52 void LineSearch<ProblemType>::save_sampled_values(
53 const std::string &filename,
54 const typename ProblemType::TVector &x,
55 const typename ProblemType::TVector &delta_x,
56 ProblemType &objFunc,
57 const double starting_step_size,
58 const int num_samples)
59 {
60 std::ofstream samples(filename, std::ios::out);
61 if (!samples.is_open())
62 {
63 spdlog::error("Unable to save sampled values to file \"{}\" !", filename);
64 return;
65 }
66
67 samples << "alpha,f(x + alpha * delta_x),valid,decrease\n";
68
69 objFunc.solution_changed(x);
70 double fx = objFunc.value(x);
71
72 Eigen::VectorXd alphas = Eigen::VectorXd::LinSpaced(2 * num_samples - 1, -starting_step_size, starting_step_size);
73 for (int i = 0; i < alphas.size(); i++)
74 {
75 typename ProblemType::TVector new_x = x + alphas[i] * delta_x;
76 objFunc.solution_changed(new_x);
77 double fxi = objFunc.value(new_x);
78 samples << alphas[i] << ","
79 << fxi << ","
80 << (objFunc.is_step_valid(x, new_x) ? "true" : "false") << ","
81 << (fxi < fx ? "true" : "false") << "\n";
82 }
83
84 samples.close();
85 }

87 template <typename ProblemType>
88 double LineSearch<ProblemType>::compute_nan_free_step_size(
89 const TVector &x,
90 const TVector &delta_x,
91 ProblemType &objFunc,
92 const double starting_step_size,
93 const double rate)
94 {
95 double step_size = starting_step_size;
96 TVector new_x = x + step_size * delta_x;
97
98 // Find step that does not result in nan or infinite energy
99 while (step_size > min_step_size && cur_iter < max_step_size_iter)
100 {
101 // Compute the new energy value without contacts
102 const double energy = objFunc.value(new_x, /*only_elastic=*/true);
103 const bool is_step_valid = objFunc.is_step_valid(x, new_x);
104
105 if (!std::isfinite(energy) || !is_step_valid)
106 {
107 step_size *= rate;
108 new_x = x + step_size * delta_x;
109 }
110 else
111 {
112 break;
113 }
114 cur_iter++;
115 }
116
117 if (cur_iter >= max_step_size_iter || step_size <= min_step_size)
118 {
119 logger().error(
120 "Line search failed to find a valid finite energy step (cur_iter={:d} step_size={:g})!",
121 cur_iter, step_size);
122 return std::nan("");
123 }
124
125 return step_size;
126 }
