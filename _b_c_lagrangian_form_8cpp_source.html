44 void BCLagrangianForm::init_masked_lumped_mass(
45 const int ndof,
46 const StiffnessMatrix &mass,
47 const size_t obstacle_ndof)
48 {
49 std::vector<bool> is_boundary_dof(ndof, true);
50 for (const auto bn : boundary_nodes_)
51 is_boundary_dof[bn] = false;
52
53 masked_lumped_mass_sqrt_ = mass.size() == 0 ? polyfem::utils::sparse_identity(ndof, ndof) : polyfem::utils::lump_matrix(mass);
54 assert(ndof == masked_lumped_mass_sqrt_.rows() && ndof == masked_lumped_mass_sqrt_.cols());
55
56 // Give the collision obstacles a entry in the lumped mass matrix
57 if (obstacle_ndof != 0)
58 {
59 const int n_fe_dof = ndof - obstacle_ndof;
60 const double avg_mass = masked_lumped_mass_sqrt_.diagonal().head(n_fe_dof).mean();
61 for (int i = n_fe_dof; i < ndof; ++i)
62 {
63 masked_lumped_mass_sqrt_.coeffRef(i, i) = avg_mass;
64 }
65 }
66
67 // Remove non-boundary ndof from mass matrix
68 masked_lumped_mass_sqrt_.prune([&](const int &row, const int &col, const double &value) -> bool {
69 assert(row == col); // matrix should be diagonal
70 return !is_boundary_dof[row];
71 });
72
73 std::vector<Eigen::Triplet<double>> tmp_triplets;
74 tmp_triplets.reserve(masked_lumped_mass_sqrt_.nonZeros());
75 for (int k = 0; k < masked_lumped_mass_sqrt_.outerSize(); ++k)
76 {
77 for (StiffnessMatrix::InnerIterator it(masked_lumped_mass_sqrt_, k); it; ++it)
78 {
79 assert(it.col() == k);
80 tmp_triplets.emplace_back(it.row(), it.col(), sqrt(it.value()));
81 }
82 }
83
84 masked_lumped_mass_sqrt_.setFromTriplets(tmp_triplets.begin(), tmp_triplets.end());
85 masked_lumped_mass_sqrt_.makeCompressed();
86
87 lagr_mults_.resize(ndof);
88 lagr_mults_.setZero();
89 }
