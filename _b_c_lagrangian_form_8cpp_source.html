48 void BCLagrangianForm::init_masked_lumped_mass(
49 const int ndof,
50 const StiffnessMatrix &mass,
51 const size_t obstacle_ndof)
52 {
53 std::vector<bool> is_boundary_dof(ndof, true);
54 for (const auto bn : boundary_nodes_)
55 is_boundary_dof[bn] = false;
56
57 masked_lumped_mass_sqrt_ = mass.size() == 0 ? polyfem::utils::sparse_identity(ndof, ndof) : polyfem::utils::lump_matrix(mass);
58 assert(ndof == masked_lumped_mass_sqrt_.rows() && ndof == masked_lumped_mass_sqrt_.cols());
59
60 // Give the collision obstacles a entry in the lumped mass matrix
61 if (obstacle_ndof != 0)
62 {
63 const int n_fe_dof = ndof - obstacle_ndof;
64 const double avg_mass = masked_lumped_mass_sqrt_.diagonal().head(n_fe_dof).mean();
65 for (int i = n_fe_dof; i < ndof; ++i)
66 {
67 masked_lumped_mass_sqrt_.coeffRef(i, i) = avg_mass;
68 }
69 }
70
71 // Remove non-boundary ndof from mass matrix
72 masked_lumped_mass_sqrt_.prune([&](const int &row, const int &col, const double &value) -> bool {
73 assert(row == col); // matrix should be diagonal
74 return !is_boundary_dof[row];
75 });
76 masked_lumped_mass_ = masked_lumped_mass_sqrt_;
77 mask_.resize(masked_lumped_mass_.rows(), masked_lumped_mass_.cols());
78 mask_.setIdentity();
79 mask_.prune([&](const int &row, const int &col, const double &value) -> bool {
80 assert(row == col); // matrix should be diagonal
81 return !is_boundary_dof[row];
82 });
83
84 std::vector<Eigen::Triplet<double>> tmp_triplets;
85 tmp_triplets.reserve(masked_lumped_mass_sqrt_.nonZeros());
86 for (int k = 0; k < masked_lumped_mass_sqrt_.outerSize(); ++k)
87 {
88 for (StiffnessMatrix::InnerIterator it(masked_lumped_mass_sqrt_, k); it; ++it)
89 {
90 assert(it.col() == k);
91 tmp_triplets.emplace_back(it.row(), it.col(), sqrt(it.value()));
92 }
93 }
94
95 masked_lumped_mass_sqrt_.setFromTriplets(tmp_triplets.begin(), tmp_triplets.end());
96 masked_lumped_mass_sqrt_.makeCompressed();
97
98 lagr_mults_.resize(ndof);
99 lagr_mults_.setZero();
100 }

102 double BCLagrangianForm::value_unweighted(const Eigen::VectorXd &x) const
103 {
104 const Eigen::VectorXd dist = x - target_x_;
105 const double L_penalty = -lagr_mults_.transpose() * masked_lumped_mass_sqrt_ * dist;
106 const double A_penalty = 0.5 * dist.transpose() * masked_lumped_mass_ * dist;
107
108 return L_penalty + k_al_ * A_penalty;
109 }
