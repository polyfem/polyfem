88 virtual bool compute_update_direction(
89 ProblemType &objFunc,
90 const TVector &x,
91 const TVector &grad,
92 TVector &direction) override
93 {
94 if (this->descent_strategy == 2)
95 {
96 direction = -grad;
97 return true;
98 }
99
100 {
101 POLYFEM_SCOPED_TIMER("assembly time", this->assembly_time);
102
103 if (this->descent_strategy == 1)
104 objFunc.set_project_to_psd(true);
105 else if (this->descent_strategy == 0)
106 objFunc.set_project_to_psd(false);
107 else
108 assert(false);
109
110 objFunc.hessian(x, hessian);
111
112 if (reg_weight > 0)
113 {
114 hessian += reg_weight * polyfem::utils::sparse_identity(hessian.rows(), hessian.cols());
115 }
116 }
117
118 {
119 POLYFEM_SCOPED_TIMER("linear solve", this->inverting_time);
120 // TODO: get the correct size
121 linear_solver->analyzePattern(hessian, hessian.rows());
122
123 try
124 {
125 linear_solver->factorize(hessian);
126 }
127 catch (const std::runtime_error &err)
128 {
129 increase_descent_strategy();
130 // warn if using gradient descent
131 polyfem::logger().log(
132 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
133 "Unable to factorize Hessian: \"{}\"; reverting to {}",
134 err.what(), this->descent_strategy_name());
135 // polyfem::write_sparse_matrix_csv("problematic_hessian.csv", hessian);
136 return compute_update_direction(objFunc, x, grad, direction);
137 }
138
139 linear_solver->solve(-grad, direction); // H Δx = -g
140 }
141
142 // gradient descent, check descent direction
143 const double residual = (hessian * direction + grad).norm(); // H Δx + g = 0
144 if (std::isnan(residual))
145 {
146 increase_descent_strategy();
147 polyfem::logger().log(
148 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
149 "nan linear solve residual {} (||∇f||={}); reverting to {}",
150 residual, grad.norm(), this->descent_strategy_name());
151 return compute_update_direction(objFunc, x, grad, direction);
152 }
153 else if (residual > std::max(1e-8 * grad.norm(), 1e-5))
154 {
155 increase_descent_strategy();
156 polyfem::logger().log(
157 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
158 "large linear solve residual {} (||∇f||={}); reverting to {}",
159 residual, grad.norm(), this->descent_strategy_name());
160 return compute_update_direction(objFunc, x, grad, direction);
161 }
162 else
163 {
164 polyfem::logger().trace("linear solve residual {}", residual);
165 }
166
167 // do this check here because we need to repeat the solve without resetting reg_weight
168 if (grad.dot(direction) >= 0)
169 {
170 increase_descent_strategy();
171 polyfem::logger().log(
172 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
173 "[{}] direction is not a descent direction (Δx⋅g={}≥0); reverting to {}",
174 name(), direction.dot(grad), descent_strategy_name());
175 return compute_update_direction(objFunc, x, grad, direction);
176 }
177
178 json info;
179 linear_solver->getInfo(info);
180 internal_solver_info.push_back(info);
181
182 reg_weight /= reg_weight_dec;
183 if (reg_weight < reg_weight_min)
184 reg_weight = 0;
185
186 return true;
187 }
