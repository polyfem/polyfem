52 template <class FullMat, class ReducedMat>
53 static void full_to_reduced_aux(const State &state, const int full_size, const int reduced_size, const FullMat &full, ReducedMat &reduced)
54 {
55 using namespace polyfem;
56
57 // Reduced is already at the full size
58 if (full_size == reduced_size || full.size() == reduced_size)
59 {
60 reduced = full;
61 return;
62 }
63
64 assert(full.size() == full_size);
65 assert(full.cols() == 1);
66 reduced.resize(reduced_size, 1);
67
68 long j = 0;
69 size_t k = 0;
70 for (int i = 0; i < full.size(); ++i)
71 {
72 if (k < state.boundary_nodes.size() && state.boundary_nodes[k] == i)
73 {
74 ++k;
75 continue;
76 }
77
78 reduced(j++) = full(i);
79 }
80 assert(j == reduced.size());
81 }

83 template <class ReducedMat, class FullMat>
84 static void reduced_to_full_aux(const State &state, const int full_size, const int reduced_size, const ReducedMat &reduced, const Eigen::MatrixXd &rhs, FullMat &full)
85 {
86 using namespace polyfem;
87
88 // Full is already at the reduced size
89 if (full_size == reduced_size || full_size == reduced.size())
90 {
91 full = reduced;
92 return;
93 }
94
95 assert(reduced.size() == reduced_size);
96 assert(reduced.cols() == 1);
97 full.resize(full_size, 1);
98
99 long j = 0;
100 size_t k = 0;
101 for (int i = 0; i < full.size(); ++i)
102 {
103 if (k < state.boundary_nodes.size() && state.boundary_nodes[k] == i)
104 {
105 ++k;
106 full(i) = rhs(i);
107 continue;
108 }
109
110 full(i) = reduced(j++);
111 }
112
113 assert(j == reduced.size());
114 }
