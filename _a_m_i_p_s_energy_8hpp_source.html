75 {
76 using std::pow;
77 using MatrixNT = Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3>;
78
79 MatrixNT F = MatrixNT::Zero(size(), size());
80 for (int i = 0; i < size(); ++i)
81 {
82 for (int j = 0; j < size(); ++j)
83 {
84 for (int k = 0; k < size(); ++k)
85 {
86 F(i, j) += def_grad(i, k) * canonical_transformation_[el_id](k, j);
87 }
88 }
89 }
90
91 T J = polyfem::utils::determinant(F);
92 if (J <= 0)
93 J = T(std::nan(""));
94 return (F.transpose() * F).trace() / pow(J, 2. / size());
95 }
