332 T HookeLinearElasticity::compute_energy_aux(const NonLinearAssemblerData &data) const
333 {
334 typedef Eigen::Matrix<T, Eigen::Dynamic, 1> AutoDiffVect;
335 typedef Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> AutoDiffGradMat;
336
337 AutoDiffVect local_disp;
338 get_local_disp(data, size(), local_disp);
339
340 AutoDiffGradMat disp_grad(size(), size());
341
342 T energy = T(0.0);
343
344 const int n_pts = data.da.size();
345 for (long p = 0; p < n_pts; ++p)
346 {
347 compute_disp_grad_at_quad(data, local_disp, p, size(), disp_grad);
348
349 AutoDiffGradMat strain = strain_from_disp_grad(disp_grad);
350 AutoDiffGradMat stress_tensor(size(), size());
351
352 if (size() == 2)
353 {
354 std::array<T, 3> eps;
355 eps[0] = strain(0, 0);
356 eps[1] = strain(1, 1);
357 eps[2] = 2 * strain(0, 1);
358
359 stress_tensor << stress(elasticity_tensor_, eps, 0), stress(elasticity_tensor_, eps, 2),
360 stress(elasticity_tensor_, eps, 2), stress(elasticity_tensor_, eps, 1);
361 }
362 else
363 {
364 std::array<T, 6> eps;
365 eps[0] = strain(0, 0);
366 eps[1] = strain(1, 1);
367 eps[2] = strain(2, 2);
368 eps[3] = 2 * strain(1, 2);
369 eps[4] = 2 * strain(0, 2);
370 eps[5] = 2 * strain(0, 1);
371
372 stress_tensor << stress(elasticity_tensor_, eps, 0), stress(elasticity_tensor_, eps, 5), stress(elasticity_tensor_, eps, 4),
373 stress(elasticity_tensor_, eps, 5), stress(elasticity_tensor_, eps, 1), stress(elasticity_tensor_, eps, 3),

375 }
