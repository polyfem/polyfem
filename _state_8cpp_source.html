80 // Only one node per vertex, so this is an identity map.
81 in_node_to_in_primitive.head(num_vertex_nodes).setLinSpaced(num_vertex_nodes, 0, num_vertex_nodes - 1); // vertex nodes
82 in_node_offset.head(num_vertex_nodes).setZero();
83
84 int prim_offset = n_vertices;

86 auto foo = [&](const int num_prims, const int num_prim_nodes) {
87 if (num_prims <= 0 || num_prim_nodes <= 0)
88 return;
89 const Eigen::VectorXi range = Eigen::VectorXi::LinSpaced(num_prim_nodes, 0, num_prim_nodes - 1);
90 // TODO: This assumes isotropic degree of element.
91 const int node_per_prim = num_prim_nodes / num_prims;
92
93 in_node_to_in_primitive.segment(node_offset, num_prim_nodes) =
94 range.array() / node_per_prim + prim_offset;
95
96 in_node_offset.segment(node_offset, num_prim_nodes) =
97 range.unaryExpr([&](const int x) { return x % node_per_prim; });
98
99 prim_offset += num_prims;
100 node_offset += num_prim_nodes;
101 };
102
103 foo(mesh.n_edges(), num_edge_nodes);

182 void State::build_node_mapping()
183 {
184 if (!mesh->is_conforming())
185 return;
186 const int num_vertex_nodes = mesh_nodes->num_vertex_nodes();
187 const int num_edge_nodes = mesh_nodes->num_edge_nodes();
188 const int num_face_nodes = mesh_nodes->num_face_nodes();
189 const int num_cell_nodes = mesh_nodes->num_cell_nodes();
190
191 const int num_nodes = num_vertex_nodes + num_edge_nodes + num_face_nodes + num_cell_nodes;
192
193 const long n_vertices = num_vertex_nodes;
194 const int num_in_primitives = n_vertices + mesh->n_edges() + mesh->n_faces() + mesh->n_cells();
195 const int num_primitives = mesh->n_vertices() + mesh->n_edges() + mesh->n_faces() + mesh->n_cells();
196
197 Eigen::VectorXi in_node_to_in_primitive;
198 Eigen::VectorXi in_node_offset;
199 build_in_node_to_in_primitive(*mesh, *mesh_nodes, in_node_to_in_primitive, in_node_offset);
200
201 build_in_primitive_to_primitive(
202 *mesh, *mesh_nodes,
203 mesh->in_ordered_vertices(),
204 mesh->in_ordered_edges(),
205 mesh->in_ordered_faces(),
206 in_primitive_to_primitive);
207
208 const auto primitive_offset = [&](int node) {
209 if (mesh_nodes->is_vertex_node(node))
210 return 0;
211 else if (mesh_nodes->is_edge_node(node))
212 return mesh->n_vertices();
213 else if (mesh_nodes->is_face_node(node))
214 return mesh->n_vertices() + mesh->n_edges();
215 else if (mesh_nodes->is_cell_node(node))
216 return mesh->n_vertices() + mesh->n_edges() + mesh->n_faces();
217 throw std::runtime_error("Invalid node ID!");
218 };
219
220 std::vector<std::vector<int>> primitive_to_nodes(num_primitives);
221 const std::vector<int> &grouped_nodes = mesh_nodes->primitive_to_node();
222 int node_count = 0;
223 for (int i = 0; i < grouped_nodes.size(); i++)
224 {
225 int node = grouped_nodes[i];
226 assert(node < num_nodes);
227 if (node >= 0)
228 {
229 int primitive = mesh_nodes->node_to_primitive_gid().at(node) + primitive_offset(i);
230 assert(primitive < num_primitives);
231 primitive_to_nodes[primitive].push_back(node);
232 node_count++;
233 }
234 }
235 assert(node_count == num_nodes);
236
237 in_node_to_node.resize(num_nodes);
238 for (int i = 0; i < num_nodes; i++)
239 {
240 // input node id -> input primitive -> primitive -> node
241 const std::vector<int> &possible_nodes =
242 primitive_to_nodes[in_primitive_to_primitive[in_node_to_in_primitive[i]]];
243 assert(possible_nodes.size() > 0);
244 if (possible_nodes.size() > 1)
245 {
246#ifndef NDEBUG
247 // assert(mesh_nodes->is_edge_node(i)); // TODO: Handle P4+
248 for (int possible_node : possible_nodes)
249 assert(mesh_nodes->is_edge_node(possible_node)); // TODO: Handle P4+
250#endif
251
252 int e_id = in_primitive_to_primitive[in_node_to_in_primitive[i]] - mesh->n_vertices();
253 assert(e_id < mesh->n_edges());
254 RowVectorNd v0 = mesh_nodes->node_position(in_node_to_node[mesh->edge_vertex(e_id, 0)]);
255 RowVectorNd a = mesh_nodes->node_position(possible_nodes[0]);
256 RowVectorNd b = mesh_nodes->node_position(possible_nodes[1]);
257 // Assume possible nodes are ordered, so only need to check order of two nodes
258
259 // Input edges are sorted, so if a is closer to v0 then the order is correct
260 // otherwise the nodes are flipped.
261 assert(mesh->edge_vertex(e_id, 0) < mesh->edge_vertex(e_id, 1));
262 int offset = (a - v0).squaredNorm() < (b - v0).squaredNorm()
263 ? in_node_offset[i]
264 : (possible_nodes.size() - in_node_offset[i] - 1);
265 in_node_to_node[i] = possible_nodes[offset];
266 }
267 else
268 in_node_to_node[i] = possible_nodes[0];
269 }

329 void State::compute_mesh_size(const Mesh &mesh_in, const std::vector<ElementBases> &bases_in, const int n_samples)
330 {
331 Eigen::MatrixXd samples_simplex, samples_cube, mapped, p0, p1, p;
332
333 mesh_size = 0;
334 average_edge_length = 0;
335 min_edge_length = std::numeric_limits<double>::max();
336
337 if (!args["output"]["advanced"]["curved_mesh_size"])
338 {
339 mesh_in.get_edges(p0, p1);
340 p = p0 - p1;
341 min_edge_length = p.rowwise().norm().minCoeff();
342 average_edge_length = p.rowwise().norm().mean();
343 mesh_size = p.rowwise().norm().maxCoeff();
344
345 logger().info("hmin: {}", min_edge_length);
346 logger().info("hmax: {}", mesh_size);
347 logger().info("havg: {}", average_edge_length);
348
349 return;
350 }
351
352 if (mesh_in.is_volume())
353 {
354 EdgeSampler::sample_3d_simplex(n_samples, samples_simplex);
355 EdgeSampler::sample_3d_cube(n_samples, samples_cube);
356 }
357 else
358 {
359 EdgeSampler::sample_2d_simplex(n_samples, samples_simplex);
360 EdgeSampler::sample_2d_cube(n_samples, samples_cube);
361 }
362
363 int n = 0;
364 for (size_t i = 0; i < bases_in.size(); ++i)
365 {
366 if (mesh_in.is_polytope(i))
367 continue;
368 int n_edges;
369
370 if (mesh_in.is_simplex(i))
371 {
372 n_edges = mesh_in.is_volume() ? 6 : 3;
373 bases_in[i].eval_geom_mapping(samples_simplex, mapped);
374 }
375 else
376 {
377 n_edges = mesh_in.is_volume() ? 12 : 4;
378 bases_in[i].eval_geom_mapping(samples_cube, mapped);
379 }
380
381 for (int j = 0; j < n_edges; ++j)
382 {
383 double current_edge = 0;
384 for (int k = 0; k < n_samples - 1; ++k)
385 {
386 p0 = mapped.row(j * n_samples + k);
387 p1 = mapped.row(j * n_samples + k + 1);
388 p = p0 - p1;
389
390 current_edge += p.norm();
391 }
392
393 mesh_size = std::max(current_edge, mesh_size);
394 min_edge_length = std::min(current_edge, min_edge_length);
395 average_edge_length += current_edge;
396 ++n;
397 }
398 }
399
400 average_edge_length /= n;
401
402 logger().info("hmin: {}", min_edge_length);
403 logger().info("hmax: {}", mesh_size);
404 logger().info("havg: {}", average_edge_length);
405 }

474 void compute_integral_constraints(
475 const Mesh3D &mesh,
476 const int n_bases,
477 const std::vector<ElementBases> &bases,
478 const std::vector<ElementBases> &gbases,
479 Eigen::MatrixXd &basis_integrals)

482 {
483 logger().error("Works only on volumetric meshes!");
484 return;
485 }
486 assert(mesh.is_volume());
487
488 basis_integrals.resize(n_bases, 9);
489 basis_integrals.setZero();
490 Eigen::MatrixXd rhs(n_bases, 9);
491 rhs.setZero();
492
493 const int n_elements = mesh.n_elements();
494 for (int e = 0; e < n_elements; ++e)
495 {
496 // if (mesh.is_polytope(e)) {
497 // continue;
498 // }
499 // ElementAssemblyValues vals = values[e];
500 // const ElementAssemblyValues &gvals = gvalues[e];
501 ElementAssemblyValues vals;
502 vals.compute(e, mesh.is_volume(), bases[e], gbases[e]);
503
504 // Computes the discretized integral of the PDE over the element
505 const int n_local_bases = int(vals.basis_values.size());
506 for (int j = 0; j < n_local_bases; ++j)
507 {
508 const AssemblyValues &v = vals.basis_values[j];
509 const double integral_100 = (v.grad_t_m.col(0).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
510 const double integral_010 = (v.grad_t_m.col(1).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
511 const double integral_001 = (v.grad_t_m.col(2).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
512
513 const double integral_110 = ((vals.val.col(1).array() * v.grad_t_m.col(0).array() + vals.val.col(0).array() * v.grad_t_m.col(1).array()) * vals.det.array() * vals.quadrature.weights.array()).sum();
514 const double integral_011 = ((vals.val.col(2).array() * v.grad_t_m.col(1).array() + vals.val.col(1).array() * v.grad_t_m.col(2).array()) * vals.det.array() * vals.quadrature.weights.array()).sum();
515 const double integral_101 = ((vals.val.col(0).array() * v.grad_t_m.col(2).array() + vals.val.col(2).array() * v.grad_t_m.col(0).array()) * vals.det.array() * vals.quadrature.weights.array()).sum();
516
517 const double integral_200 = 2 * (vals.val.col(0).array() * v.grad_t_m.col(0).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
518 const double integral_020 = 2 * (vals.val.col(1).array() * v.grad_t_m.col(1).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
519 const double integral_002 = 2 * (vals.val.col(2).array() * v.grad_t_m.col(2).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
520
521 const double area = (v.val.array() * vals.det.array() * vals.quadrature.weights.array()).sum();
522
523 for (size_t ii = 0; ii < v.global.size(); ++ii)
524 {
525 basis_integrals(v.global[ii].index, 0) += integral_100 * v.global[ii].val;
526 basis_integrals(v.global[ii].index, 1) += integral_010 * v.global[ii].val;
527 basis_integrals(v.global[ii].index, 2) += integral_001 * v.global[ii].val;
528
529 basis_integrals(v.global[ii].index, 3) += integral_110 * v.global[ii].val;
530 basis_integrals(v.global[ii].index, 4) += integral_011 * v.global[ii].val;
531 basis_integrals(v.global[ii].index, 5) += integral_101 * v.global[ii].val;
532
533 basis_integrals(v.global[ii].index, 6) += integral_200 * v.global[ii].val;
534 basis_integrals(v.global[ii].index, 7) += integral_020 * v.global[ii].val;
535 basis_integrals(v.global[ii].index, 8) += integral_002 * v.global[ii].val;
536
537 rhs(v.global[ii].index, 6) += -2.0 * area * v.global[ii].val;
538 rhs(v.global[ii].index, 7) += -2.0 * area * v.global[ii].val;
539 rhs(v.global[ii].index, 8) += -2.0 * area * v.global[ii].val;
540 }
541 }
542 }
543
544 basis_integrals -= rhs;
545 }
