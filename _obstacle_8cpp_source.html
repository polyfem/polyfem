155 void Obstacle::update_displacement(const double t, Eigen::MatrixXd &sol) const
156 {
157 // NOTE: assumes obstacle displacements is stored at the bottom of sol
158 const int offset = sol.rows() - v_.rows() * (sol.cols() == 1 ? dim_ : 1);
159
160 int start = 0;
161
162 for (int k = 0; k < endings_.size(); ++k)
163 {
164 const int to = endings_[k];
165 const auto &disp = displacements_[k];
166 const auto &interp = displacements_interpolation_[k];
167
168 for (int i = start; i < to; ++i)
169 {
170 double x = v_(i, 0), y = v_(i, 1), z = dim_ == 2 ? 0 : v_(i, 2);
171 const double interp_val = interp->eval(t);
172
173 for (int d = 0; d < dim_; ++d)
174 {
175 if (sol.cols() == 1)
176 {
177 sol(offset + i * dim_ + d) = disp[d](x, y, z, t) * interp_val;
178 }
179 else
180 {
181 sol(offset + i, d) = disp[d](x, y, z, t) * interp_val;
182 }
183 }
184 }
185
186 start = to;
187 }
188 assert(endings_.empty() || (start * (sol.cols() == 1 ? dim_ : 1) + offset) == sol.rows());
189 }

197 void Obstacle::Plane::construct_vis_mesh()
198 {
199 constexpr int size_x = 10;
200 constexpr int size_y = 10;
201
202 if (dim_ == 2)
203 {
204 Eigen::Vector2d tangent(normal().y(), -normal().x());
205
206 vis_v_.resize(size_x + 1, 2);
207 for (int x = 0; x <= size_x; ++x)
208 {
209 vis_v_.row(x) = (x - size_x / 2.0) * tangent;
210 }
211
212 vis_e_.resize(size_x, 2);
213 for (int x = 0; x < size_x; ++x)
214 {
215 vis_e_.row(x) << x, x + 1;
216 }
217 }
218 else
219 {
220 assert(dim_ == 3);
221
222 Eigen::Vector3d cross_x = ipc::cross(Eigen::Vector3d::UnitX(), normal());
223 Eigen::Vector3d cross_y = ipc::cross(Eigen::Vector3d::UnitY(), normal());
224
225 Eigen::Vector3d tangent_x, tangent_y;
226 if (cross_x.squaredNorm() > cross_y.squaredNorm())
227 {
228 tangent_x = cross_x.normalized();
229 tangent_y = ipc::cross(normal(), cross_x).normalized();
230 }
231 else
232 {
233 tangent_x = cross_y.normalized();
234 tangent_y = ipc::cross(normal(), cross_y).normalized();
235 }
236
237 vis_v_.resize((size_x + 1) * (size_y + 1), 3);
238 for (int x = 0; x <= size_x; ++x)
239 {
240 for (int y = 0; y <= size_y; ++y)
241 {
242 vis_v_.row(x * size_y + y) = (x - size_x / 2.0) * tangent_x + (y - size_y / 2.0) * tangent_y;
243 }
244 }
245
246 vis_f_.resize(2 * size_x * size_y, 3);
247 for (int x = 0; x < size_x; ++x)
248 {
249 for (int y = 0; y < size_y; ++y)
250 {
251 vis_f_.row(2 * (x * size_x + y)) << x * size_x + y, x * size_x + y + 1, (x + 1) * size_x + y;
252 vis_f_.row(2 * (x * size_x + y) + 1) << x * size_x + y + 1, (x + 1) * size_x + (y + 1), (x + 1) * size_x + y;
253 }
254 }
255
256 igl::edges(vis_f_, vis_e_);
257 }
