54 void minimize(ProblemType &objFunc, TVector &x)
55 {
56 using namespace polyfem;
57
58 // ---------------------------
59 // Initialize the minimization
60 // ---------------------------
61
62 reset(objFunc, x); // place for children to initialize their fields
63
64 TVector grad = TVector::Zero(x.rows());
65 TVector delta_x = TVector::Zero(x.rows());
66
67 // double factor = 1e-5;
68
69 // Set these to nan to indicate they have not been computed yet
70 double old_energy = std::nan("");
71
72 {
73 POLYFEM_SCOPED_TIMER("constraint set update", constraint_set_update_time);
74 objFunc.solution_changed(x);
75 }
76
77 {
78 POLYFEM_SCOPED_TIMER("compute gradient", grad_time);
79 objFunc.gradient(x, grad);
80 }
81 double first_grad_norm = grad.norm();
82 if (std::isnan(first_grad_norm))
83 {
84 this->m_status = Status::UserDefined;
85 polyfem::logger().error("[{}] Initial gradient is nan; stopping", name());
86 m_error_code = ErrorCode::NanEncountered;
87 throw std::runtime_error("Gradient is nan; stopping");
88 return;
89 }
90 this->m_current.gradNorm = first_grad_norm / (normalize_gradient ? first_grad_norm : 1);
91 this->m_current.fDelta = old_energy;
92
93 this->m_status = checkConvergence(this->m_stop, this->m_current);
94 if (this->m_status != Status::Continue)
95 {
96 POLYFEM_SCOPED_TIMER("compute objective function", obj_fun_time);
97 this->m_current.fDelta = objFunc.value(x);
98 polyfem::logger().log(
99 spdlog::level::info, "[{}] {} (f={} ||∇f||={} g={} tol={})",
100 name(), "Not even starting, grad is small enough", this->m_current.fDelta, first_grad_norm,
101 this->m_current.gradNorm, this->m_stop.gradNorm);
102 update_solver_info();
103 return;
104 }
105
106 utils::Timer timer("non-linear solver", this->total_time);
107 timer.start();
108
109 m_line_search->use_grad_norm_tol = use_grad_norm_tol;
110
111 do
112 {
113 {
114 POLYFEM_SCOPED_TIMER("constraint set update", constraint_set_update_time);
115 objFunc.solution_changed(x);
116 }
117
118 double energy;
119 {
120 POLYFEM_SCOPED_TIMER("compute objective function", obj_fun_time);
121 energy = objFunc.value(x);
122 }
123 if (!std::isfinite(energy))
124 {
125 this->m_status = Status::UserDefined;
126 polyfem::logger().error("[{}] f(x) is nan or inf; stopping", name());
127 m_error_code = ErrorCode::NanEncountered;
128 throw std::runtime_error("f(x) is nan or inf; stopping");
129 break;
130 }
131
132 {
133 POLYFEM_SCOPED_TIMER("compute gradient", grad_time);
134 objFunc.gradient(x, grad);
135 }
136
137 const double grad_norm = grad.norm();
138 if (std::isnan(grad_norm))
139 {
140 this->m_status = Status::UserDefined;
141 polyfem::logger().error("[{}] Gradient is nan; stopping", name());
142 m_error_code = ErrorCode::NanEncountered;
143 throw std::runtime_error("Gradient is nan; stopping");
144 break;
145 }
146
147 // ------------------------
148 // Compute update direction
149 // ------------------------
150
151 // Compute a Δx to update the variable
152 if (!compute_update_direction(objFunc, x, grad, delta_x))
153 {
154 this->m_status = Status::Continue;
155 continue;
156 }
157
158 if (grad_norm != 0 && delta_x.dot(grad) >= 0)
159 {
160 increase_descent_strategy();
161 polyfem::logger().log(
162 spdlog::level::debug,
163 "[{}] direction is not a descent direction (Δx⋅g={}≥0); reverting to {}",
164 name(), delta_x.dot(grad), descent_strategy_name());
165 this->m_status = Status::Continue;
166 continue;
167 }
168
169 const double delta_x_norm = delta_x.norm();
170 if (std::isnan(delta_x_norm))
171 {
172 increase_descent_strategy();
173 this->m_status = Status::UserDefined;
174 polyfem::logger().debug("[{}] Δx is nan; reverting to {}", name(), descent_strategy_name());
175 this->m_status = Status::Continue;
176 continue;
177 }
178
179 if (!use_gradient_norm)
180 {
181 //TODO, we shold remove this
182 // Use the maximum absolute displacement value divided by the timestep,
183 // so the units are in velocity units.
184 // TODO: Set this to the actual timestep
185 double dt = 1;
186 // TODO: Also divide by the world scale to make this criteria scale invariant.
187 this->m_current.gradNorm = delta_x.template lpNorm<Eigen::Infinity>() / dt;
188 }
189 else
190 {
191 //if normalize_gradient, use relative to first norm
192 this->m_current.gradNorm = grad_norm / (normalize_gradient ? first_grad_norm : 1);
193 }
194 this->m_current.fDelta = std::abs(old_energy - energy); // / std::abs(old_energy);
195
196 this->m_status = checkConvergence(this->m_stop, this->m_current);
197
198 old_energy = energy;
199
200 // ---------------
201 // Variable update
202 // ---------------
203
204 // Perform a line_search to compute step scale
205 double rate = line_search(x, delta_x, objFunc);
206 if (std::isnan(rate))
207 {
208 // descent_strategy set by line_search upon failure
209 if (this->m_status == Status::Continue)
210 continue;
211 else
212 break;
213 }
214
215 x += rate * delta_x;
216
217 // -----------
218 // Post update
219 // -----------
220
221 descent_strategy = default_descent_strategy(); // Reset this for the next iterations
222
223 const double step = (rate * delta_x).norm();
224
225 if (objFunc.stop(x))
226 {
227 this->m_status = Status::UserDefined;
228 m_error_code = ErrorCode::Success;
229 polyfem::logger().debug("[{}] Objective decided to stop", name());
230 }
231
232 objFunc.post_step(this->m_current.iterations, x);
233
234 polyfem::logger().debug(
235 "[{}] iter={:} f={} ‖∇f‖={} ‖Δx‖={} Δx⋅∇f(x)={} g={} tol={} rate={} ‖step‖={}",
236 name(), this->m_current.iterations, energy, grad_norm, delta_x_norm, delta_x.dot(grad),
237 this->m_current.gradNorm, this->m_stop.gradNorm, rate, step);
238 ++this->m_current.iterations;
239 } while (objFunc.callback(this->m_current, x) && (this->m_status == Status::Continue));
240
241 timer.stop();
242
243 // -----------
244 // Log results
245 // -----------
246
247 std::string msg = "Finished";
248 spdlog::level::level_enum level = spdlog::level::info;
249 if (this->m_status == Status::IterationLimit)
250 {
251 const std::string msg = fmt::format("[{}] Reached iteration limit", name());
252 polyfem::logger().error(msg);
253 throw std::runtime_error(msg);
254 level = spdlog::level::err;
255 }
256 else if (this->m_current.iterations == 0)
257 {
258 const std::string msg = fmt::format("[{}] Unable to take a step", name());
259 polyfem::logger().error(msg);
260 throw std::runtime_error(msg);
261 level = this->m_status == Status::UserDefined ? spdlog::level::err : spdlog::level::warn;
262 }
263 else if (this->m_status == Status::UserDefined)
264 {
265 const std::string msg = fmt::format("[{}] Failed to find minimizer", name());
266 polyfem::logger().error(msg);
267 throw std::runtime_error(msg);
268 level = spdlog::level::err;
269 }
270 polyfem::logger().log(
271 level, "[{}] {}, took {}s (niters={} f={} ||∇f||={} ||Δx||={} Δx⋅∇f(x)={} g={} tol={})",
272 name(), msg, timer.getElapsedTimeInSec(), this->m_current.iterations, old_energy, grad.norm(), delta_x.norm(),
273 delta_x.dot(grad), this->m_current.gradNorm, this->m_stop.gradNorm);
274
275 log_times();
276 update_solver_info();
277 }

279 double
280 line_search(const TVector &x, const TVector &delta_x, ProblemType &objFunc)
281 {
282 POLYFEM_SCOPED_TIMER("line search", line_search_time);
283
284 if (!m_line_search)
285 return 1; // no linesearch
286
287 double rate = m_line_search->line_search(x, delta_x, objFunc);
288
289 if (std::isnan(rate) && descent_strategy < 2) // 2 is the max, grad descent
290 {
291 increase_descent_strategy();
292 polyfem::logger().warn(
293 "[{}] Line search failed; reverting to {}", name(), descent_strategy_name());
294 this->m_status = Status::Continue; // Try the step again with gradient descent
295 }
296 else if (std::isnan(rate))
297 {
298 assert(descent_strategy == 2); // failed on gradient descent
299 polyfem::logger().error("[{}] Line search failed on gradient descent; stopping", name());
300 this->m_status = Status::UserDefined; // Line search failed on gradient descent, so quit!
301 throw std::runtime_error("Line search failed on gradient descent");
302 }
303
304 return rate;
305 }
