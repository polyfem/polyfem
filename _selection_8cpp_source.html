29 std::shared_ptr<Selection> Selection::build(
30 const json &selection,
31 const Selection::BBox &mesh_bbox,
32 const size_t start_element_id,
33 const size_t end_element_id)
34 {
35 if (!selection.contains("id"))
36 {
37 logger().error("Selection does not contain an id field: {}", selection.dump());
38 throw std::runtime_error("Selection id not found");
39 }
40
41 std::shared_ptr<Selection> res = nullptr;
42 if (selection.contains("box"))
43 res = std::make_shared<BoxSelection>(selection, mesh_bbox, start_element_id, end_element_id);
44 else if (selection.contains("center"))
45 res = std::make_shared<SphereSelection>(selection, mesh_bbox, start_element_id, end_element_id);
46 else if (selection.contains("axis"))
47 res = std::make_shared<AxisPlaneSelection>(selection, mesh_bbox, start_element_id, end_element_id);
48 else if (selection.contains("normal"))
49 res = std::make_shared<PlaneSelection>(selection, mesh_bbox, start_element_id, end_element_id);
50 else if (selection["id"].is_string()) // assume ID is a file path
51 res = std::make_shared<FileSelection>(selection["id"], start_element_id, end_element_id, selection.value("id_offset", 0));
52 else if (selection["id"].is_number_integer()) // assume ID is uniform
53 res = std::make_shared<UniformSelection>(selection["id"], start_element_id, end_element_id);
54 else
55 {
56 logger().error("Selection not recognized: {}", selection.dump());
57 throw std::runtime_error("Selection not recognized");
58 }
59
60 return res;
61 }

88 bool BoxSelection::inside(const size_t element_id, const RowVectorNd &p) const
89 {
90 if (!Selection::inside(element_id, p))
91 return false;
92
93 assert(bbox_[0].size() == p.size());
94 assert(bbox_[1].size() == p.size());
95 bool inside = true;
96
97 for (int d = 0; d < p.size(); ++d)
98 {
99 if (p[d] < bbox_[0][d] || p[d] > bbox_[1][d])
100 {
101 inside = false;
102 break;
103 }
104 }
105
106 return inside;

145 AxisPlaneSelection::AxisPlaneSelection(
146 const json &selection,
147 const Selection::BBox &mesh_bbox,
148 const size_t start_element_id,
149 const size_t end_element_id)
150 : Selection(selection["id"].get<int>(), start_element_id, end_element_id)
151 {
152 position_ = selection["position"];
153
154 if (selection["axis"].is_string())
155 {
156 std::string axis = selection["axis"];
157 int sign = axis[0] == '-' ? -1 : 1;
158 int dim = std::tolower(axis.back()) - 'x' + 1;
159 assert(dim >= 1 && dim <= 3);
160 axis_ = sign * dim;
161 }
162 else
163 {
164 assert(selection["axis"].is_number_integer());
165 axis_ = selection["axis"];
166 assert(std::abs(axis_) >= 1 && std::abs(axis_) <= 3);
167 }
168
169 if (selection.value("relative", false))
170 {
171 int dim = std::abs(axis_) - 1;
172 position_ = (mesh_bbox[1][dim] - mesh_bbox[0][dim]) * position_ + mesh_bbox[0][dim];
173 }
174 }

176 bool AxisPlaneSelection::inside(const size_t element_id, const RowVectorNd &p) const
177 {
178 if (!Selection::inside(element_id, p))
179 return false;
180
181 const double v = p[std::abs(axis_) - 1];
182
183 if (axis_ > 0)
184 return v >= position_;
185 else
186 return v <= position_;

241 FileSelection::FileSelection(
242 const std::string &file_path,
243 const size_t start_element_id,
244 const size_t end_element_id,
245 const int id_offset)
246 : SpecifiedSelection(std::vector<int>(end_element_id - start_element_id), start_element_id, end_element_id)
247 {
248 std::ifstream file(file_path);
249 if (!file.is_open())
250 {
251 logger().error("Unable to open selection file \"{}\"!", file_path);
252 return;
253 }
254
255 this->ids_.resize(end_element_id - start_element_id, 0);
256
257 std::string line;
258 int i = 0;
259 while (std::getline(file, line))
260 {
261 if (line.empty())
262 continue;
263 assert(i < this->ids_.size());
264 int id;
265 std::istringstream(line) >> id;
266 this->ids_[i++] = id + id_offset;
267 }
268
269 if (i != this->ids_.size())
270 {
271 logger().warn(
272 "Selection file \"{}\" is missing {} tag(s). Using 0 as default.",
273 file_path, this->ids_.size() - i);
274 }
275 }
