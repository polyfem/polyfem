16 double get_quality(const WildRemesher<WMTKMesh> &m, const typename WMTKMesh::Tuple &t)
17 {
18 // Global ids of the vertices of the triangle
19 auto vids = m.element_vids(t);
20
21 // Temporary variable to store the stacked coordinates of the triangle
22 constexpr int NDOF = WildRemesher<WMTKMesh>::DIM * WildRemesher<WMTKMesh>::VERTICES_PER_ELEMENT;
23 std::array<double, NDOF> T;
24 for (auto i = 0; i < WildRemesher<WMTKMesh>::VERTICES_PER_ELEMENT; i++)
25 for (auto j = 0; j < WildRemesher<WMTKMesh>::DIM; j++)
26 T[i * WildRemesher<WMTKMesh>::DIM + j] =
27 m.vertex_attrs[vids[i]].rest_position[j];
28
29 // Energy evaluation
30 double energy;
31 if constexpr (std::is_same_v<wmtk::TriMesh, WMTKMesh>)
32 energy = wmtk::AMIPS2D_energy(T);

34 energy = wmtk::AMIPS_energy(T);
35
36 // Filter for numerical issues
37 if (!std::isfinite(energy))
38 return std::numeric_limits<double>::max();
39
40 return energy;

183 template <class WMTKMesh>
184 bool WildRemesher<WMTKMesh>::smooth_after(const Tuple &v)
185 {
186 if (!WMTKMesh::smooth_before(v))
187 return false;
188
189 const size_t vid = v.vid(*this);
190
191 const std::vector<Tuple> one_ring = this->get_one_ring_elements_for_vertex(v);
192 assert(one_ring.size() > 0);
193
194 // ---------------------------------------------------------------------
195 // 1. update rest position of new vertex
196
197 // Computes the maximal error around the one ring
198 // that is needed to ensure the operation will decrease the error measure
199 double max_quality = 0;
200 for (const Tuple &t : get_one_ring_elements_for_vertex(v))
201 max_quality = std::max(max_quality, get_quality(*this, t));
202 assert(max_quality > 0); // If max quality is zero it is likely that the triangles are flipped
203
204 // Collects the coordinate of all vertices in the 1-ring
205 constexpr int NDOF = DIM * VERTICES_PER_ELEMENT;
206 std::vector<std::array<double, NDOF>> assembles(one_ring.size());
207
208 // For each triangle, make a reordered copy of the vertices so that
209 // the vertex to optimize is always the first
210 for (int i = 0; i < one_ring.size(); i++)
211 {
212 const Tuple &t = one_ring[i];
213 auto t_id = t.fid(*this);
214
215 assert(!is_inverted(t));
216 const auto local_verts = orient_preserve_element_reorder(element_vids(t), vid);
217
218 for (auto j = 0; j < VERTICES_PER_ELEMENT; j++)
219 for (auto d = 0; d < DIM; d++)
220 assembles[i][j * DIM + d] = vertex_attrs[local_verts[j]].rest_position[d];
221 }
222
223 // Make a backup of the current configuration
224 LocalMesh<This> old_local_mesh(*this, one_ring, false);
225 const VectorNd old_rest_position = vertex_attrs[vid].rest_position;
226
227 // Minimize distortion using newton's method
228 const auto log_lvl = wmtk::logger().level();
229 wmtk::logger().set_level(spdlog::level::warn);
230 if constexpr (DIM == 2)
231 vertex_attrs[vid].rest_position = wmtk::newton_method_from_stack_2d(
232 assembles, wmtk::AMIPS2D_energy, wmtk::AMIPS2D_jacobian, wmtk::AMIPS2D_hessian);
233 else
234 vertex_attrs[vid].rest_position = wmtk::newton_method_from_stack(
235 assembles, wmtk::AMIPS_energy, wmtk::AMIPS_jacobian, wmtk::AMIPS_hessian);
236 wmtk::logger().set_level(log_lvl);
237
238 // The AMIPS energy should have prevented inversions
239 if (std::any_of(one_ring.begin(), one_ring.end(), [this](const Tuple &t) {
240 return this->is_rest_inverted(t);
241 }))
242 {
243 assert(false);
244 return false;
245 }
246
247 // ---------------------------------------------------------------------
248 // 2. project quantities so to minimize the L2 error
249
250 // Adjust the previous displacements so they results in the same previous positions
251 vertex_attrs[vid].projection_quantities.leftCols(n_quantities() / 3) +=
252 old_rest_position - vertex_attrs[vid].rest_position;
253
254 LocalMesh<This> new_local_mesh(*this, one_ring, false);
255
256 project_local_quantities(*this, old_local_mesh, new_local_mesh);
257
258 // The Constrained L2 Projection should have prevented inversions
259 if (std::any_of(one_ring.begin(), one_ring.end(), [this](const Tuple &t) {
260 return this->is_inverted(t);
261 }))
262 {
263 assert(false);
264 return false;
265 }
266
267 return true;
268 }
