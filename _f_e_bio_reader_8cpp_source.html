63 std::string load_materials(const XMLNode *febio, std::map<int, std::tuple<double, double, double, std::string>> &materials)
64 {
65 double E;
66 double nu;
67 double rho;
68 std::vector<std::string> material_names;
69
70 const tinyxml2::XMLElement *material_parent = febio->FirstChildElement("Material");
71 for (const tinyxml2::XMLElement *material_node = material_parent->FirstChildElement("material"); material_node != NULL; material_node = material_node->NextSiblingElement("material"))
72 {
73 const std::string material = std::string(material_node->Attribute("type"));
74 const int mid = material_node->IntAttribute("id");
75
76 E = material_node->FirstChildElement("E")->DoubleText();
77 nu = material_node->FirstChildElement("v")->DoubleText();
78 if (material_node->FirstChildElement("density"))
79 rho = material_node->FirstChildElement("density")->DoubleText();
80 else
81 rho = 1;
82
83 std::string mat = "";
84 if (material == "neo-Hookean")
85 mat = "NeoHookean";
86 else if (material == "isotropic elastic")
87 mat = "LinearElasticity";
88 else
89 {
90 logger().error("Unsupported material {}, reverting to isotropic elastic", material);
91 mat = "LinearElasticity";
92 }
93
94 material_names.push_back(mat);
95 materials[mid] = std::tuple<double, double, double, std::string>(E, nu, rho, mat);
96 }
97
98 std::sort(material_names.begin(), material_names.end());
99 material_names.erase(std::unique(material_names.begin(), material_names.end()), material_names.end());
100 // assert(material_names.size() == 1);
101 if (material_names.size() != 1)
102 {
103 return "MultiModels";
104 }
105 return material_names.front();
106 }

130 template <typename XMLNode>
131 int load_elements(const XMLNode *geometry, const int numV, const std::map<int, std::tuple<double, double, double, std::string>> &materials, Eigen::MatrixXi &T, std::vector<std::vector<int>> &nodes, Eigen::MatrixXd &Es, Eigen::MatrixXd &nus, Eigen::MatrixXd &rhos, std::vector<std::string> &mats, std::vector<int> &mids)
132 {
133 std::vector<Eigen::VectorXi> els;
134 nodes.clear();
135 mids.clear();
136 int order = 1;
137 bool is_hex = false;
138 std::string type = "";
139
140 for (const tinyxml2::XMLElement *elements = geometry->FirstChildElement("Elements"); elements != NULL; elements = elements->NextSiblingElement("Elements"))
141 {
142 const std::string el_type = std::string(elements->Attribute("type"));
143 const int mid = elements->IntAttribute("mat");
144
145 if (el_type != "tet4" && el_type != "tet10" && el_type != "tet20" && el_type != "hex8")
146 {
147 logger().error("Unsupported elemet type {}", el_type);
148 continue;
149 }
150
151 if (type.empty())
152 {
153 if (el_type == "tet4" || el_type == "tet10" || el_type == "tet20")
154 type = "tet";
155 else
156 type = "hex";
157 }
158 else if (el_type.rfind(type, 0) != 0)
159 {
160 logger().error("Unsupported elemet type {} since the mesh contains also {}", el_type, type);
161 continue;
162 }
163
164 if (el_type == "tet4")
165 order = std::max(1, order);
166 else if (el_type == "tet10")
167 order = std::max(2, order);
168 else if (el_type == "tet20")
169 order = std::max(3, order);
170 else if (el_type == "hex8")
171 {
172 order = std::max(1, order);
173 is_hex = true;
174 }
175
176 for (const tinyxml2::XMLElement *child = elements->FirstChildElement("elem"); child != NULL; child = child->NextSiblingElement("elem"))
177 {
178 const std::string ids = std::string(child->GetText());
179 const auto tt = StringUtils::split(ids, ",");
180 assert(tt.size() >= 4);
181 const int node_size = is_hex ? 8 : 4;
182
183 els.emplace_back(node_size);
184
185 for (int n = 0; n < node_size; ++n)
186 {
187 els.back()[n] = atoi(tt[n].c_str()) - 1;
188 assert(els.back()[n] < numV);
189 }
190 nodes.emplace_back();
191 mids.emplace_back(mid);
192 for (int n = 0; n < tt.size(); ++n)
193 nodes.back().push_back(atoi(tt[n].c_str()) - 1);
194
195 if (el_type == "tet10")
196 {
197 assert(nodes.back().size() == 10);
198 std::swap(nodes.back()[8], nodes.back()[9]);
199 }
200 else if (el_type == "tet20")
201 {
202 assert(nodes.back().size() == 20);
203 std::swap(nodes.back()[8], nodes.back()[9]);
204 std::swap(nodes.back()[10], nodes.back()[11]);
205 std::swap(nodes.back()[12], nodes.back()[15]);
206 std::swap(nodes.back()[13], nodes.back()[14]);
207 std::swap(nodes.back()[16], nodes.back()[19]);
208 std::swap(nodes.back()[17], nodes.back()[19]);
209 }
210 }
211 }
212
213 T.resize(els.size(), is_hex ? 8 : 4);
214 Es.resize(els.size(), 1);
215 nus.resize(els.size(), 1);
216 rhos.resize(els.size(), 1);
217 mats.resize(els.size());
218 for (int i = 0; i < els.size(); ++i)
219 {
220 T.row(i) = els[i].transpose();
221 const auto it = materials.find(mids[i]);
222 assert(it != materials.end());
223 if (it == materials.end())
224 {
225 logger().error("Unable to find material {}", mids[i]);
226 throw std::runtime_error("Invalid material");
227 }
228 Es(i) = std::get<0>(it->second);
229 nus(i) = std::get<1>(it->second);
230 rhos(i) = std::get<2>(it->second);
231 mats[i] = std::get<3>(it->second);
232 }
233
234 return order;
235 }

338 {
339 std::map<int, BCData> allbc;
340 for (const tinyxml2::XMLElement *child = boundaries->FirstChildElement("fix"); child != NULL; child = child->NextSiblingElement("fix"))
341 {
342 const std::string name = std::string(child->Attribute("node_set"));
343 if (names.find(name) == names.end())
344 {
345 logger().error("Sideset {} not present, skipping", name);
346 continue;
347 }
348 const int id = names.at(name);
349 const std::string bc = std::string(child->Attribute("bc"));
350 const auto bcs = StringUtils::split(bc, ",");
351
352 BCData bcdata;
353 bcdata.val = Eigen::RowVector3d::Zero();
354 bcdata.isx = false;
355 bcdata.isy = false;
356 bcdata.isz = false;
357
358 for (const auto &s : bcs)
359 {

361 bcdata.isx = true;
362 else if (s == "y")
363 bcdata.isy = true;
364 else if (s == "z")
365 bcdata.isz = true;
366 }
367
368 auto it = allbc.find(id);
369 if (it == allbc.end())
370 {
371 allbc[id] = bcdata;
372 }
373 else
374 {

378 it->second.isx = true;
379 it->second.val(0) = 0;
380 }
381 if (bcdata.isy)
382 {
383 assert(!it->second.isz);
384 it->second.isy = true;
385 it->second.val(1) = 0;
386 }
387 if (bcdata.isz)
388 {
389 assert(!it->second.isz);
390 it->second.isz = true;
391 it->second.val(2) = 0;
392 }
393 }

400 {
401 logger().error("Sideset {} not present, skipping", name);
402 continue;
403 }
404 const int id = names.at(name);
405 const std::string bc = std::string(child->Attribute("bc"));
406
407 BCData bcdata;
408 bcdata.isx = bc == "x";
409 bcdata.isy = bc == "y";
410 bcdata.isz = bc == "z";
411
412 const double value = atof(child->FirstChildElement("scale")->GetText()) * (gproblem.is_time_dependent() ? dt : 1);
413 bcdata.val = Eigen::RowVector3d::Zero();
414
415 if (bcdata.isx)
416 bcdata.val(0) = value;
417 else if (bcdata.isy)
418 bcdata.val(1) = value;
419 else if (bcdata.isz)
420 bcdata.val(2) = value;
421
422 auto it = allbc.find(id);
423 if (it == allbc.end())
424 {
425 allbc[id] = bcdata;
426 }
427 else
428 {
429 if (bcdata.isx)
430 {
431 assert(!it->second.isx);
432 it->second.isx = true;
433 it->second.val(0) = bcdata.val(0);
434 }
435 if (bcdata.isy)

437 assert(!it->second.isy);
438 it->second.isy = true;
439 it->second.val(1) = bcdata.val(1);
440 }
441 if (bcdata.isz)

443 assert(!it->second.isz);

447 }
448 }
449
450 for (auto it = allbc.begin(); it != allbc.end(); ++it)
451 {
452 logger().trace("adding Dirichlet id={} value=({}) fixed=({}, {}, {})", it->first, it->second.val, it->second.isx, it->second.isy, it->second.isz);
453 gproblem.add_dirichlet_boundary(it->first, it->second.val, it->second.isx, it->second.isy, it->second.isz, get_interpolation(gproblem.is_time_dependent()));
454 }
455
456 for (const tinyxml2::XMLElement *child = boundaries->FirstChildElement("vector_bc"); child != NULL; child = child->NextSiblingElement("vector_bc"))
457 {
458 const std::string name = std::string(child->Attribute("node_set"));
459 if (names.find(name) == names.end())

462 continue;
463 }

466 const std::string values = resolve_path(std::string(child->Attribute("values")), root_file);

468 const double eps = 1e-3; //TODO
469 //TODO add is x,y,z
470
471 Eigen::MatrixXd centers_mat, values_mat;
472 read_matrix(centers, centers_mat);
473 read_matrix(values, values_mat);
474
475 RBFInterpolation interp(values_mat, centers_mat, rbf, eps);

477
478 gproblem.add_dirichlet_boundary(
479 id, [interp](double x, double y, double z, double t) {
480 Eigen::Matrix<double, 3, 1> v;
481 v[0] = x;
482 v[1] = y;
483 v[2] = z;
484 return interp.interpolate(v);
485 },

487 }

499 }

504 {
505 logger().error("Skipping scaling, center is not 3d");
506 continue;
507 }
508 const Eigen::Vector3d center(
509 atof(centrec[0].c_str()),
510 atof(centrec[1].c_str()),
511 atof(centrec[2].c_str()));
512
513 const double scaling = atof(factors.c_str());
514 logger().trace("adding scaling Dirichlet id={} center=({}) scaling={}", id, center.transpose(), scaling);

516 id, [center, scaling, is_time_dept](double x, double y, double z, double t) {
517 Eigen::Matrix<double, 3, 1> v;
518 Eigen::Matrix<double, 3, 1> target;
519 v[0] = x;
520 v[1] = y;
521 v[2] = z;
522 target = v;
523
524 const double s = is_time_dept ? (scaling * t) : scaling;
525 target -= center;
526 target *= s;
527 target += center;
528 return (target - v).eval();
529 },
530 true, true, true);
531 }

566 force *= dt;

574 //TODO added minus here

576 gproblem.add_pressure_boundary(names.at(name), -pressure, get_interpolation(gproblem.is_time_dependent()));
577 }

580 logger().error("Unsupported surface load {}", type);
581 }
582 }

622 void FEBioReader::load(const std::string &path, const json &args_in, State &state, const std::string &export_solution)
623 {
624 igl::Timer timer;
625 timer.start();
626 logger().info("Loading feb file...");
627
628 if (!args_in.contains("normalize_mesh"))
629 state.args["normalize_mesh"] = false;
630 if (!args_in.contains("quadrature_order"))
631 state.args["quadrature_order"] = 0;
632
633 if (args_in.contains("export"))
634 {
635 if (!export_solution.empty() && !args_in["export"].contains("solution_mat"))
636 state.args["export"]["solution_mat"] = export_solution;
637
638 if (!args_in["export"].contains("body_ids"))
639 state.args["export"]["body_ids"] = true;
640 }
641 else
642 {
643 if (!export_solution.empty())
644 state.args["export"]["solution_mat"] = export_solution;
645
646 state.args["export"]["body_ids"] = true;
647 }
648
649 state.args["root_path"] = path;
650
651 tinyxml2::XMLDocument doc;
652 doc.LoadFile(path.c_str());
653 if (doc.Error())
654 {
655 logger().error("Unable to read {}, error {}", path, doc.ErrorStr());
656 throw std::runtime_error("Invalid XML");
657 }
658
659 const auto *febio = doc.FirstChildElement("febio_spec");
660 const std::string ver = std::string(febio->Attribute("version"));
661 assert(ver == "2.5");
662 if (ver != "2.5")
663 {
664 logger().error("Unsuppoted FEBio version {}, use 2.5", ver);
665 throw std::runtime_error("Unsuppoted FEBio version");
666 }
667
668 const auto *control = febio->FirstChildElement("Control");
669 bool time_dependent = false;
