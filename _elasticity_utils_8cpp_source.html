325 double von_mises_stress_for_stress_tensor(const Eigen::MatrixXd &stress)
326 {
327 double von_mises_stress;
328
329 if (stress.rows() == 3)
330 {
331 von_mises_stress = 0.5 * (stress(0, 0) - stress(1, 1)) * (stress(0, 0) - stress(1, 1)) + 3.0 * stress(0, 1) * stress(1, 0);
332 von_mises_stress += 0.5 * (stress(2, 2) - stress(1, 1)) * (stress(2, 2) - stress(1, 1)) + 3.0 * stress(2, 1) * stress(2, 1);
333 von_mises_stress += 0.5 * (stress(2, 2) - stress(0, 0)) * (stress(2, 2) - stress(0, 0)) + 3.0 * stress(2, 0) * stress(2, 0);
334 }
335 else
336 {
337 // von_mises_stress = ( stress(0, 0) - stress(1, 1) ) * ( stress(0, 0) - stress(1, 1) ) + 3.0 * stress(0, 1) * stress(1, 0);
338 von_mises_stress = stress(0, 0) * stress(0, 0) - stress(0, 0) * stress(1, 1) + stress(1, 1) * stress(1, 1) + 3.0 * stress(0, 1) * stress(1, 0);
339 }
340
341 von_mises_stress = sqrt(fabs(von_mises_stress));
342
343 return von_mises_stress;
