12
13 double operator()(const double d, const double dhat) const override
14 {
15 if (d > dhat)
16 return 0;
17 else
18 return weight_ * (d - dhat) * (d - dhat);
19 }

21 {
22 if (d > dhat)

246 void LayerThicknessForm::build_collision_mesh()
247 {
248 Eigen::MatrixXd node_positions;
249 Eigen::MatrixXi boundary_edges, boundary_triangles;
250 std::vector<Eigen::Triplet<double>> displacement_map_entries;
251 io::OutGeometryData::extract_boundary_mesh(*state_.mesh, state_.n_geom_bases, state_.geom_bases(), state_.total_local_boundary,
252 node_positions, boundary_edges, boundary_triangles, displacement_map_entries);
253
254 std::vector<bool> is_on_surface;
255 is_on_surface.resize(node_positions.rows(), false);
256
257 assembler::ElementAssemblyValues vals;
258 Eigen::MatrixXd points, uv, normals;
259 Eigen::VectorXd weights;
260 Eigen::VectorXi global_primitive_ids;
261 for (const auto &lb : state_.total_local_boundary)
262 {
263 const int e = lb.element_id();
264 bool has_samples = utils::BoundarySampler::boundary_quadrature(lb, state_.n_boundary_samples(), *state_.mesh, false, uv, points, normals, weights, global_primitive_ids);
265
266 if (!has_samples)
267 continue;
268
269 const basis::ElementBases &gbs = state_.geom_bases()[e];
270
271 vals.compute(e, state_.mesh->is_volume(), points, gbs, gbs);
272
273 for (int i = 0; i < lb.size(); ++i)
274 {
275 const int primitive_global_id = lb.global_primitive_id(i);
276 const auto nodes = gbs.local_nodes_for_primitive(primitive_global_id, *state_.mesh);
277 const int boundary_id = state_.mesh->get_boundary_id(primitive_global_id);
278
279 if (!std::count(boundary_ids_.begin(), boundary_ids_.end(), boundary_id))
280 continue;
281
282 for (long n = 0; n < nodes.size(); ++n)
283 {
284 const assembler::AssemblyValues &v = vals.basis_values[nodes(n)];
285 is_on_surface[v.global[0].index] = true;
286 assert(v.global[0].index < node_positions.rows());
287 boundary_ids_to_dof_[boundary_id].insert(v.global[0].index);
288 }
289 }
290 }
291
292 Eigen::SparseMatrix<double> displacement_map;
293 if (!displacement_map_entries.empty())
294 {
295 displacement_map.resize(node_positions.rows(), state_.n_geom_bases);
296 displacement_map.setFromTriplets(displacement_map_entries.begin(), displacement_map_entries.end());
297 }
298
299 // Fix boundary edges and boundary triangles to exclude vertices not on triangles
300 Eigen::MatrixXi boundary_edges_alt(0, 2), boundary_triangles_alt(0, 3);
301 {
302 for (int i = 0; i < boundary_edges.rows(); ++i)
303 {
304 bool on_surface = true;
305 for (int j = 0; j < boundary_edges.cols(); ++j)
306 on_surface &= is_on_surface[boundary_edges(i, j)];
307 if (on_surface)
308 {
309 boundary_edges_alt.conservativeResize(boundary_edges_alt.rows() + 1, 2);
310 boundary_edges_alt.row(boundary_edges_alt.rows() - 1) = boundary_edges.row(i);
311 }
312 }
313
314 if (state_.mesh->is_volume())
315 {
316 for (int i = 0; i < boundary_triangles.rows(); ++i)
317 {
318 bool on_surface = true;
319 for (int j = 0; j < boundary_triangles.cols(); ++j)
320 on_surface &= is_on_surface[boundary_triangles(i, j)];
321 if (on_surface)
322 {
323 boundary_triangles_alt.conservativeResize(boundary_triangles_alt.rows() + 1, 3);
324 boundary_triangles_alt.row(boundary_triangles_alt.rows() - 1) = boundary_triangles.row(i);
325 }
326 }
327 }
328 else
329 boundary_triangles_alt.resize(0, 0);
330 }
331
332 collision_mesh_ = ipc::CollisionMesh(is_on_surface,
333 node_positions,
334 boundary_edges_alt,
335 boundary_triangles_alt,
336 displacement_map);
337
338 can_collide_cache_.resize(collision_mesh_.num_vertices(), collision_mesh_.num_vertices());
339 for (int i = 0; i < can_collide_cache_.rows(); ++i)
340 {
341 int dof_idx_i = collision_mesh_.to_full_vertex_id(i);
342 if (!is_on_surface[dof_idx_i])
343 continue;
344 for (int j = 0; j < can_collide_cache_.cols(); ++j)
345 {
346 int dof_idx_j = collision_mesh_.to_full_vertex_id(j);
347 if (!is_on_surface[dof_idx_j])
348 continue;
349
350 bool collision_allowed = true;
351 for (const auto &id : boundary_ids_)
352 if (boundary_ids_to_dof_[id].count(dof_idx_i) && boundary_ids_to_dof_[id].count(dof_idx_j))
353 collision_allowed = false;
354 can_collide_cache_(i, j) = collision_allowed;
355 }
356 }
357
358 collision_mesh_.can_collide = [&](size_t vi, size_t vj) {
359 return (bool)can_collide_cache_(vi, vj);
360 };
361
362 collision_mesh_.init_area_jacobians();
363 }
