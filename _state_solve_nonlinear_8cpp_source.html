38 std::shared_ptr<cppoptlib::NonlinearSolver<ProblemType>> State::make_nl_solver(
39 const std::string &linear_solver_type) const
40 {
41 const std::string name = args["solver"]["nonlinear"]["solver"];
42 const double dt = problem->is_time_dependent() ? args["time"]["dt"].get<double>() : 1.0;
43 if (name == "newton" || name == "Newton")
44 {
45 json linear_solver_params = args["solver"]["linear"];
46 if (!linear_solver_type.empty())
47 linear_solver_params["solver"] = linear_solver_type;
48 return std::make_shared<cppoptlib::SparseNewtonDescentSolver<ProblemType>>(
49 args["solver"]["nonlinear"], linear_solver_params, dt, units.characteristic_length());
50 }
51 else if (name == "lbfgs" || name == "LBFGS" || name == "L-BFGS")
52 {
53 return std::make_shared<cppoptlib::LBFGSSolver<ProblemType>>(args["solver"]["nonlinear"], dt, units.characteristic_length());
54 }
55 else
56 {
57 throw std::invalid_argument(fmt::format("invalid nonlinear solver type: {}", name));
58 }

285 void State::solve_tensor_nonlinear(Eigen::MatrixXd &sol, const int t, const bool init_lagging)
286 {
287 assert(solve_data.nl_problem != nullptr);
288 NLProblem &nl_problem = *(solve_data.nl_problem);
289
290 assert(sol.size() == rhs.size());
291
292 if (nl_problem.uses_lagging())
293 {
294 if (init_lagging)
295 {
296 POLYFEM_SCOPED_TIMER("Initializing lagging");
297 nl_problem.init_lagging(sol); // TODO: this should be u_prev projected
298 }
299 logger().info("Lagging iteration 1:");
300 }
301
302 // ---------------------------------------------------------------------
303
304 // Save the subsolve sequence for debugging
305 int subsolve_count = 0;
306 save_subsolve(subsolve_count, t, sol, Eigen::MatrixXd()); // no pressure
307
308 // ---------------------------------------------------------------------
309
310 std::shared_ptr<cppoptlib::NonlinearSolver<NLProblem>> nl_solver = make_nl_solver<NLProblem>();
311
312 ALSolver al_solver(
313 nl_solver, solve_data.al_lagr_form, solve_data.al_pen_form,
314 args["solver"]["augmented_lagrangian"]["initial_weight"],
315 args["solver"]["augmented_lagrangian"]["scaling"],
316 args["solver"]["augmented_lagrangian"]["max_weight"],
317 args["solver"]["augmented_lagrangian"]["eta"],
318 args["solver"]["augmented_lagrangian"]["max_solver_iters"],
319 [&](const Eigen::VectorXd &x) {
320 this->solve_data.update_barrier_stiffness(sol);
321 });
322
323 al_solver.post_subsolve = [&](const double al_weight) {
324 stats.solver_info.push_back(
325 {{"type", al_weight > 0 ? "al" : "rc"},
326 {"t", t}, // TODO: null if static?
327 {"info", nl_solver->get_info()}});
328 if (al_weight > 0)
329 stats.solver_info.back()["weight"] = al_weight;
330 save_subsolve(++subsolve_count, t, sol, Eigen::MatrixXd()); // no pressure
331 };
332
333 Eigen::MatrixXd prev_sol = sol;
334 al_solver.solve(nl_problem, sol, args["solver"]["augmented_lagrangian"]["force"]);
335
336 // ---------------------------------------------------------------------
337
338 // TODO: Make this more general
339 const double lagging_tol = args["solver"]["contact"].value("friction_convergence_tol", 1e-2) * units.characteristic_length();
340
341 if (!optimization_enabled)
342 {
343 // Lagging loop (start at 1 because we already did an iteration above)
344 bool lagging_converged = !nl_problem.uses_lagging();
345 for (int lag_i = 1; !lagging_converged; lag_i++)
346 {
347 Eigen::VectorXd tmp_sol = nl_problem.full_to_reduced(sol);
348
349 // Update the lagging before checking for convergence
350 nl_problem.update_lagging(tmp_sol, lag_i);
351
352 // Check if lagging converged
353 Eigen::VectorXd grad;
354 nl_problem.gradient(tmp_sol, grad);
355 const double delta_x_norm = (prev_sol - sol).lpNorm<Eigen::Infinity>();
356 logger().debug("Lagging convergence grad_norm={:g} tol={:g} (||x||={:g})", grad.norm(), lagging_tol, delta_x_norm);
357 if (grad.norm() <= lagging_tol)
358 {
359 logger().info(
360 "Lagging converged in {:d} iteration(s) (grad_norm={:g} tol={:g})",
361 lag_i, grad.norm(), lagging_tol);
362 lagging_converged = true;
363 break;
364 }
365
366 if (delta_x_norm <= 1e-12)
367 {
368 logger().warn(
369 "Lagging produced tiny update between iterations {:d} and {:d} (grad_norm={:g} grad_tol={:g} ||x||={:g} x_tol={:g}); stopping early",
370 lag_i - 1, lag_i, grad.norm(), lagging_tol, delta_x_norm, 1e-6);
371 lagging_converged = false;
372 break;
373 }
374
375 // Check for convergence first before checking if we can continue
376 if (lag_i >= nl_problem.max_lagging_iterations())
377 {
378 logger().warn(
379 "Lagging failed to converge with {:d} iteration(s) (grad_norm={:g} tol={:g})",
380 lag_i, grad.norm(), lagging_tol);
381 lagging_converged = false;
382 break;
383 }
384
385 // Solve the problem with the updated lagging
386 logger().info("Lagging iteration {:d}:", lag_i + 1);
387 nl_problem.init(sol);
388 solve_data.update_barrier_stiffness(sol);
389 nl_solver->minimize(nl_problem, tmp_sol);
390 prev_sol = sol;
391 sol = nl_problem.reduced_to_full(tmp_sol);
392
393 // Save the subsolve sequence for debugging and info
394 stats.solver_info.push_back(
395 {{"type", "rc"},
396 {"t", t}, // TODO: null if static?
397 {"lag_i", lag_i},
398 {"info", nl_solver->get_info()}});
399 save_subsolve(++subsolve_count, t, sol, Eigen::MatrixXd()); // no pressure
400 }
401 }
