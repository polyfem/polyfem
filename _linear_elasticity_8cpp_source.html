18 Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 9, 1>
19 LinearElasticity::assemble(const LinearAssemblerData &data) const
20 {
21 // mu ((gradi' gradj) Id + ((gradi gradj')') + lambda gradi *gradj';
22 const Eigen::MatrixXd &gradi = data.vals.basis_values[data.i].grad_t_m;
23 const Eigen::MatrixXd &gradj = data.vals.basis_values[data.j].grad_t_m;
24
25 Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 9, 1> res(size() * size());
26 res.setZero();
27
28 for (long k = 0; k < gradi.rows(); ++k)
29 {
30 Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 9, 1> res_k(size() * size());
31 // res_k.setZero();
32 const Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> outer = gradi.row(k).transpose() * gradj.row(k);
33 const double dot = gradi.row(k).dot(gradj.row(k));
34
35 double lambda, mu;
36 params_.lambda_mu(data.vals.quadrature.points.row(k), data.vals.val.row(k), data.vals.element_id, lambda, mu);
37
38 for (int ii = 0; ii < size(); ++ii)
39 {
40 for (int jj = 0; jj < size(); ++jj)
41 {
42 res_k(jj * size() + ii) = outer(ii * size() + jj) * mu + outer(jj * size() + ii) * lambda;
43 if (ii == jj)
44 res_k(jj * size() + ii) += mu * dot;
45 }
46 }
47 res += res_k * data.da(k);
48 }
49
50 return res;
51 }

93 // Compute \int mu eps : eps + lambda/2 tr(eps)^2 = \int mu tr(eps^2) + lambda/2 tr(eps)^2
94 template <typename T>
95 T LinearElasticity::compute_energy_aux(const NonLinearAssemblerData &data) const
96 {
97 typedef Eigen::Matrix<T, Eigen::Dynamic, 1> AutoDiffVect;
98 typedef Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> AutoDiffGradMat;
99
100 AutoDiffVect local_disp;
101 get_local_disp(data, size(), local_disp);
102
103 AutoDiffGradMat disp_grad(size(), size());
104
105 T energy = T(0.0);
106
107 const int n_pts = data.da.size();
108 for (long p = 0; p < n_pts; ++p)
109 {
110 compute_disp_grad_at_quad(data, local_disp, p, size(), disp_grad);
111
112 const AutoDiffGradMat strain = (disp_grad + disp_grad.transpose()) / T(2);
113
114 double lambda, mu;
115 params_.lambda_mu(data.vals.quadrature.points.row(p), data.vals.val.row(p), data.vals.element_id, lambda, mu);
116
117 const T val = mu * (strain.transpose() * strain).trace() + lambda / 2 * strain.trace() * strain.trace();
118
119 energy += val * data.da(p);
120 }
121 return energy;
122 }

162 void LinearElasticity::assign_stress_tensor(const int el_id, const ElementBases &bs, const ElementBases &gbs, const Eigen::MatrixXd &local_pts, const Eigen::MatrixXd &displacement, const int all_size, const ElasticityTensorType &type, Eigen::MatrixXd &all, const std::function<Eigen::MatrixXd(const Eigen::MatrixXd &)> &fun) const
163 {
164 all.resize(local_pts.rows(), all_size);
165 assert(displacement.cols() == 1);
166
167 Eigen::MatrixXd displacement_grad(size(), size());
168
169 ElementAssemblyValues vals;
170 vals.compute(el_id, size() == 3, local_pts, bs, gbs);
171
172 for (long p = 0; p < local_pts.rows(); ++p)
173 {
174 compute_diplacement_grad(size(), bs, vals, local_pts, p, displacement, displacement_grad);
175
176 if (type == ElasticityTensorType::F)
177 {
178 all.row(p) = fun(displacement_grad + Eigen::MatrixXd::Identity(size(), size()));
179 continue;
180 }
181
182 double lambda, mu;
183 params_.lambda_mu(local_pts.row(p), vals.val.row(p), vals.element_id, lambda, mu);
184
185 const Eigen::MatrixXd strain = (displacement_grad + displacement_grad.transpose()) / 2;
186 Eigen::MatrixXd stress = 2 * mu * strain + lambda * strain.trace() * Eigen::MatrixXd::Identity(size(), size());
187 if (type == ElasticityTensorType::PK1)
188 stress = pk1_from_cauchy(stress, displacement_grad + Eigen::MatrixXd::Identity(size(), size()));
189 else if (type == ElasticityTensorType::PK2)
190 stress = pk2_from_cauchy(stress, displacement_grad + Eigen::MatrixXd::Identity(size(), size()));
191
192 all.row(p) = fun(stress);
193 }

196 Eigen::Matrix<AutodiffScalarGrad, Eigen::Dynamic, 1, 0, 3, 1> LinearElasticity::kernel(const int dim, const AutodiffGradPt &r) const
197 {
198 Eigen::Matrix<AutodiffScalarGrad, Eigen::Dynamic, 1, 0, 3, 1> res(dim);
199 assert(r.size() == dim);
200
201 double mu, nu, lambda;
202 // per body lame parameter dont work here!
203 params_.lambda_mu(0, 0, 0, 0, 0, 0, 0, lambda, mu);
204
205 // convert to nu!
206 nu = lambda / (lambda * (dim - 1) + 2 * mu);
207
208 if (dim == 2)
209 {
210 res(0) = 1. / (8 * M_PI * mu * (1 - nu)) * ((3 - 4 * nu) * log(1. / r.norm()) + r(0) * r(0) / r.squaredNorm());
211 res(1) = 1. / (8 * M_PI * mu * (1 - nu)) * r(0) * r(1) / r.squaredNorm();
212 }
213 else if (dim == 3)
214 {
215 res(0) = 1. / (16 * M_PI * mu * (1 - nu)) * ((3 - 4 * nu) / r.norm() + r(0) * r(0) / r.norm() / r.squaredNorm());
216 res(1) = 1. / (16 * M_PI * mu * (1 - nu)) * r(0) * r(1) / r.norm() / r.squaredNorm();
217 res(2) = 1. / (16 * M_PI * mu * (1 - nu)) * r(0) * r(2) / r.norm() / r.squaredNorm();
218 }
219 else
220 assert(false);
221
222 return res;
223 }
