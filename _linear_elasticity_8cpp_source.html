21 Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 9, 1>
22 LinearElasticity::assemble(const ElementAssemblyValues &vals, const int i, const int j, const QuadratureVector &da) const
23 {
24 // mu ((gradi' gradj) Id + ((gradi gradj')') + lambda gradi *gradj';
25 const Eigen::MatrixXd &gradi = vals.basis_values[i].grad_t_m;
26 const Eigen::MatrixXd &gradj = vals.basis_values[j].grad_t_m;
27
28 Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 9, 1> res(size() * size());
29 res.setZero();
30
31 for (long k = 0; k < gradi.rows(); ++k)
32 {
33 Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 9, 1> res_k(size() * size());
34 // res_k.setZero();
35 const Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> outer = gradi.row(k).transpose() * gradj.row(k);
36 const double dot = gradi.row(k).dot(gradj.row(k));
37
38 double lambda, mu;
39 params_.lambda_mu(vals.quadrature.points.row(k), vals.val.row(k), vals.element_id, lambda, mu);
40
41 for (int ii = 0; ii < size(); ++ii)
42 {
43 for (int jj = 0; jj < size(); ++jj)
44 {
45 res_k(jj * size() + ii) = outer(ii * size() + jj) * mu + outer(jj * size() + ii) * lambda;
46 if (ii == jj)
47 res_k(jj * size() + ii) += mu * dot;
48 }
49 }
50 res += res_k * da(k);
51 }
52
53 return res;
54 }

98 T LinearElasticity::compute_energy_aux(const ElementAssemblyValues &vals, const Eigen::MatrixXd &displacement, const QuadratureVector &da) const
99 {
100 typedef Eigen::Matrix<T, Eigen::Dynamic, 1> AutoDiffVect;
101 typedef Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> AutoDiffGradMat;
102
103 assert(displacement.cols() == 1);
104
105 const int n_pts = da.size();
106
107 Eigen::Matrix<double, Eigen::Dynamic, 1> local_dispv(vals.basis_values.size() * size(), 1);
108 local_dispv.setZero();
109 for (size_t i = 0; i < vals.basis_values.size(); ++i)
110 {
111 const auto &bs = vals.basis_values[i];
112 for (size_t ii = 0; ii < bs.global.size(); ++ii)
113 {
114 for (int d = 0; d < size(); ++d)
115 {
116 local_dispv(i * size() + d) += bs.global[ii].val * displacement(bs.global[ii].index * size() + d);
117 }
118 }
119 }
120
121 DiffScalarBase::setVariableCount(local_dispv.rows());
122 AutoDiffVect local_disp(local_dispv.rows(), 1);
123 T energy = T(0.0);
124
125 const AutoDiffAllocator<T> allocate_auto_diff_scalar;
126
127 for (long i = 0; i < local_dispv.rows(); ++i)
128 {
129 local_disp(i) = allocate_auto_diff_scalar(i, local_dispv(i));
130 }
131
132 AutoDiffGradMat def_grad(size(), size());
133
134 for (long p = 0; p < n_pts; ++p)
135 {
136 for (long k = 0; k < def_grad.size(); ++k)
137 def_grad(k) = T(0);
138
139 for (size_t i = 0; i < vals.basis_values.size(); ++i)
140 {
141 const auto &bs = vals.basis_values[i];
142 const Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 3, 1> grad = bs.grad.row(p);
143 assert(grad.size() == size());
144
145 for (int d = 0; d < size(); ++d)
146 {
147 for (int c = 0; c < size(); ++c)
148 {
149 def_grad(d, c) += grad(c) * local_disp(i * size() + d);
150 }
151 }
152 }
153
154 AutoDiffGradMat jac_it(size(), size());
155 for (long k = 0; k < jac_it.size(); ++k)
156 jac_it(k) = T(vals.jac_it[p](k));
157 def_grad = def_grad * jac_it;
158
159 const AutoDiffGradMat strain = (def_grad + def_grad.transpose()) / T(2);
160
161 double lambda, mu;
162 params_.lambda_mu(vals.quadrature.points.row(p), vals.val.row(p), vals.element_id, lambda, mu);
163
164 const T val = mu * (strain.transpose() * strain).trace() + lambda / 2 * strain.trace() * strain.trace();
165
166 energy += val * da(p);
167 }
168 return energy;
169 }

255 Eigen::Matrix<AutodiffScalarGrad, Eigen::Dynamic, 1, 0, 3, 1> LinearElasticity::kernel(const int dim, const AutodiffGradPt &r) const
256 {
257 Eigen::Matrix<AutodiffScalarGrad, Eigen::Dynamic, 1, 0, 3, 1> res(dim);
258 assert(r.size() == dim);
259
260 double mu, nu, lambda;
261 //per body lame parameter dont work here!
262 params_.lambda_mu(0, 0, 0, 0, 0, 0, 0, lambda, mu);
263
264 // convert to nu!
265 nu = lambda / (lambda * (dim - 1) + 2 * mu);
266
267 if (dim == 2)
268 {
269 res(0) = 1. / (8 * M_PI * mu * (1 - nu)) * ((3 - 4 * nu) * log(1. / r.norm()) + r(0) * r(0) / r.squaredNorm());
270 res(1) = 1. / (8 * M_PI * mu * (1 - nu)) * r(0) * r(1) / r.squaredNorm();
271 }
272 else if (dim == 3)
273 {
274 res(0) = 1. / (16 * M_PI * mu * (1 - nu)) * ((3 - 4 * nu) / r.norm() + r(0) * r(0) / r.norm() / r.squaredNorm());
275 res(1) = 1. / (16 * M_PI * mu * (1 - nu)) * r(0) * r(1) / r.norm() / r.squaredNorm();
276 res(2) = 1. / (16 * M_PI * mu * (1 - nu)) * r(0) * r(2) / r.norm() / r.squaredNorm();
277 }
278 else
279 assert(false);
280
281 return res;
282 }
