237 // Distance from harmonic kernels to polygon boundary
238 double compute_epsilon(const Mesh2D &mesh, int e)
239 {
240 double area = 0;
241 const int n_edges = mesh.n_face_vertices(e);
242 Navigation::Index index = mesh.get_index_from_face(e);
243
244 for (int i = 0; i < n_edges; ++i)
245 {
246 Eigen::Matrix2d det_mat;
247 det_mat.row(0) = mesh.point(index.vertex);
248 det_mat.row(1) = mesh.point(mesh.switch_vertex(index).vertex);
249
250 area += det_mat.determinant();
251
252 index = mesh.next_around_face(index);
253 }
254 area = std::fabs(area);
255 // const double eps = use_harmonic ? (0.08*area) : 0;
256 const double eps = 0.08 * area;
257
258 return eps;
259 }
