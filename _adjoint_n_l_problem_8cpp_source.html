54 // A recursive function used by topologicalSort
55 void Graph::topologicalSortUtil(int v, vector<bool> &visited,
56 stack<int> &Stack)
57 {
58 // Mark the current node as visited.
59 visited[v] = true;
60
61 // Recur for all the vertices adjacent to this vertex
62 list<int>::iterator i;
63 for (i = adj[v].begin(); i != adj[v].end(); ++i)
64 if (!visited[*i])
65 topologicalSortUtil(*i, visited, Stack);
66
67 // Push current vertex to stack which stores result
68 Stack.push(v);
69 }

71 // The function to do Topological Sort. It uses recursive
72 // topologicalSortUtil()
73 vector<int> Graph::topologicalSort()
74 {
75 stack<int> Stack;
76
77 // Mark all the vertices as not visited
78 vector<bool> visited(V, false);
79
80 // Call the recursive helper function to store Topological
81 // Sort starting from all vertices one by one
82 for (int i = 0; i < V; i++)
83 if (visited[i] == false)
84 topologicalSortUtil(i, visited, Stack);
85
86 // Print contents of stack
87 vector<int> sorted;
88 while (Stack.empty() == false)
89 {
90 sorted.push_back(Stack.top());
91 Stack.pop();
92 }
