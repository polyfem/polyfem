379 void BoundedBiharmonicWeights2Dto3D::compute_faces_for_partial_vertices(const Eigen::MatrixXd &V, Eigen::MatrixXi &F) const
380 {
381 // The following implementation is maybe a bit wasteful, but is independent of state or surface selections
382 std::unordered_map<int, int> full_to_reduced_indices;
383
384 Eigen::MatrixXd BV;
385 BV.setZero(3, 2);
386 for (int i = 0; i < V.rows(); ++i)
387 {
388 if (i == 0)
389 {
390 BV.col(0) = V.row(i);
391 BV.col(1) = V.row(i);
392 }
393 else
394 {
395 for (int j = 0; j < 3; ++j)
396 {
397 BV(j, 0) = std::min(BV(j, 0), V(i, j));
398 BV(j, 1) = std::max(BV(j, 1), V(i, j));
399 }
400 }
401 }
402
403 Eigen::VectorXd bbox_width = (BV.col(1) - BV.col(0));
404 for (int j = 0; j < 3; ++j)
405 if (bbox_width(j) < 1e-12)
406 bbox_width(j) = 1e-3;
407
408 // Pad the bbox to make it conservative
409 BV.col(0) -= 0.05 * (BV.col(1) - BV.col(0));
410 BV.col(1) += 0.05 * (BV.col(1) - BV.col(0));
411
412 auto in_bbox = [&BV](const Eigen::VectorXd &x) {
413 bool in = true;
414 in &= (x(0) >= BV(0, 0)) && (x(0) <= BV(0, 1));
415 in &= (x(1) >= BV(1, 0)) && (x(1) <= BV(1, 1));
416 in &= (x(2) >= BV(2, 0)) && (x(0) <= BV(2, 1));
417 return in;
418 };
419
420 for (int i = 0; i < V_surface_.rows(); ++i)
421 for (int j = 0; j < V.rows(); ++j)
422 {
423 // if (in_bbox(V_surface_.row(i)))
424 if ((V_surface_.row(i) - V.row(j)).norm() < 1e-12)
425 full_to_reduced_indices[i] = j;
426 }
427
428 F.resize(0, 3);
429 for (int i = 0; i < F_surface_.rows(); ++i)
430 {
431 bool contains_face = true;
432 for (int j = 0; j < 3; ++j)
433 contains_face &= (full_to_reduced_indices.count(F_surface_(i, j)) == 1);
434
435 if (contains_face)
436 {
437 F.conservativeResize(F.rows() + 1, 3);
438 for (int j = 0; j < 3; ++j)
439 F(F.rows() - 1, j) = full_to_reduced_indices.at(F_surface_(i, j));
440 }
441 }
