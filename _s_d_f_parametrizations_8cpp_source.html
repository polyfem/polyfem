276 PeriodicMeshToMesh::PeriodicMeshToMesh(const Eigen::MatrixXd &V)
277 {
278 dim_ = V.cols();
279
280 for (auto &list : periodic_dependence)
281 list.clear();
282
283 assert(dim_ == V.cols());
284 const int n_verts = V.rows();
285
286 Eigen::VectorXd min = V.colwise().minCoeff();
287 Eigen::VectorXd max = V.colwise().maxCoeff();
288 Eigen::VectorXd scale_ = max - min;
289
290 n_periodic_dof_ = 0;
291 dependent_map.resize(n_verts);
292 dependent_map.setConstant(-1);
293
294 const double eps = 1e-4 * scale_.maxCoeff();
295 Eigen::VectorXi boundary_indices;
296 {
297 Eigen::VectorXi boundary_mask1 = ((V.rowwise() - min.transpose()).rowwise().minCoeff().array() < eps).select(Eigen::VectorXi::Ones(V.rows()), Eigen::VectorXi::Zero(V.rows()));
298 Eigen::VectorXi boundary_mask2 = ((V.rowwise() - max.transpose()).rowwise().maxCoeff().array() > -eps).select(Eigen::VectorXi::Ones(V.rows()), Eigen::VectorXi::Zero(V.rows()));
299 Eigen::VectorXi boundary_mask = boundary_mask1.array() + boundary_mask2.array();
300
301 boundary_indices.setZero(boundary_mask.sum());
302 for (int i = 0, j = 0; i < boundary_mask.size(); i++)
303 if (boundary_mask[i])
304 boundary_indices[j++] = i;
305 }
306
307 // find corresponding periodic boundary nodes
308 Eigen::MatrixXd V_boundary = V(boundary_indices, Eigen::all);
309 for (int d = 0; d < dim_; d++)
310 {
311 Eigen::VectorXi mask1 = (V_boundary.col(d).array() < min(d) + eps).select(Eigen::VectorXi::Ones(V_boundary.rows()), Eigen::VectorXi::Zero(V_boundary.rows()));
312 Eigen::VectorXi mask2 = (V_boundary.col(d).array() > max(d) - eps).select(Eigen::VectorXi::Ones(V_boundary.rows()), Eigen::VectorXi::Zero(V_boundary.rows()));
313
314 for (int i = 0; i < mask1.size(); i++)
315 {
316 if (!mask1(i))
317 continue;
318
319 bool found_target = false;
320 for (int j = 0; j < mask2.size(); j++)
321 {
322 if (!mask2(j))
323 continue;
324
325 RowVectorNd projected_diff = V_boundary.row(j) - V_boundary.row(i);
326 projected_diff(d) = 0;
327 if (projected_diff.norm() < eps)
328 {
329 dependent_map(boundary_indices[j]) = boundary_indices[i];
330 std::array<int, 2> pair = {{boundary_indices[i], boundary_indices[j]}};
331 periodic_dependence[d].insert(pair);
332 found_target = true;
333 break;
334 }
335 }
336 if (!found_target)
337 throw std::runtime_error("Periodic mesh failed to find corresponding nodes!");
338 }
339 }
340
341 // break dependency chains into direct dependency
342 for (int d = 0; d < dim_; d++)
343 for (int i = 0; i < dependent_map.size(); i++)
344 if (dependent_map(i) >= 0 && dependent_map(dependent_map(i)) >= 0)
345 dependent_map(i) = dependent_map(dependent_map(i));
346
347 Eigen::VectorXi reduce_map;
348 reduce_map.setZero(dependent_map.size());
349 for (int i = 0; i < dependent_map.size(); i++)
350 if (dependent_map(i) < 0)
351 reduce_map(i) = n_periodic_dof_++;
352 for (int i = 0; i < dependent_map.size(); i++)
353 if (dependent_map(i) >= 0)
354 reduce_map(i) = reduce_map(dependent_map(i));
355
356 dependent_map = std::move(reduce_map);
357 }
