128
129 for (int e = start; e < end; ++e)
130 {
131 assembler::ElementAssemblyValues &vals = local_storage.vals;
132 rhs_assembler_.ass_vals_cache().compute(e, rhs_assembler_.mesh().is_volume(), bases[e], gbases[e], vals);
133 assembler::ElementAssemblyValues &gvals = local_storage.gvals;
134 gvals.compute(e, rhs_assembler_.mesh().is_volume(), vals.quadrature.points, gbases[e], gbases[e]);

136 const quadrature::Quadrature &quadrature = vals.quadrature;

144 rhs_function *= -1;
145 for (int q = 0; q < vals.val.rows(); q++)
146 {
147 const double rho = density_(quadrature.points.row(q), vals.val.row(q), e);
148 rhs_function.row(q) *= rho;
149 }
150
151 for (int q = 0; q < local_storage.da.size(); ++q)
152 {
153 const double value = p.row(q).dot(rhs_function.row(q)) * local_storage.da(q);
154 for (auto &v : gvals.basis_values)
155 {
156 local_storage.vec.block(v.global[0].index * dim, 0, dim, 1) += value * v.grad_t_m.row(q).transpose();

158 }

193 for (int n = 0; n < gvals.jac_it.size(); ++n)
194 {
195 Eigen::MatrixXd trafo = gvals.jac_it[n].inverse();

197 Eigen::MatrixXd deform_mat(dim, dim);
198 deform_mat.setZero();
199 Eigen::MatrixXd jac_mat(dim, gvals.basis_values.size());
200 int b_idx = 0;
201 for (const auto &b : gvals.basis_values)
202 {
203 jac_mat.col(b_idx++) = b.grad.row(n);
204
205 for (const auto &g : b.global)
206 for (int d = 0; d < dim; ++d)
207 deform_mat.row(d) += x(g.index * dim + d) * b.grad.row(n);
208 }
209
210 trafo += deform_mat;
211 trafo = trafo.inverse();
212
213 Eigen::VectorXd displaced_normal = normals.row(n) * trafo;
214 normals.row(n) = displaced_normal / displaced_normal.norm();

216 std::vector<Eigen::MatrixXd> grad;
217 {
218 Eigen::MatrixXd vec = -(jac_mat.transpose() * trafo * reference_normals.row(n).transpose());
219 // Gradient of the displaced normal computation
220 for (int k = 0; k < dim; ++k)
221 {
222 Eigen::MatrixXd grad_i(jac_mat.rows(), jac_mat.cols());
223 grad_i.setZero();

225 for (int l = 0; l < jac_mat.cols(); ++l)

227 grad.push_back(grad_i);
228 }
229 }
230
231 {
232 Eigen::MatrixXd normalization_chain_rule = (normals.row(n).transpose() * normals.row(n));
233 normalization_chain_rule = Eigen::MatrixXd::Identity(dim, dim) - normalization_chain_rule;
234 normalization_chain_rule /= displaced_normal.norm();
235
236 Eigen::VectorXd vec(dim);
237 b_idx = 0;
238 for (const auto &b : gvals.basis_values)
239 {
240 for (int d = 0; d < dim; ++d)

242 for (int k = 0; k < dim; ++k)

247 }
248 b_idx++;
249 }
250 }

253 }

293 }
294 }
295
296 // integrate j * div(gbases) over the whole boundary
297 for (int d = 0; d < dim; d++)

299 assert(v.global.size() == 1);

305
