286 template <typename T>
287 T NeoHookeanElasticity::compute_energy_aux(const ElementAssemblyValues &vals, const Eigen::MatrixXd &displacement, const QuadratureVector &da) const
288 {
289 typedef Eigen::Matrix<T, Eigen::Dynamic, 1> AutoDiffVect;
290 typedef Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> AutoDiffGradMat;
291
292 assert(displacement.cols() == 1);
293
294 const int n_pts = da.size();
295
296 Eigen::Matrix<double, Eigen::Dynamic, 1> local_dispv(vals.basis_values.size() * size(), 1);
297 local_dispv.setZero();
298 for (size_t i = 0; i < vals.basis_values.size(); ++i)
299 {
300 const auto &bs = vals.basis_values[i];
301 for (size_t ii = 0; ii < bs.global.size(); ++ii)
302 {
303 for (int d = 0; d < size(); ++d)
304 {
305 local_dispv(i * size() + d) += bs.global[ii].val * displacement(bs.global[ii].index * size() + d);
306 }
307 }
308 }
309
310 DiffScalarBase::setVariableCount(local_dispv.rows());
311 AutoDiffVect local_disp(local_dispv.rows(), 1);
312 T energy = T(0.0);
313
314 const AutoDiffAllocator<T> allocate_auto_diff_scalar;
315
316 for (long i = 0; i < local_dispv.rows(); ++i)
317 {
318 local_disp(i) = allocate_auto_diff_scalar(i, local_dispv(i));
319 }
320
321 AutoDiffGradMat def_grad(size(), size());
322
323 for (long p = 0; p < n_pts; ++p)
324 {
325 for (long k = 0; k < def_grad.size(); ++k)
326 def_grad(k) = T(0);
327
328 for (size_t i = 0; i < vals.basis_values.size(); ++i)
329 {
330 const auto &bs = vals.basis_values[i];
331 const Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 3, 1> grad = bs.grad.row(p);
332 assert(grad.size() == size());
333
334 for (int d = 0; d < size(); ++d)
335 {
336 for (int c = 0; c < size(); ++c)
337 {
338 def_grad(d, c) += grad(c) * local_disp(i * size() + d);
339 }
340 }
341 }
342
343 AutoDiffGradMat jac_it(size(), size());
344 for (long k = 0; k < jac_it.size(); ++k)
345 jac_it(k) = T(vals.jac_it[p](k));
346 def_grad = def_grad * jac_it;
347
348 //Id + grad d
349 for (int d = 0; d < size(); ++d)
350 def_grad(d, d) += T(1);
351
352 double lambda, mu;
353 params_.lambda_mu(vals.quadrature.points.row(p), vals.val.row(p), vals.element_id, lambda, mu);
354
355 const T log_det_j = log(polyfem::utils::determinant(def_grad));
356 const T val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
357
358 energy += val * da(p);
359 }
360 return energy;
361 }

364 Eigen::Matrix<double, dim, dim> hat(const Eigen::Matrix<double, dim, 1> &x)
365 {
366
367 Eigen::Matrix<double, dim, dim> prod;
368 prod.setZero();
369
370 prod(0, 1) = -x(2);
371 prod(0, 2) = x(1);
372 prod(1, 0) = x(2);
373 prod(1, 2) = -x(0);
374 prod(2, 0) = -x(1);
375 prod(2, 1) = x(0);
376

394 template <int n_basis, int dim>
395 void NeoHookeanElasticity::compute_energy_aux_gradient_fast(const ElementAssemblyValues &vals, const Eigen::MatrixXd &displacement, const QuadratureVector &da, Eigen::Matrix<double, Eigen::Dynamic, 1> &G_flattened) const
396 {
397
398 assert(displacement.cols() == 1);
399
400 const int n_pts = da.size();
401
402 Eigen::Matrix<double, n_basis, dim> local_disp(vals.basis_values.size(), size());
403 local_disp.setZero();
404 for (size_t i = 0; i < vals.basis_values.size(); ++i)
405 {
406 const auto &bs = vals.basis_values[i];
407 for (size_t ii = 0; ii < bs.global.size(); ++ii)
408 {
409 for (int d = 0; d < size(); ++d)
410 {
411 local_disp(i, d) += bs.global[ii].val * displacement(bs.global[ii].index * size() + d);
412 }
413 }
414 }
415
416 Eigen::Matrix<double, dim, dim> def_grad(size(), size());
417
418 Eigen::Matrix<double, n_basis, dim> G(vals.basis_values.size(), size());
419 G.setZero();
420
421 for (long p = 0; p < n_pts; ++p)
422 {
423 Eigen::Matrix<double, n_basis, dim> grad(vals.basis_values.size(), size());
424
425 for (size_t i = 0; i < vals.basis_values.size(); ++i)
426 {
427 grad.row(i) = vals.basis_values[i].grad.row(p);
428 }
429
430 Eigen::Matrix<double, dim, dim> jac_it = vals.jac_it[p];
431
432 //Id + grad d
433 def_grad = local_disp.transpose() * grad * jac_it + Eigen::Matrix<double, dim, dim>::Identity(size(), size());
434
435 const double J = def_grad.determinant();
436 const double log_det_j = log(J);
437
438 Eigen::Matrix<double, dim, dim> delJ_delF(size(), size());
439 delJ_delF.setZero();
440
441 if (dim == 2)
442 {
443
444 delJ_delF(0, 0) = def_grad(1, 1);
445 delJ_delF(0, 1) = -def_grad(1, 0);
446 delJ_delF(1, 0) = -def_grad(0, 1);
447 delJ_delF(1, 1) = def_grad(0, 0);
448 }
449
450 else if (dim == 3)
451 {
452
453 Eigen::Matrix<double, dim, 1> u(def_grad.rows());
454 Eigen::Matrix<double, dim, 1> v(def_grad.rows());
455 Eigen::Matrix<double, dim, 1> w(def_grad.rows());
456
457 u = def_grad.col(0);
458 v = def_grad.col(1);
459 w = def_grad.col(2);
460
461 delJ_delF.col(0) = cross<dim>(v, w);
462 delJ_delF.col(1) = cross<dim>(w, u);
463 delJ_delF.col(2) = cross<dim>(u, v);
464 }
465
466 double lambda, mu;
467 params_.lambda_mu(vals.quadrature.points.row(p), vals.val.row(p), vals.element_id, lambda, mu);
468
469 Eigen::Matrix<double, n_basis, dim> delF_delU = grad * jac_it;
470
471 Eigen::Matrix<double, dim, dim> gradient_temp = mu * def_grad - mu * (1 / J) * delJ_delF + lambda * log_det_j * (1 / J) * delJ_delF;
472 Eigen::Matrix<double, n_basis, dim> gradient = delF_delU * gradient_temp.transpose();
473
474 double val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
475
476 G.noalias() += gradient * da(p);
477 }
478
479 Eigen::Matrix<double, dim, n_basis> G_T = G.transpose();
480
481 constexpr int N = (n_basis == Eigen::Dynamic) ? Eigen::Dynamic : n_basis * dim;
482 Eigen::Matrix<double, N, 1> temp(Eigen::Map<Eigen::Matrix<double, N, 1>>(G_T.data(), G_T.size()));
483 G_flattened = temp;
484 }

529 Eigen::Matrix<double, dim * dim, dim * dim> del2J_delF2(size() * size(), size() * size());
530 del2J_delF2.setZero();
531
532 if (dim == 2)
533 {
534 delJ_delF(0, 0) = def_grad(1, 1);
535 delJ_delF(0, 1) = -def_grad(1, 0);
536 delJ_delF(1, 0) = -def_grad(0, 1);
537 delJ_delF(1, 1) = def_grad(0, 0);
538
539 del2J_delF2(0, 3) = 1;
540 del2J_delF2(1, 2) = -1;
541 del2J_delF2(2, 1) = -1;
542 del2J_delF2(3, 0) = 1;
543 }
544 else if (size() == 3)
545 {
546 Eigen::Matrix<double, dim, 1> u(def_grad.rows());
547 Eigen::Matrix<double, dim, 1> v(def_grad.rows());
548 Eigen::Matrix<double, dim, 1> w(def_grad.rows());
549
550 u = def_grad.col(0);
551 v = def_grad.col(1);
552 w = def_grad.col(2);
553
554 delJ_delF.col(0) = cross<dim>(v, w);
555 delJ_delF.col(1) = cross<dim>(w, u);
556 delJ_delF.col(2) = cross<dim>(u, v);
557
558 del2J_delF2.template block<dim, dim>(0, 6) = hat<dim>(v);
559 del2J_delF2.template block<dim, dim>(6, 0) = -hat<dim>(v);
560 del2J_delF2.template block<dim, dim>(0, 3) = -hat<dim>(w);
561 del2J_delF2.template block<dim, dim>(3, 0) = hat<dim>(w);
562 del2J_delF2.template block<dim, dim>(3, 6) = -hat<dim>(u);
563 del2J_delF2.template block<dim, dim>(6, 3) = hat<dim>(u);
564 }
565
566 double lambda, mu;
567 params_.lambda_mu(vals.quadrature.points.row(p), vals.val.row(p), vals.element_id, lambda, mu);
568
569 Eigen::Matrix<double, dim * dim, dim *dim> id = Eigen::Matrix<double, dim * dim, dim * dim>::Identity(size() * size(), size() * size());
570
571 Eigen::Matrix<double, dim * dim, 1> g_j = Eigen::Map<const Eigen::Matrix<double, dim * dim, 1>>(delJ_delF.data(), delJ_delF.size());
572
573 Eigen::Matrix<double, dim * dim, dim *dim> hessian_temp = (mu * id) + (((mu + lambda * (1 - log_det_j)) / (J * J)) * (g_j * g_j.transpose())) + (((lambda * log_det_j - mu) / (J)) * del2J_delF2);
574
575 Eigen::Matrix<double, dim * dim, N> delF_delU_tensor(jac_it.size(), grad.size());
576
577 for (size_t i = 0; i < local_disp.rows(); ++i)
578 {
579 for (size_t j = 0; j < local_disp.cols(); ++j)
580 {
581 Eigen::Matrix<double, dim, dim> temp(size(), size());
582 temp.setZero();
583 temp.row(j) = grad.row(i);
584 temp = temp * jac_it;
585 Eigen::Matrix<double, dim * dim, 1> temp_flattened(Eigen::Map<Eigen::Matrix<double, dim * dim, 1>>(temp.data(), temp.size()));
586 delF_delU_tensor.col(i * size() + j) = temp_flattened;
587 }
588 }
589
590 Eigen::Matrix<double, N, N> hessian = delF_delU_tensor.transpose() * hessian_temp * delF_delU_tensor;
591
592 double val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
593
594 H += hessian * da(p);
595 }
