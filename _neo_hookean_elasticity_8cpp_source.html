336 // Compute ∫ ½μ (tr(FᵀF) - 3 - 2ln(J)) + ½λ ln²(J) du
337 template <typename T, int n_basis, int dim>
338 T NeoHookeanElasticity::compute_energy_aux(const NonLinearAssemblerData &data) const
339 {
340 if constexpr (std::is_same_v<T, double>)
341 {
342 Eigen::Matrix<T, n_basis, dim> local_disp(data.vals.basis_values.size(), size());
343 local_disp.setZero();
344 for (size_t i = 0; i < data.vals.basis_values.size(); ++i)
345 {
346 const auto &bs = data.vals.basis_values[i];
347 for (size_t ii = 0; ii < bs.global.size(); ++ii)
348 {
349 for (int d = 0; d < dim; ++d)
350 {
351 local_disp(i, d) += bs.global[ii].val * data.x(bs.global[ii].index * size() + d);
352 }
353 }
354 }
355
356 Eigen::VectorXd jacs;
357 if (use_robust_jacobian)
358 jacs = data.vals.eval_deformed_jacobian_determinant(data.x);
359
360 Eigen::Matrix<T, dim, dim> def_grad(size(), size());
361
362 T energy = T(0.0);
363
364 const int n_pts = data.da.size();
365 for (long p = 0; p < n_pts; ++p)
366 {
367 Eigen::Matrix<T, n_basis, dim> grad(data.vals.basis_values.size(), size());
368
369 for (size_t i = 0; i < data.vals.basis_values.size(); ++i)
370 {
371 grad.row(i) = data.vals.basis_values[i].grad.row(p);
372 }
373
374 const Eigen::Matrix<T, dim, dim> jac_it = data.vals.jac_it[p];
375 // Id + grad d
376 def_grad = (local_disp.transpose() * grad) * jac_it + Eigen::Matrix<T, dim, dim>::Identity(size(), size());
377
378 double lambda, mu;
379 params_.lambda_mu(data.vals.quadrature.points.row(p), data.vals.val.row(p), data.t, data.vals.element_id, lambda, mu);
380
381 const T J = use_robust_jacobian ? jacs(p) * jac_it.determinant() : def_grad.determinant();
382 const T log_det_j = log(J);
383 const T val = mu / 2 * (def_grad.squaredNorm() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
384 energy += val * data.da(p);
385 }
386 return energy;
387 }
388 else
389 {
390 typedef Eigen::Matrix<T, Eigen::Dynamic, 1> AutoDiffVect;
391 typedef Eigen::Matrix<T, dim, dim> AutoDiffGradMat;
392
393 AutoDiffVect local_disp;
394 get_local_disp(data, size(), local_disp);
395
396 AutoDiffGradMat def_grad(size(), size());
397
398 T energy = T(0.0);
399
400 const int n_pts = data.da.size();
401 for (long p = 0; p < n_pts; ++p)
402 {
403 compute_disp_grad_at_quad(data, local_disp, p, size(), def_grad);
404
405 // Id + grad d
406 for (int d = 0; d < dim; ++d)
407 def_grad(d, d) += T(1);
408
409 double lambda, mu;
410 params_.lambda_mu(data.vals.quadrature.points.row(p), data.vals.val.row(p), data.t, data.vals.element_id, lambda, mu);
411
412 const T log_det_j = log(polyfem::utils::determinant(def_grad));
413 const T val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
414
415 energy += val * data.da(p);
416 }
417 return energy;
418 }

421 template <int dim>
422 Eigen::Matrix<double, dim, dim> hat(const Eigen::Matrix<double, dim, 1> &x)
423 {
424
425 Eigen::Matrix<double, dim, dim> prod;
426 prod.setZero();
427
428 prod(0, 1) = -x(2);
429 prod(0, 2) = x(1);
430 prod(1, 0) = x(2);
431 prod(1, 2) = -x(0);
432 prod(2, 0) = -x(1);
433 prod(2, 1) = x(0);
434
435 return prod;
436 }

438 template <int dim>
439 Eigen::Matrix<double, dim, 1> cross(const Eigen::Matrix<double, dim, 1> &x, const Eigen::Matrix<double, dim, 1> &y)
440 {
441
442 Eigen::Matrix<double, dim, 1> z;
443 z.setZero();
444
445 z(0) = x(1) * y(2) - x(2) * y(1);
446 z(1) = x(2) * y(0) - x(0) * y(2);
447 z(2) = x(0) * y(1) - x(1) * y(0);
448
449 return z;

453 void NeoHookeanElasticity::compute_energy_aux_gradient_fast(const NonLinearAssemblerData &data, Eigen::Matrix<double, Eigen::Dynamic, 1> &G_flattened) const
454 {
455 assert(data.x.cols() == 1);
456
457 const int n_pts = data.da.size();
458
459 Eigen::Matrix<double, n_basis, dim> local_disp(data.vals.basis_values.size(), size());
460 local_disp.setZero();
461 for (size_t i = 0; i < data.vals.basis_values.size(); ++i)
462 {
463 const auto &bs = data.vals.basis_values[i];
464 for (size_t ii = 0; ii < bs.global.size(); ++ii)
465 {
466 for (int d = 0; d < dim; ++d)
467 {
468 local_disp(i, d) += bs.global[ii].val * data.x(bs.global[ii].index * size() + d);
469 }
470 }
471 }
472
473 Eigen::VectorXd jacs;
474 if (use_robust_jacobian)
475 jacs = data.vals.eval_deformed_jacobian_determinant(data.x);
476
477 Eigen::Matrix<double, dim, dim> def_grad(size(), size());
478
479 Eigen::Matrix<double, n_basis, dim> G(data.vals.basis_values.size(), size());
480 G.setZero();
481
482 for (long p = 0; p < n_pts; ++p)
483 {
484 Eigen::Matrix<double, n_basis, dim> grad(data.vals.basis_values.size(), size());
485
486 for (size_t i = 0; i < data.vals.basis_values.size(); ++i)
487 {
488 grad.row(i) = data.vals.basis_values[i].grad.row(p);
489 }
490
491 Eigen::Matrix<double, dim, dim> jac_it = data.vals.jac_it[p];
492 Eigen::Matrix<double, n_basis, dim> delF_delU = grad * jac_it;
493
494 // Id + grad d
495 def_grad = local_disp.transpose() * delF_delU + Eigen::Matrix<double, dim, dim>::Identity(size(), size());
496
497 const double J = use_robust_jacobian ? jacs(p) * jac_it.determinant() : def_grad.determinant();
498 const double log_det_j = log(J);
499
500 Eigen::Matrix<double, dim, dim> delJ_delF(size(), size());
501 delJ_delF.setZero();
502
503 if (dim == 2)
504 {
505
506 delJ_delF(0, 0) = def_grad(1, 1);
507 delJ_delF(0, 1) = -def_grad(1, 0);
508 delJ_delF(1, 0) = -def_grad(0, 1);
509 delJ_delF(1, 1) = def_grad(0, 0);
510 }
511
512 else if (dim == 3)
513 {
514
515 Eigen::Matrix<double, dim, 1> u(def_grad.rows());
516 Eigen::Matrix<double, dim, 1> v(def_grad.rows());
517 Eigen::Matrix<double, dim, 1> w(def_grad.rows());
518
519 u = def_grad.col(0);
520 v = def_grad.col(1);
521 w = def_grad.col(2);
522
523 delJ_delF.col(0) = cross<dim>(v, w);
524 delJ_delF.col(1) = cross<dim>(w, u);
525 delJ_delF.col(2) = cross<dim>(u, v);
526 }
527
528 double lambda, mu;
529 params_.lambda_mu(data.vals.quadrature.points.row(p), data.vals.val.row(p), data.t, data.vals.element_id, lambda, mu);
530
531 Eigen::Matrix<double, dim, dim> gradient_temp = mu * def_grad - mu * (1 / J) * delJ_delF + lambda * log_det_j * (1 / J) * delJ_delF;
532 Eigen::Matrix<double, n_basis, dim> gradient = delF_delU * gradient_temp.transpose();
533
534 double val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
535
536 G.noalias() += gradient * data.da(p);
537 }
538
539 Eigen::Matrix<double, dim, n_basis> G_T = G.transpose();
540
541 constexpr int N = (n_basis == Eigen::Dynamic) ? Eigen::Dynamic : n_basis * dim;
542 Eigen::Matrix<double, N, 1> temp(Eigen::Map<Eigen::Matrix<double, N, 1>>(G_T.data(), G_T.size()));
543 G_flattened = temp;

617 delJ_delF.col(1) = cross<dim>(w, u);
618 delJ_delF.col(2) = cross<dim>(u, v);

633 Eigen::Matrix<double, dim * dim, 1> g_j = Eigen::Map<const Eigen::Matrix<double, dim * dim, 1>>(delJ_delF.data(), delJ_delF.size());
634
635 Eigen::Matrix<double, dim * dim, dim * dim> hessian_temp = (mu * id) + (((mu + lambda * (1 - log_det_j)) / (J * J)) * (g_j * g_j.transpose())) + (((lambda * log_det_j - mu) / (J)) * del2J_delF2);
636
637 Eigen::Matrix<double, dim * dim, N> delF_delU_tensor = Eigen::Matrix<double, dim * dim, N>::Zero(jac_it.size(), grad.size());
638
639 for (size_t i = 0; i < local_disp.rows(); ++i)
640 {
641 for (size_t j = 0; j < local_disp.cols(); ++j)
642 {
643 Eigen::Matrix<double, dim, dim> temp(size(), size());
644 temp.setZero();
645 temp.row(j) = delF_delU.row(i);
646 Eigen::Matrix<double, dim * dim, 1> temp_flattened(Eigen::Map<Eigen::Matrix<double, dim * dim, 1>>(temp.data(), temp.size()));
647 delF_delU_tensor.col(i * size() + j) = temp_flattened;
648 }
649 }
650
651 Eigen::Matrix<double, N, N> hessian = delF_delU_tensor.transpose() * hessian_temp * delF_delU_tensor;
652
653 double val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
654
655 H += hessian * data.da(p);
656 }
657 }

659 void NeoHookeanElasticity::compute_stress_grad_multiply_mat(
660 const OptAssemblerData &data,
661 const Eigen::MatrixXd &mat,
662 Eigen::MatrixXd &stress,
663 Eigen::MatrixXd &result) const
664 {
665 const double t = data.t;
666 const int el_id = data.el_id;
667 const Eigen::MatrixXd &local_pts = data.local_pts;
668 const Eigen::MatrixXd &global_pts = data.global_pts;
669 const Eigen::MatrixXd &grad_u_i = data.grad_u_i;
670
671 double lambda, mu;
672 params_.lambda_mu(local_pts, global_pts, t, el_id, lambda, mu);
673
674 Eigen::MatrixXd def_grad = Eigen::MatrixXd::Identity(grad_u_i.rows(), grad_u_i.cols()) + grad_u_i;
675 Eigen::MatrixXd FmT = def_grad.inverse().transpose();
676
677 stress = mu * (def_grad - FmT) + lambda * std::log(def_grad.determinant()) * FmT;
678 result = mu * mat + FmT * mat.transpose() * FmT * (mu - lambda * std::log(def_grad.determinant())) + lambda * (FmT.array() * mat.array()).sum() * FmT;
679 }
