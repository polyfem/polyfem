53 {
54 val = Eigen::MatrixXd::Zero(pts.rows(), mesh.dimension());
55
56 double alpha = n_turns_ * t * 2 * M_PI;
57 Eigen::Matrix2d rot;
58 rot << cos(alpha), -sin(alpha),

60

354 template <typename T>
355 void compute_singularity(const T &x, const T &y, Eigen::Matrix<T, -1, 1> &res, const double a, const double mu, const double nu, const double lambda, const double t)
356 {
357 double b1, b2, b3, b4;
358 T r = sqrt(x * x + y * y);
359 T phi = atan2(y, x);
360 T ur, ut;
361
362 b1 = sin(M_PI * a / 2) * ((a - 1) * mu + (1 - 2 * nu) * lambda) * (a - 4 * nu + 3) / cos(M_PI * a / 2) / ((a - 4 * nu + 2) * mu + (1 - 2 * nu) * lambda) / (a + 1);
363 b2 = (a + 4 * nu - 3) / (a + 1);
364 b3 = -sin(M_PI * a / 2) * ((a - 1) * mu + (1 - 2 * nu) * lambda) / cos(M_PI * a / 2) / ((a - 4 * nu + 2) * mu + (1 - 2 * nu) * lambda);
365 b4 = -1;
366
367 ur = -1 / mu * pow(r, a) * (b4 * (a + (4 * nu) - 3) * cos((a - 1) * phi) + b3 * (a + (4 * nu) - 3) * sin((a - 1) * phi) + (b2 * cos((a + 1) * phi) + b1 * sin((a + 1) * phi)) * (a + 1)) / 2;
368 ut = -1 / mu * pow(r, a) * (b3 * (a - (4 * nu) + 3) * cos((a - 1) * phi) - b4 * (a - (4 * nu) + 3) * sin((a - 1) * phi) + (b1 * cos((a + 1) * phi) - b2 * sin((a + 1) * phi)) * (a + 1)) / 2;
369
370 res(0) = 79.17 * t * ur * cos(ut);
371 res(1) = 79.17 * t * ur * sin(ut);
372 }

374 template <typename T>
375 Eigen::Matrix<T, -1, 1> function_cantilever(T x, T y, const double t, const double delta, const double E, const double nu, const int dim, const double L, const double D)
376 {
377 Eigen::Matrix<T, -1, 1> res, res1, res2;
378 res.setZero(dim, 1);
379 res1.setZero(dim, 1);
380 res2.setZero(dim, 1);
381 const double lambda = (E * nu) / (1 + nu) / (1 - (dim - 1) * nu);
382 const double mu = E / (2 * (1 + nu));
383
384 // Boundary condition related formulas in Rossel's paper
385 double a = 0.71117293;
386 double P = 100; // force
387 double I = D * D * D / 12; // second moment of area of the cross-section
388
389 // Add 2 singular solutions for an infinite wedge with Dirichlet/Neumann sides
390 compute_singularity(y + delta, x + delta, res1, a, mu, nu, lambda, t);
391 compute_singularity(D - y + delta, x + delta, res2, a, mu, nu, lambda, t);
392 res = res1 + res2;
393
394 // Formulas in Charles's paper
395 res(0) += t * P * (y - D/2) / (6 * E * I) * ((6 * L - 3 * x) * x + (2 + nu) * ((y - D/2) * (y - D/2) - D * D / 4)) / 3;
396 res(1) += -t * P / (6 * E * I) * (3 * nu * (y - D/2) * (y - D/2) * (L - x) + (4 + 5 * nu) * D * D * x / 4 + (3 * L - x) * x * x) / 3;
397
398 return res;
399 }

676 {

737 void ElasticCantileverExact::neumann_bc(const mesh::Mesh &mesh, const Eigen::MatrixXi &global_ids, const Eigen::MatrixXd &uv, const Eigen::MatrixXd &pts, const Eigen::MatrixXd &normals, const double t, Eigen::MatrixXd &val) const
738 {
739 val = Eigen::MatrixXd::Zero(pts.rows(), mesh.dimension());
740
741 const double lambda = (E * nu) / (1.0 + nu) / (1.0 - (mesh.dimension() - 1.0) * nu);
742 const double mu = E / 2.0 / (1.0 + nu);
743
744 for (long i = 0; i < pts.rows(); ++i)
745 {
746 const int id = mesh.get_boundary_id(global_ids(i));
747
748 DiffScalarBase::setVariableCount(pts.cols());
749 AutodiffHessianPt pt(pts.cols());
750 Eigen::VectorXd neumann_bc_normal = normals.row(i);
751 Eigen::MatrixXd sigma;
752
753 for (long d = 0; d < pts.cols(); ++d)
754 pt(d) = AutodiffScalarHessian(d, pts(i, d));
755
756 AutodiffHessianPt res = eval_fun(pt, t);
757 Eigen::MatrixXd grad_u(mesh.dimension(), mesh.dimension());
758 for (int d1 = 0; d1 < mesh.dimension(); d1++) {
759 for (int d2 = 0; d2 < mesh.dimension(); d2++) {
760 grad_u(d1, d2) = res(d1).getGradient()(d2);
761 }
762 }
763
764 if (formulation == "LinearElasticity") {
765 sigma = mu * (grad_u + grad_u.transpose()) + lambda * grad_u.trace() * Eigen::MatrixXd::Identity(mesh.dimension(), mesh.dimension());
766 } else if (formulation == "NeoHookean") {
767 Eigen::MatrixXd def_grad = Eigen::MatrixXd::Identity(grad_u.rows(), grad_u.cols()) + grad_u;
768 Eigen::MatrixXd FmT = def_grad.inverse().transpose();
769 sigma = mu * (def_grad - FmT) + lambda * std::log(def_grad.determinant()) * FmT;
770 } else {
771 throw std::invalid_argument("No specified formulation in params!");
772 assert(false);
773 }
774
775 val.row(i) = sigma * neumann_bc_normal;
776 }
777 }
