315 const int prev_dim = vertices.cols();
316 vertices.conservativeResize(vertices.rows(), dim);

318 vertices.rightCols(dim - prev_dim).setZero();

320 // --------------------------------------------------------------------
321
322 {

330 VectorNd b;
331 construct_affine_transformation(unit_scale, j_mesh["transformation"], mesh_dimensions, A, b);
332 vertices = vertices * A.transpose();
333 vertices.rowwise() += b.transpose();
334 }
335
336 std::string extract = j_mesh["extract"];
337 // Default: "volume" clashes with defaults for non obstacle, here assume volume is suface

342 {
343 // points -> vertices (drop edges and faces)
344 codim_edges.resize(0, 0);
345 faces.resize(0, 0);

372 // volume -> undefined
373 log_and_throw_error("Volumetric elements not supported for collision obstacles!");
374 }
375
376 if (j_mesh["n_refs"].get<int>() != 0)
377 {
378 log_and_throw_error("Option \"n_refs\" in obstacles not implement yet!");
379 if (j_mesh["advanced"]["refinement_location"].get<double>() != 0.5)

591 void construct_affine_transformation(
592 const double unit_scale,
593 const json &transform,
594 const VectorNd &mesh_dimensions,
595 MatrixNd &A,
596 VectorNd &b)
597 {
598 const int dim = mesh_dimensions.size();
599
600 // -----
601 // Scale
602 // -----
603
604 RowVectorNd scale;
605 if (transform["dimensions"].is_array()) // default is nullptr
606 {
607 VectorNd modified_dimensions =
608 (mesh_dimensions.array() == 0).select(1, mesh_dimensions);
609
610 scale = transform["dimensions"];
611 const int scale_size = scale.size();
612 scale.conservativeResize(dim);
613 if (scale_size < dim)
614 scale.tail(dim - scale_size).setZero();
615
616 scale.array() /= modified_dimensions.array();
617 }
618 else if (transform["scale"].is_number())
619 {
620 scale.setConstant(dim, transform["scale"].get<double>());
621 }
622 else
623 {
624 assert(transform["scale"].is_array());
625 scale = transform["scale"];
626 const int scale_size = scale.size();
627 scale.conservativeResize(dim);
628 if (scale_size < dim)
629 scale.tail(dim - scale_size).setZero();
630
631 if (scale_size == 0)
632 scale.setOnes();
633 }
634
635 A = (unit_scale * scale).asDiagonal();
636
637 // ------
638 // Rotate
639 // ------
640
641 // Rotate around the models origin NOT the bodies center of mass.
642 // We could expose this choice as a "rotate_around" field.
643 MatrixNd R = MatrixNd::Identity(dim, dim);
644 if (!transform["rotation"].is_null())
645 {
646 if (dim == 2)
647 {
648 if (transform["rotation"].is_number())
649 R = Eigen::Rotation2Dd(deg2rad(transform["rotation"].get<double>()))
650 .toRotationMatrix();
651 else if (!transform["rotation"].is_array() || !transform["rotation"].empty())
652 log_and_throw_error("Invalid 2D rotation; 2D rotations can only be a angle in degrees.");
653 }
654 else if (dim == 3)
655 {
656 R = to_rotation_matrix(transform["rotation"], transform["rotation_mode"]);
657 }
658 }
659
660 A = R * A; // Scale first, then rotate
661
662 // ---------
663 // Translate
664 // ---------
665
666 b = transform["translation"];
667 const int translation_size = b.size();
668 b.conservativeResize(dim);
669 if (translation_size < dim)
670 b.tail(dim - translation_size).setZero();
671 }
