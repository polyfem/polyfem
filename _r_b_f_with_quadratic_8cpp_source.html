237void RBFWithQuadratic::bases_grads(const int axis, const Eigen::MatrixXd &samples, Eigen::MatrixXd &val) const
238{
239 const int num_kernels = centers_.rows();
240 const int dim = (is_volume() ? 3 : 2);
241
242 // Compute xA
243 Eigen::MatrixXd A_prime(samples.rows(), num_kernels + 1 + dim + dim * (dim + 1) / 2);
244 A_prime.setZero();
245
246 for (int j = 0; j < num_kernels; ++j)
247 {
248 A_prime.col(j) = (samples.rowwise() - centers_.row(j)).rowwise().norm().unaryExpr([this](double x) { return kernel_prime(is_volume(), x) / x; });
249 A_prime.col(j) = (samples.col(axis).array() - centers_(j, axis)) * A_prime.col(j).array();
250 }
251 // Linear terms
252 A_prime.middleCols(num_kernels + 1 + axis, 1).setOnes();
253 // Mixed terms
254 if (dim == 2)
255 {
256 A_prime.col(num_kernels + 1 + dim) = samples.col(1 - axis);
257 }
258 else
259 {
260 A_prime.col(num_kernels + 1 + dim + axis) = samples.col((axis + 1) % dim);
261 A_prime.col(num_kernels + 1 + dim + (axis + 2) % dim) = samples.col((axis + 2) % dim);
262 }
263 // Quadratic terms
264 A_prime.rightCols(dim).col(axis) = 2.0 * samples.col(axis);
265
266 // Apply weights
267 val = A_prime * weights_;
268}

271//
272// For each FEM basis that is nonzero on the element E, we want to
273// solve the least square system A w = rhs, where:
274//
275// _k(pi) ... 1 xi yi xi*yi xi^2 yi^2
276// A = ^{#S x (#K+1+dim+dim*(dim+1)/2)}
277//
278//
279// ^
280// w = w_k ... a00 a10 a01 a11 a20 a02 ^{#K+1+dim+dim*(dim+1)/2}
281//
282// - A is the RBF kernels evaluated over the collocation points (#S)
283// - b is the expected value of the basis sampled on the boundary (#S)
284// - w is the weight of the kernels defining the basis
285// - pi = (xi, yi) is the i-th collocation point
286//
287// Moreover, we want to impose a constraint on the weight vector w so that each
288// monomial Q(x,y) = x^*y^ with + <= 2 is in the span of the FEM bases {_j}_j.
289//
290// In the case of Laplace's equation, we recall the weak form of the PDE as:
291//
292// Find u such that: _ u v = - _ uv v
293//
294// For our bases to exactly represent a monomial Q(x,y), it means that its
295// approximation by the finite element bases {_j}_j must be equal to Q(x,y).
296// In particular, for any _j that is nonzero on the polyhedral element E, we must have:
297//
298// _{ in } Q() _j() d = - _{ \in } Q()_j() d (1)
299//
300// Now, for each of the 5 non-constant monomials (9 in 3D), we need to compute
301// (x^*y^). For (,) {(1,0), (0,1), (1,1), (2,0), (0,2)}, this yields
302// the following equalities:
303//
304// x = 0 (2a)
305// y = 0 (2b)
306// xy = 0 (2c)
307// x = 1 (2d)
308// y = 1 (2e)
309//
310// If we plug these back into (1), and split the integral between the polyhedral
311// element E and \E, we obtain the following constraints:
312//
313// _E Q_j + _E Q _j = - _{\E} Q_j - _{\E} Q _j (3)
314//
315// Note that the right-hand side of (3) is already known, since no two polyhedral
316// cells are adjacent to each other, and the bases overlapping a polyhedron vanish
317// on the boundary of the domain . This right-hand side is computed in advance
318// and passed to our functions as in argument `local_basis_integral`.
319//
320// The left-hand side of equation (3) reduces to the following (in 2D):
321//
322// _E x(_j) = c10 (4a)
323// _E y(_j) = c01 (4b)
324// _E (yx(_j) + yx(_jj)) = c11 (4c)
325// _E 2xx(_j) + _E 2 _j = c20 (4d)
326// _E 2yy(_j) + _E 2 _j = c02 (4e)
327//
328// The next step is to express the basis _j in terms of the harmonic kernels and
329// quadratic polynomials:
330//
331// _j(x,y) = _k w_k _k(x,y) + a00 + a10*x + a01*y + a11*x*y + a20*x + a02*y
332//
333// The five equations in (4) become:
334//
335// _j w_k x(_k) = q10 (_j w_k (_k) + a00) + _j w_k q10 . (_k + a00)
336// _j w_k x(_k) + a10 |E| + a11 y + a20 2x = c10
337// _j w_k y(_k) + a01 |E| + a11 x + a02 2y = c01
338// _j w_k (yx(_k) + xy(_k)) + a10 y + a01 x + a11 (x+y) + a20 2xy + a02 2xy = c11
339// _j w_k (2xx(_k) + 2_k) + a10 4x + a01 2y + a11 4xy + a20 6x + a02 2y = c20
340// _j w_k (2yy(_k) + 2_k) + a10 2x + a01 4y + a11 4xy + a20 2x + a02 6y = c02
341// _j w_k (2yy(_k) + 2_k) = q20 (_j w_k (_k) + a00) + _j w_k q20 . (_k + a00) = -2 (_j w_k (_k) + a00) + _j w_k 2x x(_k)
342//
343// This system gives us a relationship between the fives a10, a01, a11, a20, a02
344// and the rest of the w_k + a constant translation term. We can write down the
345// corresponding system:
346//
347// a10 a01 a11 a20 a02
348//
349// |E| y 2x w_k
350//
351// |E| x 2y
352//
353// M = y x x+y 2xy 2xy = \tilde{L} + \tilde{t}
354//
355// 4x 2y 4xy 6x 2y
356// w_#K
357// 2x 4y 4xy 2x 6y a00
358//
359//
360// Now, if we want to express w as w = Lv + t, and solve our least-square
361// system as before, we need to invert M and compute L and t in terms of
362// \tilde{L} and \tilde{t}
363//
364//
365// 1
366// 1
367//
368// L = ^{ (#K+1+dim+dim*(dim+1)/2) x (#K+1}) }
369// 1
370// M^{-1} \tilde{L}
371//
372//
373// 0
374//
375// t = ^{#K+1+dim+dim*(dim+1)/2}
376// 0
377// M^{-1} \tilde{t}
378//
379// After solving the new least square system A L v = rhs - A t, we can retrieve
380// w = L v
381//
383
384void RBFWithQuadratic::compute_kernels_matrix(const Eigen::MatrixXd &samples, Eigen::MatrixXd &A) const
385{
386 // Compute A
387 const int num_kernels = centers_.rows();
388 const int dim = (is_volume() ? 3 : 2);
389
390 A.resize(samples.rows(), num_kernels + 1 + dim + dim * (dim + 1) / 2);
391 for (int j = 0; j < num_kernels; ++j)
392 {
393 A.col(j) = (samples.rowwise() - centers_.row(j)).rowwise().norm().unaryExpr([this](double x) { return kernel(is_volume(), x); });
394 }
395 A.col(num_kernels).setOnes(); // constant term
396 A.middleCols(num_kernels + 1, dim) = samples; // linear terms
397 if (dim == 2)
398 {
399 A.middleCols(num_kernels + dim + 1, 1) = samples.rowwise().prod(); // mixed terms
400 }
401 else if (dim == 3)
402 {
403 A.middleCols(num_kernels + dim + 1, 3) = samples;
404 A.middleCols(num_kernels + dim + 1 + 0, 1).array() *= samples.col(1).array();
405 A.middleCols(num_kernels + dim + 1 + 1, 1).array() *= samples.col(2).array();
406 A.middleCols(num_kernels + dim + 1 + 2, 1).array() *= samples.col(0).array();
407 }
408 A.rightCols(dim) = samples.array().square(); // quadratic terms
409}

413void RBFWithQuadratic::compute_constraints_matrix_2d_old(
414 const int num_bases,
415 const Quadrature &quadr,
416 const Eigen::MatrixXd &local_basis_integral,
417 Eigen::MatrixXd &L,
418 Eigen::MatrixXd &t) const
419{
420 const int num_kernels = centers_.rows();
421 const int dim = centers_.cols();
422 assert(dim == 2);
423
424 // K_cst = _k
425 // K_lin = x(_k), y(_k)
426 // K_mix = yx(_k), xy(_k)
427 // K_sqr = xx(_k), yy(_k)
428 Eigen::VectorXd K_cst = Eigen::VectorXd::Zero(num_kernels);
429 Eigen::MatrixXd K_lin = Eigen::MatrixXd::Zero(num_kernels, dim);
430 Eigen::MatrixXd K_mix = Eigen::MatrixXd::Zero(num_kernels, dim);
431 Eigen::MatrixXd K_sqr = Eigen::MatrixXd::Zero(num_kernels, dim);
432 for (int j = 0; j < num_kernels; ++j)
433 {
434 // x(_k)(p) = _q (xq - xk) * 1/r * h'(r) * wq
435 // - xq is the x coordinate of the q-th quadrature point
436 // - wq is the q-th quadrature weight
437 // - r is the distance from pq to the kernel center
438 // - h is the RBF kernel (scalar function)
439 for (int q = 0; q < quadr.points.rows(); ++q)
440 {
441 const RowVectorNd p = quadr.points.row(q) - centers_.row(j);
442 const double r = p.norm();
443 const RowVectorNd gradPhi = p * kernel_prime(is_volume(), r) / r * quadr.weights(q);
444 K_cst(j) += kernel(is_volume(), r) * quadr.weights(q);
445 K_lin.row(j) += gradPhi;
446 K_mix(j, 0) += quadr.points(q, 1) * gradPhi(0);
447 K_mix(j, 1) += quadr.points(q, 0) * gradPhi(1);
448 K_sqr.row(j) += (quadr.points.row(q).array() * gradPhi.array()).matrix();
449 }
450 }
451
452 // I_lin = x, y
453 // I_mix = xy
454 // I_sqr = x, y
455 Eigen::RowVectorXd I_lin = (quadr.points.array().colwise() * quadr.weights.array()).colwise().sum();
456 Eigen::RowVectorXd I_mix = (quadr.points.rowwise().prod().array() * quadr.weights.array()).colwise().sum();
457 Eigen::RowVectorXd I_sqr = (quadr.points.array().square().colwise() * quadr.weights.array()).colwise().sum();
458 double volume = quadr.weights.sum();
459
460 // std::cout << I_lin << std::endl;
461 // std::cout << I_mix << std::endl;
462 // std::cout << I_sqr << std::endl;
463
464 // Compute M
465 Eigen::Matrix<double, 5, 5> M;
466 M << volume, 0, I_lin(1), 2 * I_lin(0), 0,
467 0, volume, I_lin(0), 0, 2 * I_lin(1),
468 I_lin(1), I_lin(0), I_sqr(0) + I_sqr(1), 2 * I_mix(0), 2 * I_mix(0),
469 4 * I_lin(0), 2 * I_lin(1), 4 * I_mix(0), 6 * I_sqr(0), 2 * I_sqr(1),
470 2 * I_lin(0), 4 * I_lin(1), 4 * I_mix(0), 2 * I_sqr(0), 6 * I_sqr(1);
471 Eigen::FullPivLU<Eigen::Matrix<double, 5, 5>> lu(M);
472 assert(lu.isInvertible());
473
474 // show_matrix_stats(M);
475
476 // Compute L
477 L.resize(num_kernels + 1 + dim + dim * (dim + 1) / 2, num_kernels + 1);
478 L.setZero();
479 L.diagonal().setOnes();
480
481 L.block(num_kernels + 1, 0, dim, num_kernels) = -K_lin.transpose();
482 L.block(num_kernels + 1 + dim, 0, 1, num_kernels) = -K_mix.transpose().colwise().sum();
483 L.block(num_kernels + 1 + dim + 1, 0, dim, num_kernels) = -2.0 * (K_sqr.colwise() + K_cst).transpose();
484 L.bottomRightCorner(dim, 1).setConstant(-2.0 * volume);
485 // j \in [0, 4]
486 // i \in [0, num_kernels]
487 // ass_val = [q_10, q_01, q_11, q_20, q_02, psi_0, ..., psi_k]
488
489 // strong rows is the evaluation at quadrature points
490 // strong.col(0) = pde(q_10) (probably 0)
491 // strong.col(4) = pde(q_02) (it is 2 for laplacian)
492 //L.block(num_kernels + 1 + i, j) = +/- assembler.assemble(ass_val, j, 5 + i) +/- (strong.col(j).array() * ass_val.basis_values[5+i].val.array() * quadr.weights.array()).sum();
493
494 L.block(num_kernels + 1, 0, 5, num_kernels + 1) = lu.solve(L.block(num_kernels + 1, 0, 5, num_kernels + 1));
495 // std::cout << L.bottomRightCorner(10, 10) << std::endl;
496
497 // Compute t
498 t.resize(L.rows(), num_bases);
499 t.setZero();
500 t.bottomRows(5) = local_basis_integral.transpose();
501 t.bottomRows(5) = lu.solve(weights_.bottomRows(5));
502}

504void RBFWithQuadratic::compute_constraints_matrix_2d(
505 const AssemblerUtils &assembler,
506 const std::string &assembler_name,
507 const int num_bases,
508 const Quadrature &quadr,
509 const Eigen::MatrixXd &local_basis_integral,
510 Eigen::MatrixXd &L,
511 Eigen::MatrixXd &t) const
512{
513 const int num_kernels = centers_.rows();
514 const int space_dim = centers_.cols();
515 const int assembler_dim = assembler.is_tensor(assembler_name) ? 2 : 1;
516 assert(space_dim == 2);
517
518 std::array<Eigen::MatrixXd, 5> strong;
519
520 // ass_val = [q_10, q_01, q_11, q_20, q_02, psi_0, ..., psi_k]
521 ElementAssemblyValues ass_val;
522 ass_val.has_parameterization = false;
523 ass_val.basis_values.resize(5 + num_kernels);
524
525 //evaluating monomial and grad of monomials at quad points
526 setup_monomials_vals_2d(0, quadr.points, ass_val);
527 setup_monomials_strong_2d(assembler_dim, assembler, assembler_name, quadr.points, quadr.weights.array(), strong);
528
529 //evaluating psi and grad psi at quadr points
530 for (int j = 0; j < num_kernels; ++j)
531 {
532 ass_val.basis_values[5 + j].val = Eigen::MatrixXd(quadr.points.rows(), 1);
533 ass_val.basis_values[5 + j].grad = Eigen::MatrixXd(quadr.points.rows(), quadr.points.cols());
534
535 for (int q = 0; q < quadr.points.rows(); ++q)
536 {
537 const RowVectorNd p = quadr.points.row(q) - centers_.row(j);
538 const double r = p.norm();
539
540 ass_val.basis_values[5 + j].val(q) = kernel(is_volume(), r);
541 ass_val.basis_values[5 + j].grad.row(q) = p * kernel_prime(is_volume(), r) / r;
542 }
543 }
544
545 for (size_t i = 5; i < ass_val.basis_values.size(); ++i)
546 {
547 ass_val.basis_values[i].grad_t_m = ass_val.basis_values[i].grad;
548 }
549
550 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, 0, 10, 10> M(5 * assembler_dim, 5 * assembler_dim);
551 for (int i = 0; i < 5; ++i)

614void RBFWithQuadratic::compute_constraints_matrix_3d(
615 const AssemblerUtils &assembler,
616 const std::string &assembler_name,
617 const int num_bases,
618 const Quadrature &quadr,
619 const Eigen::MatrixXd &local_basis_integral,
620 Eigen::MatrixXd &L,
621 Eigen::MatrixXd &t) const
622{
623 const int num_kernels = centers_.rows();
624 const int dim = centers_.cols();
625 assert(dim == 3);
626 assert(local_basis_integral.cols() == 9);
627
628 // K_cst = _k
629 // K_lin = x(_k), y(_k), z(_k)
630 // K_mix = (yx(_k)+xy(_k)), (zy(_k)+yz(_k)), (xz(_k)+zx(_k))
631 // K_sqr = xx(_k), yy(_k), zz(_k)
632 Eigen::VectorXd K_cst = Eigen::VectorXd::Zero(num_kernels);
633 Eigen::MatrixXd K_lin = Eigen::MatrixXd::Zero(num_kernels, dim);
634 Eigen::MatrixXd K_mix = Eigen::MatrixXd::Zero(num_kernels, dim);
635 Eigen::MatrixXd K_sqr = Eigen::MatrixXd::Zero(num_kernels, dim);
636 for (int j = 0; j < num_kernels; ++j)
637 {
638 // x(_k)(p) = _q (xq - xk) * 1/r * h'(r) * wq
639 // - xq is the x coordinate of the q-th quadrature point
640 // - wq is the q-th quadrature weight
641 // - r is the distance from pq to the kernel center
642 // - h is the RBF kernel (scalar function)
643 for (int q = 0; q < quadr.points.rows(); ++q)
644 {
645 const RowVectorNd p = quadr.points.row(q) - centers_.row(j);
646 const double r = p.norm();
647 const RowVectorNd gradPhi = p * kernel_prime(is_volume(), r) / r * quadr.weights(q);
648 K_cst(j) += kernel(is_volume(), r) * quadr.weights(q);
649 K_lin.row(j) += gradPhi;
650 for (int d = 0; d < dim; ++d)
651 {
652 K_mix(j, d) += quadr.points(q, (d + 1) % dim) * gradPhi(d) + quadr.points(q, d) * gradPhi((d + 1) % dim);
653 }
654 K_sqr.row(j) += (quadr.points.row(q).array() * gradPhi.array()).matrix();
655 }
656 }
657
658 // I_lin = x, y, z
659 // I_sqr = x, y, z
660 // I_mix = xy, yz, zx
661 Eigen::RowVectorXd I_lin = (quadr.points.array().colwise() * quadr.weights.array()).colwise().sum();
662 Eigen::RowVectorXd I_sqr = (quadr.points.array().square().colwise() * quadr.weights.array()).colwise().sum();
663 Eigen::RowVectorXd I_mix(3);
664 I_mix(0) = (quadr.points.col(0).array() * quadr.points.col(1).array() * quadr.weights.array()).sum();
665 I_mix(1) = (quadr.points.col(1).array() * quadr.points.col(2).array() * quadr.weights.array()).sum();
666 I_mix(2) = (quadr.points.col(2).array() * quadr.points.col(0).array() * quadr.weights.array()).sum();
667 double volume = quadr.weights.sum();
668
669 // std::cout << I_lin << std::endl;
670 // std::cout << I_mix << std::endl;
671 // std::cout << I_sqr << std::endl;
672
673 // Compute M
674 Eigen::Matrix<double, 9, 9> M;
675 M << volume, 0, 0, I_lin(1), 0, I_lin(2), 2 * I_lin(0), 0, 0,
676 0, volume, 0, I_lin(0), I_lin(2), 0, 0, 2 * I_lin(1), 0,
677 0, 0, volume, 0, I_lin(1), I_lin(0), 0, 0, 2 * I_lin(2),
678 I_lin(1), I_lin(0), 0, I_sqr(0) + I_sqr(1), I_mix(2), I_mix(1), 2 * I_mix(0), 2 * I_mix(0), 0,
679 0, I_lin(2), I_lin(1), I_mix(2), I_sqr(1) + I_sqr(2), I_mix(0), 0, 2 * I_mix(1), 2 * I_mix(1),
680 I_lin(2), 0, I_lin(0), I_mix(1), I_mix(0), I_sqr(2) + I_sqr(0), 2 * I_mix(2), 0, 2 * I_mix(2),
681 2 * I_lin(0), 0, 0, 2 * I_mix(0), 0, 2 * I_mix(2), 4 * I_sqr(0), 0, 0,
682 0, 2 * I_lin(1), 0, 2 * I_mix(0), 2 * I_mix(1), 0, 0, 4 * I_sqr(1), 0,
683 0, 0, 2 * I_lin(2), 0, 2 * I_mix(1), 2 * I_mix(2), 0, 0, 4 * I_sqr(2);
684 Eigen::Matrix<double, 1, 9> M_rhs;
685 M_rhs.segment<3>(0) = I_lin;
686 M_rhs.segment<3>(3) = I_mix;
687 M_rhs.segment<3>(6) = I_sqr;
688 // M_rhs << I_lin, I_mix, I_sqr;
689 M.bottomRows(dim).rowwise() += 2.0 * M_rhs;
690 Eigen::FullPivLU<Eigen::Matrix<double, 9, 9>> lu(M);
691 assert(lu.isInvertible());
692
693 // show_matrix_stats(M);
694
695 // Compute L
696 L.resize(num_kernels + 1 + dim + dim * (dim + 1) / 2, num_kernels + 1);
697 L.setZero();
698 L.diagonal().setOnes();
699 L.block(num_kernels + 1, 0, dim, num_kernels) = -K_lin.transpose();
700 L.block(num_kernels + 1 + dim, 0, dim, num_kernels) = -K_mix.transpose();
701 L.block(num_kernels + 1 + dim + dim, 0, dim, num_kernels) = -2.0 * (K_sqr.colwise() + K_cst).transpose();
702 L.bottomRightCorner(dim, 1).setConstant(-2.0 * volume);
703 L.block(num_kernels + 1, 0, 9, num_kernels + 1) = lu.solve(L.block(num_kernels + 1, 0, 9, num_kernels + 1));
704 // std::cout << L.bottomRightCorner(10, 10) << std::endl;
705
706 // Compute t
707 t.resize(L.rows(), num_bases);
708 t.setZero();
709 t.bottomRows(9) = local_basis_integral.transpose();
710 t.bottomRows(9) = lu.solve(weights_.bottomRows(9));
711}
