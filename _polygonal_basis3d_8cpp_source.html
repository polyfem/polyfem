36 std::vector<int> compute_nonzero_bases_ids(const Mesh3D &mesh, const int c,
37 const std::vector<ElementBases> &bases,
38 const std::map<int, InterfaceData> &poly_face_to_data)
39 {
40 std::vector<int> local_to_global;
41
42 for (int lf = 0; lf < mesh.n_cell_faces(c); ++lf)
43 {
44 auto index = mesh.get_index_from_element(c, lf, 0);
45 const int c2 = mesh.switch_element(index).element;
46 assert(c2 >= 0); // no boundary polytope
47 assert(poly_face_to_data.count(index.face) > 0);
48 const InterfaceData &bdata = poly_face_to_data.at(index.face);
49 const ElementBases &b = bases[c2];
50 for (int other_local_basis_id : bdata.local_indices)
51 {
52 for (const auto &x : b.bases[other_local_basis_id].global())
53 {
54 const int global_node_id = x.index;
55 local_to_global.push_back(global_node_id);
56 }
57 }
58 }
59
60 std::sort(local_to_global.begin(), local_to_global.end());
61 auto it = std::unique(local_to_global.begin(), local_to_global.end());
62 local_to_global.resize(std::distance(local_to_global.begin(), it));
63
64 return local_to_global;
65 }

84 // Assemble the surface quad mesh (V, F) corresponding to the polyhedron c
85 GetAdjacentLocalEdge compute_quad_mesh_from_cell(
86 const Mesh3D &mesh, int c, Eigen::MatrixXd &V, Eigen::MatrixXi &F)
87 {
88 std::vector<std::array<int, 4>> quads(mesh.n_cell_faces(c));
89 typedef std::tuple<int, int, bool> QuadLocalEdge;
90 std::vector<std::array<QuadLocalEdge, 4>> adj(mesh.n_cell_faces(c));
91 int num_vertices = 0;
92 std::map<int, int> vertex_g2l;
93 std::map<int, int> face_g2l;
94 for (int lf = 0; lf < mesh.n_cell_faces(c); ++lf)
95 {
96 face_g2l.emplace(mesh.get_index_from_element(c, lf, lv0).face, lf);
97 }
98 for (int lf = 0; lf < mesh.n_cell_faces(c); ++lf)
99 {
100 auto index = mesh.get_index_from_element(c, lf, lv0);
101 assert(mesh.n_face_vertices(index.face) == 4);
102 for (int lv = 0; lv < 4; ++lv)
103 {
104 if (!vertex_g2l.count(index.vertex))
105 {
106 vertex_g2l.emplace(index.vertex, num_vertices++);
107 }
108 quads[lf][lv] = vertex_g2l.at(index.vertex);
109
110 // Set adjacency info
111 auto index2 = mesh.switch_face(index);
112 int lf2 = face_g2l.at(index2.face);
113 std::get<0>(adj[lf][lv]) = lf2;
114 auto index3 = mesh.get_index_from_element(c, lf2, lv0);
115 for (int lv2 = 0; lv2 < 4; ++lv2)
116 {
117 if (index3.edge == index2.edge)
118 {
119 std::get<1>(adj[lf][lv]) = lv2;
120 if (index2.vertex != index3.vertex)
121 {
122 assert(mesh.switch_vertex(index3).vertex == index2.vertex);
123 std::get<2>(adj[lf][lv]) = true;
124 }
125 else
126 {
127 std::get<2>(adj[lf][lv]) = false;
128 }
129 }
130 index3 = mesh.next_around_face(index3);
131 }
132
133 index = mesh.next_around_face(index);
134 }
135 }
136 V.resize(num_vertices, 3);
137 for (const auto &kv : vertex_g2l)
138 {
139 V.row(kv.second) = mesh.point(kv.first);
140 }
141 F.resize(quads.size(), 4);
142 int f = 0;
143 for (auto q : quads)
144 {
145 F.row(f++) << q[0], q[1], q[2], q[3];
146 }
147
148 return [adj](int q, int lv) {
149 return adj[q][lv];
150 };
151 }

393 // Compute the integral constraints for each basis of the mesh
394 void PolygonalBasis3d::compute_integral_constraints(
395 const AssemblerUtils &assembler,
396 const std::string &assembler_name,
397 const Mesh3D &mesh,
398 const int n_bases,
399 const std::vector<ElementBases> &bases,
400 const std::vector<ElementBases> &gbases,
401 Eigen::MatrixXd &basis_integrals)
402 {
403 assert(mesh.is_volume());
404
405 basis_integrals.resize(n_bases, 9);
406 basis_integrals.setZero();
407 Eigen::MatrixXd rhs(n_bases, 9);
408 rhs.setZero();
409
410 const int n_elements = mesh.n_elements();
411 for (int e = 0; e < n_elements; ++e)
412 {
413 if (mesh.is_polytope(e))
414 {
415 continue;
416 }
417 // ElementAssemblyValues vals = values[e];
418 // const ElementAssemblyValues &gvals = gvalues[e];
419 ElementAssemblyValues vals;
420 vals.compute(e, mesh.is_volume(), bases[e], gbases[e]);
421
422 // Computes the discretized integral of the PDE over the element
423 const int n_local_bases = int(vals.basis_values.size());
424 for (int j = 0; j < n_local_bases; ++j)
425 {
426 const AssemblyValues &v = vals.basis_values[j];
427 const double integral_100 = (v.grad_t_m.col(0).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
428 const double integral_010 = (v.grad_t_m.col(1).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
429 const double integral_001 = (v.grad_t_m.col(2).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
430
431 const double integral_110 = ((vals.val.col(1).array() * v.grad_t_m.col(0).array() + vals.val.col(0).array() * v.grad_t_m.col(1).array()) * vals.det.array() * vals.quadrature.weights.array()).sum();
432 const double integral_011 = ((vals.val.col(2).array() * v.grad_t_m.col(1).array() + vals.val.col(1).array() * v.grad_t_m.col(2).array()) * vals.det.array() * vals.quadrature.weights.array()).sum();
433 const double integral_101 = ((vals.val.col(0).array() * v.grad_t_m.col(2).array() + vals.val.col(2).array() * v.grad_t_m.col(0).array()) * vals.det.array() * vals.quadrature.weights.array()).sum();
434
435 const double integral_200 = 2 * (vals.val.col(0).array() * v.grad_t_m.col(0).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
436 const double integral_020 = 2 * (vals.val.col(1).array() * v.grad_t_m.col(1).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
437 const double integral_002 = 2 * (vals.val.col(2).array() * v.grad_t_m.col(2).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
438
439 const double area = (v.val.array() * vals.det.array() * vals.quadrature.weights.array()).sum();
440
441 for (size_t ii = 0; ii < v.global.size(); ++ii)
442 {
443 basis_integrals(v.global[ii].index, 0) += integral_100 * v.global[ii].val;
444 basis_integrals(v.global[ii].index, 1) += integral_010 * v.global[ii].val;
445 basis_integrals(v.global[ii].index, 2) += integral_001 * v.global[ii].val;
446
447 basis_integrals(v.global[ii].index, 3) += integral_110 * v.global[ii].val;
448 basis_integrals(v.global[ii].index, 4) += integral_011 * v.global[ii].val;
449 basis_integrals(v.global[ii].index, 5) += integral_101 * v.global[ii].val;
450
451 basis_integrals(v.global[ii].index, 6) += integral_200 * v.global[ii].val;
452 basis_integrals(v.global[ii].index, 7) += integral_020 * v.global[ii].val;
453 basis_integrals(v.global[ii].index, 8) += integral_002 * v.global[ii].val;
454
455 rhs(v.global[ii].index, 6) += -2.0 * area * v.global[ii].val;
456 rhs(v.global[ii].index, 7) += -2.0 * area * v.global[ii].val;
457 rhs(v.global[ii].index, 8) += -2.0 * area * v.global[ii].val;
458 }
459 }
460 }
461
462 basis_integrals -= rhs;
463 }
