56 bool OBJReader::load(
57 FILE *obj_file,
58 std::vector<std::vector<double>> &V,
59 std::vector<std::vector<double>> &TC,
60 std::vector<std::vector<double>> &N,
61 std::vector<std::vector<int>> &F,
62 std::vector<std::vector<int>> &FTC,
63 std::vector<std::vector<int>> &FN,
64 std::vector<std::vector<int>> &L)
65 {
66 // File open was successful so clear outputs
67 V.clear();
68 TC.clear();
69 N.clear();
70 F.clear();
71 FTC.clear();
72 FN.clear();
73 L.clear();
74
75 // variables and constants to assist parsing the .obj file
76 // Constant strings to compare against
77 std::string v("v");
78 std::string vn("vn");
79 std::string vt("vt");
80 std::string f("f");
81 std::string l("l");
82 std::string tic_tac_toe("#");
83
84 const int LINE_MAX_LEN = 2048;
85
86 char line[LINE_MAX_LEN];
87 int line_no = 1;
88 while (fgets(line, LINE_MAX_LEN, obj_file) != NULL)
89 {
90 char type[LINE_MAX_LEN];
91 // Read first word containing type
92 if (sscanf(line, "%s", type) == 1)
93 {
94 // Get pointer to rest of line right after type
95 char *rest_of_line = &line[strlen(type)];
96 if (type == v)
97 {
98 std::istringstream ls(&line[1]);
99 std::vector<double> vertex{std::istream_iterator<double>(ls),
100 std::istream_iterator<double>()};
101
102 // if (vertex.size() < 3) {
103 // logger().error(
104 // "OBJReader::load: vertex on line {:d} should have at "
105 // "least 3 coordinates",
106 // line_no);
107 // fclose(obj_file);
108 // return false;
109 // }
110
111 V.push_back(vertex);
112 }
113 else if (type == vn)
114 {
115 double x[3];
116 int count =
117 sscanf(rest_of_line, "%lf %lf %lf\n", &x[0], &x[1], &x[2]);
118 if (count != 3)
119 {
120 logger().error(
121 "OBJReader::load: normal on line {:d} should have 3 "
122 "coordinates",
123 line_no);
124 fclose(obj_file);
125 return false;
126 }
127 std::vector<double> normal(count);
128 for (int i = 0; i < count; i++)
129 {
130 normal[i] = x[i];
131 }
132 N.push_back(normal);
133 }
134 else if (type == vt)
135 {
136 double x[3];
137 int count =
138 sscanf(rest_of_line, "%lf %lf %lf\n", &x[0], &x[1], &x[2]);
139 if (count != 2 && count != 3)
140 {
141 logger().error(
142 "OBJReader::load: texture coords on line {:d} should have "
143 "2 or 3 coordinates (has {:d})",
144 line_no, count);
145 fclose(obj_file);
146 return false;
147 }
148 std::vector<double> tex(count);
149 for (int i = 0; i < count; i++)
150 {
151 tex[i] = x[i];
152 }
153 TC.push_back(tex);
154 }
155 else if (type == f)
156 {
157 const auto &shift = [&V](const int i) -> int {
158 return i < 0 ? i + V.size() : i - 1;
159 };
160 const auto &shift_t = [&TC](const int i) -> int {
161 return i < 0 ? i + TC.size() : i - 1;
162 };
163 const auto &shift_n = [&N](const int i) -> int {
164 return i < 0 ? i + N.size() : i - 1;
165 };
166 std::vector<int> f;
167 std::vector<int> ftc;
168 std::vector<int> fn;
169 // Read each "word" after type
170 char word[LINE_MAX_LEN];
171 int offset;
172 while (sscanf(rest_of_line, "%s%n", word, &offset) == 1)
173 {
174 // adjust offset
175 rest_of_line += offset;
176 // Process word
177 long int i, it, in;
178 if (sscanf(word, "%ld/%ld/%ld", &i, &it, &in) == 3)
179 {
180 f.push_back(shift(i));
181 ftc.push_back(shift_t(it));
182 fn.push_back(shift_n(in));
183 }
184 else if (sscanf(word, "%ld/%ld", &i, &it) == 2)
185 {
186 f.push_back(shift(i));
187 ftc.push_back(shift_t(it));
188 }
189 else if (sscanf(word, "%ld//%ld", &i, &in) == 2)
190 {
191 f.push_back(shift(i));
192 fn.push_back(shift_n(in));
193 }
194 else if (sscanf(word, "%ld", &i) == 1)
195 {
196 f.push_back(shift(i));
197 }
198 else
199 {
200 logger().error(
201 "OBJReader::load: face on line {:d} has invalid "
202 "element format",
203 line_no);
204 fclose(obj_file);
205 return false;
206 }
207 }
208 if ((f.size() > 0 && fn.size() == 0 && ftc.size() == 0)
209 || (f.size() > 0 && fn.size() == f.size()
210 && ftc.size() == 0)
211 || (f.size() > 0 && fn.size() == 0
212 && ftc.size() == f.size())
213 || (f.size() > 0 && fn.size() == f.size()
214 && ftc.size() == f.size()))
215 {
216 // No matter what add each type to lists so that lists
217 // are the correct lengths
218 F.push_back(f);
219 FTC.push_back(ftc);
220 FN.push_back(fn);
221 }
222 else
223 {
224 logger().error(
225 "OBJReader::load: face on line {:d} has invalid format",
226 line_no);
227 fclose(obj_file);
228 return false;
229 }
230 }
231 else if (type == l)
232 {
233 std::istringstream ls(&line[1]);
234 std::vector<int> polyline{std::istream_iterator<int>(ls),
235 std::istream_iterator<int>()};
236
237 if (polyline.size() < 2)
238 {
239 logger().error(
240 "OBJReader::load: line element on line {:d} should have "
241 "at least 2 vertices",
242 line_no);
243 fclose(obj_file);
244 return false;
245 }
246
247 for (int i = 0; i < polyline.size(); i++)
248 {
249 polyline[i] = polyline[i] < 0 ? polyline[i] + V.size()
250 : polyline[i] - 1;
251 }
252
253 L.push_back(polyline);
254 }
255 else if (
256 strlen(type) >= 1
257 && (type[0] == '#' || type[0] == 'g' || type[0] == 's'
258 || strcmp("usemtl", type) == 0
259 || strcmp("mtllib", type) == 0))
260 {
261 // ignore comments or other stuff
262 }
263 else
264 {
265 // ignore any other lines
266 std::string line_no_newline = remove_newline(line);
267 logger().warn(
268 "OBJReader::load: ignored non-comment line {:d}: {:s}", line_no,
269 line_no_newline);
270 }
271 }
272 else
273 {
274 // ignore empty line
275 }
276 line_no++;
277 }
278 fclose(obj_file);
279
280 assert(F.size() == FN.size());
281 assert(F.size() == FTC.size());
282
283 return true;
284 }

349 bool OBJWriter::save(const std::string &path, const Eigen::MatrixXd &v, const Eigen::MatrixXi &e, const Eigen::MatrixXi &f)
350 {
351 const Eigen::IOFormat OBJ_VERTEX_FORMAT(
352 /*precision=*/Eigen::FullPrecision,
353 /*flags=*/Eigen::DontAlignCols,
354 /*coeffSeparator=*/" ",
355 /*rowSeparator=*/"",
356 /*rowPrefix=*/"v ",
357 /*rowSuffix=*/v.cols() == 2 ? " 0\n" : "\n",
358 /*matPrefix=*/"",
359 /*fill=*/"");
360
361 std::ofstream obj(path, std::ios::out);
362 if (!obj.is_open())
363 return false;
364
365 for (int i = 0; i < v.rows(); ++i)
366 obj << v.row(i).format(OBJ_VERTEX_FORMAT);
367
368 for (int i = 0; i < e.rows(); ++i)
369 obj << fmt::format("l {} {}\n", e(i, 0) + 1, e(i, 1) + 1);
370
371 for (int i = 0; i < f.rows(); ++i)
372 obj << fmt::format("f {} {} {}\n", f(i, 0) + 1, f(i, 1) + 1, f(i, 2) + 1);
373
374 return true;
375 }
