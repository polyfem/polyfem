19 inline bool intersect_segments(
20 const Eigen::RowVector2d &p1, const Eigen::RowVector2d &p2,
21 const Eigen::RowVector2d &q1, const Eigen::RowVector2d &q2,
22 Eigen::RowVector2d &result)
23 {
24
25 Eigen::RowVector2d Vp = p2 - p1;
26 Eigen::RowVector2d Vq = q2 - q1;
27 Eigen::RowVector2d pq = q1 - p1;
28
29 double a = Vp(0);
30 double b = -Vq(0);
31 double c = Vp(1);
32 double d = -Vq(1);
33
34 double delta = a * d - b * c;
35 if (delta == 0.0)
36 {
37 return false;
38 }
39
40 double tp = (d * pq(0) - b * pq(1)) / delta;
41
42 result << (1.0 - tp) * p1(0) + tp * p2(0),
43 (1.0 - tp) * p1(1) + tp * p2(1);
44
45 return true;
46 }

52void polyfem::mesh::clip_polygon_by_half_plane(const Eigen::MatrixXd &P_in,
53 const Eigen::RowVector2d &q1, const Eigen::RowVector2d &q2, Eigen::MatrixXd &P_out)
54{
55 using namespace GEO;
56 assert(P_in.cols() == 2);
57 std::vector<Eigen::RowVector2d> result;
58
59 if (P_in.rows() == 0)
60 {
61 P_out.resize(0, 2);
62 return;
63 }
64
65 if (P_in.rows() == 1)
66 {
67 if (point_is_in_half_plane(P_in.row(0), q1, q2))
68 {
69 P_out.resize(1, 2);
70 P_out << P_in.row(0);
71 }
72 else
73 {
74 P_out.resize(0, 2);
75 }
76 return;
77 }
78
79 Eigen::RowVector2d prev_p = P_in.row(P_in.rows() - 1);
80 Sign prev_status = point_is_in_half_plane(prev_p, q1, q2);
81
82 for (unsigned int i = 0; i < P_in.rows(); ++i)
83 {
84 Eigen::RowVector2d p = P_in.row(i);
85 Sign status = point_is_in_half_plane(p, q1, q2);
86 if (status != prev_status && status != ZERO && prev_status != ZERO)
87 {
88 Eigen::RowVector2d intersect;
89 if (intersect_segments(prev_p, p, q1, q2, intersect))
90 {
91 result.push_back(intersect);
92 }
93 }
94
95 switch (status)
96 {
97 case NEGATIVE:
98 break;
99 case ZERO:
100 result.push_back(p);
101 break;
102 case POSITIVE:
103 result.push_back(p);
104 break;
105 default:
106 break;
107 }
108
109 prev_p = p;
110 prev_status = status;
111 }
112
113 P_out.resize((int)result.size(), 2);
114 for (size_t i = 0; i < result.size(); ++i)
115 {
116 P_out.row((int)i) = result[i];
117 }
118}

122void polyfem::mesh::compute_visibility_kernel(const Eigen::MatrixXd &IV, Eigen::MatrixXd &OV)
123{
124 assert(IV.cols() == 2 || IV.cols() == 3);
125
126 // 1) Start from the bounding box of the input points
127 Eigen::MatrixXd src, dst;
128 const auto &minV = IV.colwise().minCoeff().array();
129 const auto &maxV = IV.colwise().maxCoeff().array();
130 src.resize(4, 2);
131 src.row(0) << minV(0), minV(1);
132 src.row(1) << maxV(0), minV(1);
133 src.row(2) << maxV(0), maxV(1);
134 src.row(3) << minV(0), maxV(1);
135 // std::cout << IV << std::endl;
136 // std::cout << minV << ' ' << maxV << std::endl;
137
138 // 2) Clip by half planes until we are left with the kernel
139 for (unsigned int i = 0; i < IV.rows(); ++i)
140 {
141 unsigned int j = ((i + 1) % (int)IV.rows());
142 const Eigen::RowVector2d &q1 = IV.row(i).head<2>();
143 const Eigen::RowVector2d &q2 = IV.row(j).head<2>();
144 clip_polygon_by_half_plane(src, q1, q2, dst);
145 std::swap(src, dst);
146 }
147 OV = src;
148}

260 // Return true iff [a,b] intersects [c,d], and store the intersection in ans
261 bool intersect_segment(const Vec2d &a, const Vec2d &b, Vec2d c, Vec2d d, Vec2d &ans)
262 {
263 const double eps = 1e-10; // small epsilon for numerical precision
264 double x = det(c - a, d - c);
265 double y = det(b - a, a - c);
266 double z = det(b - a, d - c);
267 if (std::abs(z) < eps || x * z < 0 || x * z > z * z || y * z < 0 || y * z > z * z)
268 return false;
269 ans = c + (d - c) * y / z;
270 return true;
271 }

273 bool is_point_inside(const Eigen::MatrixXd &poly, const Vec2d &outside, const Vec2d &query)
274 {
275 int n = poly.rows();
276 bool tmp, ans = false;
277 for (long i = 0; i < poly.rows(); ++i)
278 {
279 Vec2d m; // Coordinates of intersection point
280 tmp = intersect_segment(query, outside, poly.row(i), poly.row((i + 1) % n), m);
281 ans = (ans != tmp);
282 }
283 return ans;
284 }

290int polyfem::mesh::is_inside(const Eigen::MatrixXd &IV, const Eigen::MatrixXd &Q, std::vector<bool> &inside)
291{
292 assert(IV.cols() == 2);
293 Eigen::RowVector2d minV = IV.colwise().minCoeff().array();
294 Eigen::RowVector2d maxV = IV.colwise().maxCoeff().array();
295 Eigen::RowVector2d center = 0.5 * (maxV + minV);
296 Eigen::RowVector2d outside(2.0 * minV(0) - center(0), center(1));
297 inside.resize(Q.rows());
298 int num_inside = 0;
299 for (long i = 0; i < Q.rows(); ++i)
300 {
301 inside[i] = is_point_inside(IV, outside, Q.row(i));
302 if (inside[i])
303 {
304 ++num_inside;
305 }
306 }
307 return num_inside;

312void polyfem::mesh::sample_polygon(const Eigen::MatrixXd &poly, int num_samples, Eigen::MatrixXd &S)
313{
314 assert(poly.rows() >= 2);
315 int n = poly.rows();
316
317 auto length = [&](int i) {
318 return (poly.row(i) - poly.row((i + 1) % n)).norm();
319 };
320
321 // Step 1: compute starting edge + total length of the polygon
322 int i0 = 0;
323 double max_length = 0;
324 double total_length = 0;
325 for (int i = 0; i < n; ++i)
326 {
327 double len = length(i);
328 total_length += len;
329 if (len > max_length)
330 {
331 max_length = len;
332 i0 = i;
333 }
334 }
335
336 // Step 2: place a sample at regular intervals along the polygon,
337 // starting from the middle of edge i0
338 S.resize(num_samples, poly.cols());
339 double spacing = total_length / num_samples; // sampling distance
340 double offset = length(i0) / 2.0; // distance from first sample to vertex i0
341 double distance_to_next = length(i0);
342 for (int s = 0, i = i0; s < num_samples; ++s)
343 {
344 double distance_to_sample = s * spacing + offset; // next sample length
345 while (distance_to_sample > distance_to_next)
346 {
347 i = (i + 1) % n;
348 distance_to_next += length(i);
349 }
350 double t = (distance_to_next - distance_to_sample) / length(i);
351 S.row(s) = t * poly.row(i) + (1.0 - t) * poly.row((i + 1) % n);
352 }
