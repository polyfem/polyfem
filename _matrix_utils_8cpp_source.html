351Eigen::VectorXd polyfem::utils::flatten(const Eigen::MatrixXd &X)
352{
353 if (X.size() == 0)
354 return Eigen::VectorXd();
355
356 Eigen::VectorXd x(X.size());
357 for (int i = 0; i < X.rows(); ++i)
358 {
359 for (int j = 0; j < X.cols(); ++j)
360 {
361 x(i * X.cols() + j) = X(i, j);
362 }
363 }
364 assert(nanproof_equals(X(0, 0), x(0)));
365 assert(X.cols() <= 1 || nanproof_equals(X(0, 1), x(1)));
366 return x;

371{
372 if (x.size() == 0)
373 return Eigen::MatrixXd(0, dim);
374

376 Eigen::MatrixXd X(x.size() / dim, dim);
377 for (int i = 0; i < x.size(); ++i)
378 {
379 X(i / dim, i % dim) = x(i);

413void polyfem::utils::full_to_reduced_matrix(
414 const int full_size,
415 const int reduced_size,
416 const std::vector<int> &removed_vars,
417 const StiffnessMatrix &full,
418 StiffnessMatrix &reduced)
419{
420 POLYFEM_SCOPED_TIMER("full to reduced matrix");
421
422 if (reduced_size == full_size || reduced_size == full.rows())
423 {
424 assert(reduced_size == full.rows() && reduced_size == full.cols());
425 reduced = full;
426 return;
427 }
428 assert(full.rows() == full_size && full.cols() == full_size);
429
430 Eigen::VectorXi indices(full_size);
431 int index = 0;
432 size_t kk = 0;
433 for (int i = 0; i < full_size; ++i)
434 {
435 if (kk < removed_vars.size() && removed_vars[kk] == i)
436 {
437 ++kk;
438 indices(i) = -1;
439 }
440 else
441 {
442 indices(i) = index++;
443 }
444 }
445 assert(index == reduced_size);
446
447 std::vector<Eigen::Triplet<double>> entries;
448 entries.reserve(full.nonZeros()); // Conservative estimate
449 for (int k = 0; k < full.outerSize(); ++k)
450 {
451 if (indices(k) < 0)
452 continue;
453
454 for (StiffnessMatrix::InnerIterator it(full, k); it; ++it)
455 {
456 assert(it.col() == k);
457 if (indices(it.row()) < 0 || indices(it.col()) < 0)
458 continue;
459
460 assert(indices(it.row()) >= 0);
461 assert(indices(it.col()) >= 0);
462
463 entries.emplace_back(indices(it.row()), indices(it.col()), it.value());
464 }
465 }
466
467 reduced.resize(reduced_size, reduced_size);
468 reduced.setFromTriplets(entries.begin(), entries.end());
469 reduced.makeCompressed();
470}

472Eigen::MatrixXd polyfem::utils::reorder_matrix(
473 const Eigen::MatrixXd &in,
474 const Eigen::VectorXi &in_to_out,
475 int out_blocks,
476 const int block_size)
477{
478 constexpr double NaN = std::numeric_limits<double>::quiet_NaN();
479
480 assert(in.rows() % block_size == 0);
481 assert(in_to_out.size() == in.rows() / block_size);
482
483 if (out_blocks < 0)
484 out_blocks = in.rows() / block_size;
485
486 Eigen::MatrixXd out = Eigen::MatrixXd::Constant(
487 out_blocks * block_size, in.cols(), NaN);
488
489 const int in_blocks = in.rows() / block_size;
490 for (int i = 0; i < in_blocks; ++i)
491 {
492 const int j = in_to_out[i];
493 if (j < 0)
494 continue;
495
496 out.middleRows(block_size * j, block_size) =
497 in.middleRows(block_size * i, block_size);
498 }
499
500 return out;
501}
