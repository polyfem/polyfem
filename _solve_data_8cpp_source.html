459 void SolveData::update_barrier_stiffness(const Eigen::VectorXd &x)
460 {
461 if (contact_form == nullptr || !contact_form->use_adaptive_barrier_stiffness())
462 return;
463
464 Eigen::VectorXd grad_energy = Eigen::VectorXd::Zero(x.size());
465 const std::array<std::shared_ptr<Form>, 4> energy_forms{
466 {elastic_form, inertia_form, body_form, pressure_form}};
467 for (const std::shared_ptr<Form> &form : energy_forms)
468 {
469 if (form == nullptr || !form->enabled())
470 continue;
471
472 Eigen::VectorXd grad_form;
473 form->first_derivative(x, grad_form);
474 grad_energy += grad_form;
475 }
476
477 contact_form->update_barrier_stiffness(x, grad_energy);
478 }

480 void SolveData::update_dt()
481 {
482 if (time_integrator == nullptr) // if is not time dependent
483 return;
484
485 const std::array<std::shared_ptr<Form>, 6> energy_forms{
486 {elastic_form, body_form, pressure_form, damping_form, contact_form, friction_form}};
487 for (const std::shared_ptr<Form> &form : energy_forms)
488 {
489 if (form == nullptr)
490 continue;
491 form->set_weight(time_integrator->acceleration_scaling());
492 }
