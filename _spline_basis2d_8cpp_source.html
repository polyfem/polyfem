344 void basis_for_irregulard_quad(const int el_id, const Mesh2D &mesh, MeshNodes &mesh_nodes, const SpaceMatrix &space, const NodeMatrix &loc_nodes, const std::array<std::array<double, 4>, 3> &h_knots, const std::array<std::array<double, 4>, 3> &v_knots, ElementBases &b)
345 {
346 for (int y = 0; y < 3; ++y)
347 {
348 for (int x = 0; x < 3; ++x)
349 {
350 if (space(x, y).size() > 1)
351 {
352 const int mpx = 1;
353 const int mpy = y;
354
355 const int mmx = x;
356 const int mmy = 1;
357
358 std::vector<int> other_indices;
359 const auto &center = b.bases[1 * 3 + 1].global().front();
360
361 const auto &el1 = b.bases[mpy * 3 + mpx].global().front();
362 const auto &el2 = b.bases[mmy * 3 + mmx].global().front();
363
364 Navigation::Index start_index = mesh.get_index_from_face(el_id);
365 bool found = false;
366 for (int i = 0; i < 4; ++i)
367 {
368 other_indices.clear();
369 int n_neighs = 0;
370 Navigation::Index index = start_index;
371 do
372 {
373 const int f_index = mesh_nodes.node_id_from_face(index.face);
374 if (f_index != el1.index && f_index != el2.index && f_index != center.index)
375 other_indices.push_back(f_index);
376
377 ++n_neighs;
378 index = mesh.next_around_vertex(index);
379 } while (index.face != start_index.face);
380 if (n_neighs != 4)
381 {
382 found = true;
383 break;
384 }
385
386 start_index = mesh.next_around_face(start_index);
387 }
388 assert(found);
389
390 const int local_index = y * 3 + x;
391 auto &base = b.bases[local_index];
392
393 const int k = int(other_indices.size()) + 3;
394
395 base.global().resize(k);
396
397 base.global()[0].index = center.index;
398 base.global()[0].val = (4. - k) / k;
399 base.global()[0].node = center.node;
400
401 base.global()[1].index = el1.index;
402 base.global()[1].val = (4. - k) / k;
403 base.global()[1].node = el1.node;
404
405 base.global()[2].index = el2.index;
406 base.global()[2].val = (4. - k) / k;
407 base.global()[2].node = el2.node;
408
409 for (std::size_t n = 0; n < other_indices.size(); ++n)
410 {
411 base.global()[3 + n].index = other_indices[n];
412 base.global()[3 + n].val = 4. / k;
413 base.global()[3 + n].node = mesh_nodes.node_position(other_indices[n]);
414 }
415
416 const QuadraticBSpline2d spline(h_knots[x], v_knots[y]);
417 b.bases[local_index].set_basis([spline](const Eigen::MatrixXd &uv, Eigen::MatrixXd &val) { spline.interpolate(uv, val); });
418 b.bases[local_index].set_grad([spline](const Eigen::MatrixXd &uv, Eigen::MatrixXd &val) { spline.derivative(uv, val); });
419 }
420 }

548 void insert_into_global(const Local2Global &data, std::vector<Local2Global> &vec)
549 {
550 //ignore small weights

552 return;
