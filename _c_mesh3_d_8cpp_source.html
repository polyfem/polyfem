384 bool CMesh3D::build_from_matrices(const Eigen::MatrixXd &V, const Eigen::MatrixXi &F)
385 {
386 assert(F.cols() == 4 || F.cols() == 8);
387 edge_nodes_.clear();
388 face_nodes_.clear();
389 cell_nodes_.clear();
390
391 GEO::Mesh M;
392 M.vertices.create_vertices((int)V.rows());
393 for (int i = 0; i < (int)M.vertices.nb(); ++i)
394 {
395 GEO::vec3 &p = M.vertices.point(i);
396 p[0] = V(i, 0);
397 p[1] = V(i, 1);
398 p[2] = V(i, 2);
399 }
400
401 if (F.cols() == 4)
402 M.cells.create_tets((int)F.rows());
403 else if (F.cols() == 8)
404 M.cells.create_hexes((int)F.rows());
405 else
406 {
407 throw std::runtime_error("Mesh format not supported");
408 }
409
410 static const std::vector<int> permute_tet = {0, 1, 2, 3};
411 // polyfem uses the msh file format for hexes ordering!
412 static const std::vector<int> permute_hex = {1, 0, 2, 3, 5, 4, 6, 7};
413 const std::vector<int> permute = F.cols() == 4 ? permute_tet : permute_hex;
414
415 for (int c = 0; c < (int)M.cells.nb(); ++c)
416 {
417 for (int lv = 0; lv < F.cols(); ++lv)
418 {
419 M.cells.set_vertex(c, lv, F(c, permute[lv]));
420 }
421 }
422 M.cells.connect();
423
424 return load(M);

926 void CMesh3D::normalize()
927 {
928 RowVectorNd minV, maxV;
929 bounding_box(minV, maxV);
930 auto &V = mesh_.points;
931 const auto shift = V.rowwise().minCoeff().eval();
932 const double scaling = 1.0 / (V.rowwise().maxCoeff() - V.rowwise().minCoeff()).maxCoeff();
933 V = (V.colwise() - shift) * scaling;
934
935 for (int i = 0; i < n_cells(); ++i)
936 {
937 for (int d = 0; d < 3; ++d)
938 {
939 auto val = mesh_.elements[i].v_in_Kernel[d];
940 mesh_.elements[i].v_in_Kernel[d] = (val - shift(d)) * scaling;
941 }
942 }
943
944 for (auto &n : edge_nodes_)
945 {
946 if (n.nodes.size() > 0)
947 n.nodes = (n.nodes.rowwise() - shift.transpose()) * scaling;
948 }
949 for (auto &n : face_nodes_)
950 {
951 if (n.nodes.size() > 0)
952 n.nodes = (n.nodes.rowwise() - shift.transpose()) * scaling;
953 }
954 for (auto &n : cell_nodes_)
955 {
956 if (n.nodes.size() > 0)
957 n.nodes = (n.nodes.rowwise() - shift.transpose()) * scaling;
958 }
959
960 logger().debug("-- bbox before normalization:");
961 logger().debug(" min : {}", minV);
962 logger().debug(" max : {}", maxV);
963 logger().debug(" extent: {}", maxV - minV);
964 bounding_box(minV, maxV);
965 logger().debug("-- bbox after normalization:");
966 logger().debug(" min : {}", minV);
967 logger().debug(" max : {}", maxV);
968 logger().debug(" extent: {}", maxV - minV);
969

971
972 // for(int i = 0; i < n_cells(); ++i)

974 // mesh_.elements[i].v_in_Kernel[1] /= 100.;
975 // }

986 void CMesh3D::triangulate_faces(Eigen::MatrixXi &tris, Eigen::MatrixXd &pts, std::vector<int> &ranges) const
987 {
988 ranges.clear();
989
990 std::vector<Eigen::MatrixXi> local_tris(mesh_.elements.size());
991 std::vector<Eigen::MatrixXd> local_pts(mesh_.elements.size());
992 Eigen::MatrixXi tets;
993
994 int total_tris = 0;
995 int total_pts = 0;
996
997 ranges.push_back(0);
998
999 Eigen::MatrixXd face_barys;
1000 face_barycenters(face_barys);
1001
1002 Eigen::MatrixXd cell_barys;
1003 cell_barycenters(cell_barys);
1004
1005 for (std::size_t e = 0; e < mesh_.elements.size(); ++e)
1006 {
1007 const Element &el = mesh_.elements[e];
1008
1009 const int n_vertices = el.vs.size();
1010 const int n_faces = el.fs.size();
1011
1012 Eigen::MatrixXd local_pt(n_vertices + n_faces, 3);
1013
1014 std::map<int, int> global_to_local;
1015
1016 for (int i = 0; i < n_vertices; ++i)
1017 {
1018 const int global_index = el.vs[i];
1019 local_pt.row(i) = mesh_.points.col(global_index).transpose();
1020 global_to_local[global_index] = i;
1021 }
1022
1023 int n_local_faces = 0;
1024 for (int i = 0; i < n_faces; ++i)
1025 {
1026 const Face &f = mesh_.faces[el.fs[i]];
1027 n_local_faces += f.vs.size();
1028
1029 local_pt.row(n_vertices + i) = face_barys.row(f.id); // node_from_face(f.id);
1030 }
1031
1032 Eigen::MatrixXi local_faces(n_local_faces, 3);
1033
1034 int face_index = 0;
1035 for (int i = 0; i < n_faces; ++i)
1036 {
1037 const Face &f = mesh_.faces[el.fs[i]];
1038 const int n_face_vertices = f.vs.size();
1039
1040 const Eigen::RowVector3d e0 = (point(f.vs[0]) - local_pt.row(n_vertices + i));
1041 const Eigen::RowVector3d e1 = (point(f.vs[1]) - local_pt.row(n_vertices + i));
1042 const Eigen::RowVector3d normal = e0.cross(e1);
1043 // const Eigen::RowVector3d check_dir = (node_from_element(e)-p);
1044 const Eigen::RowVector3d check_dir = (cell_barys.row(e) - point(f.vs[1]));
1045
1046 const bool reverse_order = normal.dot(check_dir) > 0;
1047
1048 for (int j = 0; j < n_face_vertices; ++j)
1049 {
1050 const int jp = (j + 1) % n_face_vertices;
1051 if (reverse_order)
1052 {
1053 local_faces(face_index, 0) = global_to_local[f.vs[jp]];
1054 local_faces(face_index, 1) = global_to_local[f.vs[j]];
1055 }
1056 else
1057 {
1058 local_faces(face_index, 0) = global_to_local[f.vs[j]];
1059 local_faces(face_index, 1) = global_to_local[f.vs[jp]];
1060 }
1061 local_faces(face_index, 2) = n_vertices + i;
1062
1063 ++face_index;
1064 }
1065 }
1066
1067 local_pts[e] = local_pt;
1068 local_tris[e] = local_faces;
1069
1070 total_tris += local_tris[e].rows();
1071 total_pts += local_pts[e].rows();
1072
1073 ranges.push_back(total_tris);
1074
1075 assert(local_pts[e].rows() == local_pt.rows());
1076 }
1077
1078 tris.resize(total_tris, 3);
1079 pts.resize(total_pts, 3);
1080
1081 int tri_index = 0;
1082 int pts_index = 0;
1083 for (std::size_t i = 0; i < local_tris.size(); ++i)
1084 {
1085 tris.block(tri_index, 0, local_tris[i].rows(), local_tris[i].cols()) = local_tris[i].array() + pts_index;
1086 tri_index += local_tris[i].rows();
1087
1088 pts.block(pts_index, 0, local_pts[i].rows(), local_pts[i].cols()) = local_pts[i];
1089 pts_index += local_pts[i].rows();
1090 }
1091 }

1227 void CMesh3D::compute_elements_tag()
1228 {
1229 std::vector<ElementType> &ele_tag = elements_tag_;
1230 ele_tag.clear();
1231
1232 ele_tag.resize(mesh_.elements.size());
1233 for (auto &t : ele_tag)
1234 t = ElementType::RegularInteriorCube;
1235
1236 // boundary flags
1237 std::vector<bool> bv_flag(mesh_.vertices.size(), false), be_flag(mesh_.edges.size(), false), bf_flag(mesh_.faces.size(), false);
1238 for (auto f : mesh_.faces)
1239 if (f.boundary)
1240 bf_flag[f.id] = true;
1241 else
1242 {
1243 for (auto nhid : f.neighbor_hs)
1244 if (!mesh_.elements[nhid].hex)
1245 bf_flag[f.id] = true;
1246 }
1247 for (uint32_t i = 0; i < mesh_.faces.size(); ++i)
1248 if (bf_flag[i])
1249 for (uint32_t j = 0; j < mesh_.faces[i].vs.size(); ++j)
1250 {
1251 uint32_t eid = mesh_.faces[i].es[j];
1252 be_flag[eid] = true;
1253 bv_flag[mesh_.faces[i].vs[j]] = true;
1254 }
1255
1256 for (auto &ele : mesh_.elements)
1257 {
1258 if (ele.hex)
1259 {
1260 bool attaching_non_hex = false, on_boundary = false;
1261 ;
1262 for (auto vid : ele.vs)
1263 {
1264 for (auto eleid : mesh_.vertices[vid].neighbor_hs)
1265 if (!mesh_.elements[eleid].hex)
1266 {
1267 attaching_non_hex = true;
1268 break;
1269 }
1270 if (mesh_.vertices[vid].boundary)
1271 {
1272 on_boundary = true;
1273 break;
1274 }
1275 if (on_boundary || attaching_non_hex)
1276 break;
1277 }
1278 if (attaching_non_hex)
1279 {
1280 ele_tag[ele.id] = ElementType::InterfaceCube;
1281 continue;
1282 }
1283
1284 if (on_boundary)
1285 {
1286 ele_tag[ele.id] = ElementType::MultiSingularBoundaryCube;
1287 // has no boundary edge--> singular

1289 int n_interior_edge_singular = 0;
1290 for (auto eid : ele.es)
1291 {
1292 int en = 0;
1293 if (be_flag[eid])
1294 {
1295 boundary_edge = true;
1296 for (auto nhid : mesh_.edges[eid].neighbor_hs)
1297 if (mesh_.elements[nhid].hex)
1298 en++;
1299 if (en > 2)
1300 boundary_edge_singular = true;
1301 }
1302 else
1303 {
1304 for (auto nhid : mesh_.edges[eid].neighbor_hs)
1305 if (mesh_.elements[nhid].hex)
1306 en++;
1307 if (en != 4)
1308 {
1309 interior_edge_singular = true;
1310 n_interior_edge_singular++;
1311 }
1312 }
1313 }
1314 if (!boundary_edge || boundary_edge_singular || n_interior_edge_singular > 1)
1315 continue;
1316
1317 bool has_singular_v = false, has_iregular_v = false;
1318 int n_in_irregular_v = 0;
1319 for (auto vid : ele.vs)
1320 {
1321 int vn = 0;
1322 if (bv_flag[vid])
1323 {
1324 int nh = 0;
1325 for (auto nhid : mesh_.vertices[vid].neighbor_hs)
1326 if (mesh_.elements[nhid].hex)
1327 nh++;
1328 if (nh > 4)
1329 has_iregular_v = true;
1330 continue; // not sure the conditions
1331 }
1332 else
1333 {
1334 if (mesh_.vertices[vid].neighbor_hs.size() != 8)
