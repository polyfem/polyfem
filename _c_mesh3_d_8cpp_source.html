464 bool CMesh3D::build_from_matrices(const Eigen::MatrixXd &V, const Eigen::MatrixXi &F)
465 {
466 assert(F.cols() == 4 || F.cols() == 8);
467 edge_nodes_.clear();
468 face_nodes_.clear();
469 cell_nodes_.clear();
470
471 GEO::Mesh M;
472 M.vertices.create_vertices((int)V.rows());
473 for (int i = 0; i < (int)M.vertices.nb(); ++i)
474 {
475 GEO::vec3 &p = M.vertices.point(i);
476 p[0] = V(i, 0);
477 p[1] = V(i, 1);
478 p[2] = V(i, 2);
479 }
480
481 if (F.cols() == 4)
482 M.cells.create_tets((int)F.rows());
483 else if (F.cols() == 8)
484 M.cells.create_hexes((int)F.rows());
485 else
486 {
487 throw std::runtime_error("Mesh format not supported");
488 }
489
490 static const std::vector<int> permute_tet = {0, 1, 2, 3};
491 // polyfem uses the msh file format for hexes ordering!
492 static const std::vector<int> permute_hex = {1, 0, 2, 3, 5, 4, 6, 7};
493 const std::vector<int> permute = F.cols() == 4 ? permute_tet : permute_hex;
494
495 for (int c = 0; c < (int)M.cells.nb(); ++c)
496 {
497 for (int lv = 0; lv < F.cols(); ++lv)
498 {
499 M.cells.set_vertex(c, lv, F(c, permute[lv]));
500 }
501 }
502 M.cells.connect();
503
504 return load(M);
