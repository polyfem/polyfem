215 std::shared_ptr<Parametrization> AdjointOptUtils::create_parametrization(const json &args, const std::vector<std::shared_ptr<State>> &states, const std::vector<int> &variable_sizes)
216 {
217 std::shared_ptr<Parametrization> map;
218 const std::string type = args["type"];
219 if (type == "per-body-to-per-elem")
220 {
221 map = std::make_shared<PerBody2PerElem>(*(states[args["state"]]->mesh));
222 }
223 else if (type == "per-body-to-per-node")
224 {
225 map = std::make_shared<PerBody2PerNode>(*(states[args["state"]]->mesh), states[args["state"]]->bases, states[args["state"]]->n_bases);
226 }
227 else if (type == "E-nu-to-lambda-mu")
228 {
229 map = std::make_shared<ENu2LambdaMu>(args["is_volume"]);
230 }
231 else if (type == "slice")
232 {
233 if (args.contains("from") && args.contains("to"))
234 {
235 assert(args["from"] != -1);
236 assert(args["to"] != -1);
237 map = std::make_shared<SliceMap>(args["from"], args["to"], args["last"]);
238 }
239 else if (args.contains("parameter_index"))
240 {
241 assert(args["parameter_index"] != -1);
242 int idx = args["parameter_index"].get<int>();

244 int cumulative = 0;
245 for (int i = 0; i < variable_sizes.size(); ++i)
246 {
247 if (i == idx)
248 {
249 from = cumulative;
250 to = from + variable_sizes[i];
251 }
252 cumulative += variable_sizes[i];
253 }
254 last = cumulative;
255 map = std::make_shared<SliceMap>(from, to, last);
256 }
257 else
258 log_and_throw_error("Incorrect spec for SliceMap!");
259 }
260 else if (type == "exp")
261 {
262 map = std::make_shared<ExponentialMap>();
263 }
264 else if (type == "scale")
265 {
266 map = std::make_shared<Scaling>(args["value"]);
267 }
268 else if (type == "power")

271 }

274 Eigen::VectorXd vals;
275 nlohmann::adl_serializer<Eigen::VectorXd>::from_json(args["values"], vals);
276 map = std::make_shared<InsertConstantMap>(vals);
277 }
278 else if (type == "append-const")
279 {
280 map = std::make_shared<InsertConstantMap>(args["size"], args["value"], args["start"]);
281 }
282 else if (type == "linear-filter")
283 {
284 map = std::make_shared<LinearFilter>(*(states[args["state"]]->mesh), args["radius"]);
285 }
286 else if (type == "custom-symmetric")
287 {
288 map = std::make_shared<CustomSymmetric>(args);
289 }
290 else if (type == "periodic-mesh-tile")
291 {
292 Eigen::VectorXi dims;
293 nlohmann::adl_serializer<Eigen::VectorXi>::from_json(args["dimensions"], dims);

313 log_and_throw_error("Unkown parametrization!");

531 int AdjointOptUtils::compute_variable_size(const json &args, const std::vector<std::shared_ptr<State>> &states)
532 {
533 if (args["number"].is_number())
534 {
535 return args["number"].get<int>();
536 }
537 else if (args["number"].is_null() && args["initial"].size() > 0)
538 {
539 return args["initial"].size();
540 }
541 else if (args["number"].is_object())
542 {
543 auto selection = args["number"];
544 if (selection.contains("surface_selection"))
545 {
546 auto surface_selection = selection["surface_selection"].get<std::vector<int>>();
547 auto state_id = selection["state"];
548 std::set<int> node_ids = {};
549 for (const auto &surface : surface_selection)
550 {
551 std::vector<int> ids;
552 states[state_id]->compute_surface_node_ids(surface, ids);
553 for (const auto &i : ids)
554 node_ids.insert(i);
555 }
556 return node_ids.size() * states[state_id]->mesh->dimension();
557 }
558 else if (selection.contains("volume_selection"))
559 {
560 auto volume_selection = selection["volume_selection"].get<std::vector<int>>();
561 auto state_id = selection["state"];
562 std::set<int> node_ids = {};
563 for (const auto &volume : volume_selection)
564 {
565 std::vector<int> ids;
566 states[state_id]->compute_volume_node_ids(volume, ids);
567 for (const auto &i : ids)
568 node_ids.insert(i);
569 }
570
571 if (selection["exclude_boundary_nodes"])
572 {
573 std::vector<int> ids;
574 states[state_id]->compute_total_surface_node_ids(ids);
575 for (const auto &i : ids)
576 node_ids.erase(i);
577 }
578
579 return node_ids.size() * states[state_id]->mesh->dimension();
580 }
581 else
582 {
583 log_and_throw_error("Incorrect specification for parameters.");
584 }
585 }
586 else
587 log_and_throw_error("Incorrect specification for parameters.");
588
589 return -1;
590 }
