80 Eigen::VectorXd AdjointOptUtils::inverse_evaluation(const json &args, const int ndof, const std::vector<int> &variable_sizes, VariableToSimulationGroup &var2sim)
81 {
82 Eigen::VectorXd x;
83 x.setZero(ndof);
84 int accumulative = 0;
85 int var = 0;
86 for (const auto &arg : args)
87 {
88 const auto &arg_initial = arg["initial"];
89 Eigen::VectorXd tmp(variable_sizes[var]);
90 if (arg_initial.is_array() && arg_initial.size() > 0)
91 {
92 tmp = arg_initial;
93 x.segment(accumulative, tmp.size()) = tmp;
94 }
95 else if (arg_initial.is_number())
96 {
97 tmp.setConstant(arg_initial.get<double>());
98 x.segment(accumulative, tmp.size()) = tmp;
99 }
100 else // arg["initial"] is empty array
101 x += var2sim.data[var]->inverse_eval();
102
103 accumulative += tmp.size();
104 var++;
105 }
106
107 return x;
108 }

205 int AdjointOptUtils::compute_variable_size(const json &args, const std::vector<std::shared_ptr<State>> &states)
206 {
207 if (args["number"].is_number())
208 {
209 return args["number"].get<int>();
210 }
211 else if (args["number"].is_null() && args["initial"].size() > 0)
212 {
213 return args["initial"].size();
214 }
215 else if (args["number"].is_object())
216 {
217 auto selection = args["number"];
218 if (selection.contains("surface_selection"))
219 {
220 auto surface_selection = selection["surface_selection"].get<std::vector<int>>();
221 auto state_id = selection["state"];
222 std::set<int> node_ids = {};
223 for (const auto &surface : surface_selection)
224 {
225 std::vector<int> ids;
226 compute_surface_node_ids(*states[state_id], surface, ids);
227 for (const auto &i : ids)
228 node_ids.insert(i);
229 }
230 return node_ids.size() * states[state_id]->mesh->dimension();
231 }
232 else if (selection.contains("volume_selection"))
233 {
234 auto volume_selection = selection["volume_selection"].get<std::vector<int>>();
235 auto state_id = selection["state"];
236 std::set<int> node_ids = {};
237 for (const auto &volume : volume_selection)
238 {
239 std::vector<int> ids;
240 compute_volume_node_ids(*states[state_id], volume, ids);
241 for (const auto &i : ids)
242 node_ids.insert(i);
243 }
244
245 if (selection["exclude_boundary_nodes"])
246 {
247 std::vector<int> ids;
248 compute_total_surface_node_ids(*states[state_id], ids);
249 for (const auto &i : ids)
250 node_ids.erase(i);
251 }
252
253 return node_ids.size() * states[state_id]->mesh->dimension();
254 }
255 }
256
257 log_and_throw_adjoint_error("Incorrect specification for parameters.");
258 return -1;
259 }
