123 void MVPolygonalBasis2d::meanvalue(const Eigen::MatrixXd &polygon, const Eigen::RowVector2d &point, Eigen::MatrixXd &b, const double tol)
124 {
125 const int n_boundary = polygon.rows();
126
127 Eigen::MatrixXd segments(n_boundary, 2);
128 Eigen::VectorXd radii(n_boundary);
129 Eigen::VectorXd areas(n_boundary);
130 Eigen::VectorXd products(n_boundary);
131 Eigen::VectorXd tangents(n_boundary);
132 Eigen::Matrix2d mat;
133
134 b.resize(n_boundary, 1);
135 b.setZero();
136
137 for (int i = 0; i < n_boundary; ++i)
138 {
139 segments.row(i) = polygon.row(i) - point;
140
141 radii(i) = segments.row(i).norm();
142
143 //we are on the vertex
144 if (radii(i) < tol)
145 {
146 b(i) = 1;
147
148 return;
149 }
150 }
151
152 for (int i = 0; i < n_boundary; ++i)
153 {
154 const int ip1 = (i + 1) == n_boundary ? 0 : (i + 1);
155
156 mat.row(0) = segments.row(i);
157 mat.row(1) = segments.row(ip1);
158
159 areas(i) = mat.determinant();
160 products(i) = segments.row(i).dot(segments.row(ip1));
161
162 //we are on the edge
163 if (fabs(areas[i]) < tol && products(i) < 0)
164 {
165 const double denominator = 1.0 / (radii(i) + radii(ip1));
166
167 b(i) = radii(ip1) * denominator;
168 b(ip1) = radii(i) * denominator;
169
170 return;
171 }
172 }
173
174 for (int i = 0; i < n_boundary; ++i)
175 {
176 const int ip1 = (i + 1) == n_boundary ? 0 : (i + 1);
177
178 tangents(i) = areas(i) / (radii(i) * radii(ip1) + products(i));
179 }
180
181 double W = 0;
182 for (int i = 0; i < n_boundary; ++i)
183 {
184 const int im1 = i == 0 ? (n_boundary - 1) : (i - 1);
185
186 b(i) = (tangents(im1) + tangents(i)) / radii(i);
187 W += b(i);
188 }
189
190 b /= W;
191 }
