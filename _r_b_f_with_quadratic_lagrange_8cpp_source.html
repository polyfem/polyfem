25 // Harmonic kernel
26 double kernel(const bool is_volume, const double r)
27 {
28 if (r < 1e-8)
29 {
30 return 0;
31 }
32
33 if (is_volume)
34 {
35 return 1 / r;
36 }
37 else
38 {
39 return log(r);
40 }

43 double kernel_prime(const bool is_volume, const double r)
44 {
45 if (r < 1e-8)
46 {
47 return 0;
48 }
49
50 if (is_volume)
51 {
52 return -1 / (r * r);
53 }
54 else

56 return 1 / r;
57 }

131void RBFWithQuadraticLagrange::bases_grads(const int axis, const Eigen::MatrixXd &samples, Eigen::MatrixXd &val) const
132{
133 const int num_kernels = centers_.rows();
134 const int dim = (is_volume() ? 3 : 2);
135
136 // Compute ∇xA
137 Eigen::MatrixXd A_prime(samples.rows(), num_kernels + 1 + dim + dim * (dim + 1) / 2);
138 A_prime.setZero();
139
140 for (int j = 0; j < num_kernels; ++j)
141 {
142 A_prime.col(j) = (samples.rowwise() - centers_.row(j)).rowwise().norm().unaryExpr([this](double x) { return kernel_prime(is_volume(), x) / x; });
143 A_prime.col(j) = (samples.col(axis).array() - centers_(j, axis)) * A_prime.col(j).array();
144 }
145 // Linear terms
146 A_prime.middleCols(num_kernels + 1 + axis, 1).setOnes();
147 // Mixed terms
148 if (dim == 2)
149 {
150 A_prime.col(num_kernels + 1 + dim) = samples.col(1 - axis);
151 }
152 else
153 {
154 A_prime.col(num_kernels + 1 + dim + axis) = samples.col((axis + 1) % dim);
155 A_prime.col(num_kernels + 1 + dim + (axis + 2) % dim) = samples.col((axis + 2) % dim);
156 }
157 // Quadratic terms
158 A_prime.rightCols(dim).col(axis) = 2.0 * samples.col(axis);
159
160 // Apply weights
161 val = A_prime * weights_;
162}

166void RBFWithQuadraticLagrange::compute_kernels_matrix(const Eigen::MatrixXd &samples, Eigen::MatrixXd &A) const
167{
168 // Compute A
169 const int num_kernels = centers_.rows();
170 const int dim = (is_volume() ? 3 : 2);
171
172 A.resize(samples.rows(), num_kernels + 1 + dim + dim * (dim + 1) / 2);
173 for (int j = 0; j < num_kernels; ++j)
174 {
175 A.col(j) = (samples.rowwise() - centers_.row(j)).rowwise().norm().unaryExpr([this](double x) { return kernel(is_volume(), x); });
176 }
177 A.col(num_kernels).setOnes(); // constant term
178 A.middleCols(num_kernels + 1, dim) = samples; // linear terms
179 if (dim == 2)
180 {
181 A.middleCols(num_kernels + dim + 1, 1) = samples.rowwise().prod(); // mixed terms
182 }
183 else if (dim == 3)
184 {
185 A.middleCols(num_kernels + dim + 1, 3) = samples;
186 A.middleCols(num_kernels + dim + 1 + 0, 1).array() *= samples.col(1).array();
187 A.middleCols(num_kernels + dim + 1 + 1, 1).array() *= samples.col(2).array();
188 A.middleCols(num_kernels + dim + 1 + 2, 1).array() *= samples.col(0).array();
189 }
190 A.rightCols(dim) = samples.array().square(); // quadratic terms
191}

195void RBFWithQuadraticLagrange::compute_constraints_matrix_2d_old(
196 const int num_bases, const Quadrature &quadr, Eigen::MatrixXd &C) const
197{
198 const int num_kernels = centers_.rows();
199 const int dim = centers_.cols();
200 assert(dim == 2);
201
202 // K_cst = ∫φj
203 // K_lin = ∫∇x(φj), ∫∇y(φj)
204 // K_mix = ∫y·∇x(φj), ∫x·∇y(φj)
205 // K_sqr = ∫x·∇x(φj), ∫y·∇y(φj)
206 Eigen::VectorXd K_cst = Eigen::VectorXd::Zero(num_kernels);
207 Eigen::MatrixXd K_lin = Eigen::MatrixXd::Zero(num_kernels, dim);
208 Eigen::MatrixXd K_mix = Eigen::MatrixXd::Zero(num_kernels, dim);
209 Eigen::MatrixXd K_sqr = Eigen::MatrixXd::Zero(num_kernels, dim);
210 for (int j = 0; j < num_kernels; ++j)
211 {
212 // ∫∇x(φj)(p) = Σ_q (xq - xk) * 1/r * h'(r) * wq
213 // - xq is the x coordinate of the q-th quadrature point
214 // - wq is the q-th quadrature weight
215 // - r is the distance from pq to the kernel center
216 // - h is the RBF kernel (scalar function)
217 for (int q = 0; q < quadr.points.rows(); ++q)
218 {
219 const RowVectorNd p = quadr.points.row(q) - centers_.row(j);
220 const double r = p.norm();
221 const RowVectorNd gradPhi = p * kernel_prime(is_volume(), r) / r * quadr.weights(q);
222 K_cst(j) += kernel(is_volume(), r) * quadr.weights(q);
223 K_lin.row(j) += gradPhi;
224 K_mix(j, 0) += quadr.points(q, 1) * gradPhi(0);
225 K_mix(j, 1) += quadr.points(q, 0) * gradPhi(1);
226 K_sqr.row(j) += (quadr.points.row(q).array() * gradPhi.array()).matrix();
227 }
228 }
229
230 // I_lin = ∫x, ∫y
231 // I_mix = ∫xy
232 // I_sqr = ∫x², ∫y²
233 Eigen::RowVectorXd I_lin = (quadr.points.array().colwise() * quadr.weights.array()).colwise().sum();
234 Eigen::RowVectorXd I_mix = (quadr.points.rowwise().prod().array() * quadr.weights.array()).colwise().sum();
235 Eigen::RowVectorXd I_sqr = (quadr.points.array().square().colwise() * quadr.weights.array()).colwise().sum();
236 double volume = quadr.weights.sum();
237
238 //TODO
239 // std::cout << I_lin << std::endl;
240 // std::cout << I_mix << std::endl;
241 // std::cout << I_sqr << std::endl;
242
243 // Compute M
244 Eigen::Matrix<double, 5, 5> M;
245 M << volume, 0, I_lin(1), 2 * I_lin(0), 0,
246 0, volume, I_lin(0), 0, 2 * I_lin(1),
247 I_lin(1), I_lin(0), I_sqr(0) + I_sqr(1), 2 * I_mix(0), 2 * I_mix(0),
248 4 * I_lin(0), 2 * I_lin(1), 4 * I_mix(0), 6 * I_sqr(0), 2 * I_sqr(1),
249 2 * I_lin(0), 4 * I_lin(1), 4 * I_mix(0), 2 * I_sqr(0), 6 * I_sqr(1);
250 Eigen::FullPivLU<Eigen::Matrix<double, 5, 5>> lu(M);
251
252 show_matrix_stats(M);
253
254 // Compute L
255 C.resize(5, num_kernels + 1 + dim + dim * (dim + 1) / 2);
256 C.setZero();
257 C.block(0, 0, dim, num_kernels) = K_lin.transpose();
258 C.block(dim, 0, 1, num_kernels) = K_mix.transpose().colwise().sum();
259 C.block(dim + 1, 0, dim, num_kernels) = 2.0 * (K_sqr.colwise() + K_cst).transpose();
260 C.block(dim + 1, num_kernels, dim, 1).setConstant(2.0 * volume);
261 C.bottomRightCorner(5, 5) = M;
262 // std::cout << L.bottomRightCorner(10, 10) << std::endl;
263}

370void RBFWithQuadraticLagrange::compute_constraints_matrix_3d(
371 const int num_bases, const Quadrature &quadr, Eigen::MatrixXd &C) const
372{
373 const int num_kernels = centers_.rows();
374 const int dim = centers_.cols();
375 assert(dim == 3);
376
377 // K_cst = ∫φj
378 // K_lin = ∫∇x(φj), ∫∇y(φj), ∫∇z(φj)
379 // K_mix = ∫(y·∇x(φj)+x·∇y(φj)), ∫(z·∇y(φj)+y·∇z(φj)), ∫(x·∇z(φj)+z·∇x(φj))
380 // K_sqr = ∫x·∇x(φj), ∫y·∇y(φj), ∫z·∇z(φj)
381 Eigen::VectorXd K_cst = Eigen::VectorXd::Zero(num_kernels);
382 Eigen::MatrixXd K_lin = Eigen::MatrixXd::Zero(num_kernels, dim);
383 Eigen::MatrixXd K_mix = Eigen::MatrixXd::Zero(num_kernels, dim);
384 Eigen::MatrixXd K_sqr = Eigen::MatrixXd::Zero(num_kernels, dim);
385 for (int j = 0; j < num_kernels; ++j)
386 {
387 // ∫∇x(φj)(p) = Σ_q (xq - xk) * 1/r * h'(r) * wq
388 // - xq is the x coordinate of the q-th quadrature point
389 // - wq is the q-th quadrature weight
390 // - r is the distance from pq to the kernel center
391 // - h is the RBF kernel (scalar function)
392 for (int q = 0; q < quadr.points.rows(); ++q)
393 {
394 const RowVectorNd p = quadr.points.row(q) - centers_.row(j);
395 const double r = p.norm();
396 const RowVectorNd gradPhi = p * kernel_prime(is_volume(), r) / r * quadr.weights(q);
397 K_cst(j) += kernel(is_volume(), r) * quadr.weights(q);
398 K_lin.row(j) += gradPhi;
399 for (int d = 0; d < dim; ++d)
400 {
401 K_mix(j, d) += quadr.points(q, (d + 1) % dim) * gradPhi(d) + quadr.points(q, d) * gradPhi((d + 1) % dim);
402 }
403 K_sqr.row(j) += (quadr.points.row(q).array() * gradPhi.array()).matrix();
404 }
405 }
406
407 // I_lin = ∫x, ∫y, ∫z
408 // I_sqr = ∫x², ∫y², ∫z²
409 // I_mix = ∫xy, ∫yz, ∫zx
410 Eigen::RowVectorXd I_lin = (quadr.points.array().colwise() * quadr.weights.array()).colwise().sum();
411 Eigen::RowVectorXd I_sqr = (quadr.points.array().square().colwise() * quadr.weights.array()).colwise().sum();
412 Eigen::RowVectorXd I_mix(3);
413 I_mix(0) = (quadr.points.col(0).array() * quadr.points.col(1).array() * quadr.weights.array()).sum();
414 I_mix(1) = (quadr.points.col(1).array() * quadr.points.col(2).array() * quadr.weights.array()).sum();
415 I_mix(2) = (quadr.points.col(2).array() * quadr.points.col(0).array() * quadr.weights.array()).sum();
416 double volume = quadr.weights.sum();
417
418 // std::cout << I_lin << std::endl;
419 // std::cout << I_mix << std::endl;
420 // std::cout << I_sqr << std::endl;
421
422 // Compute M
423 Eigen::Matrix<double, 9, 9> M;
424 M << volume, 0, 0, I_lin(1), 0, I_lin(2), 2 * I_lin(0), 0, 0,
425 0, volume, 0, I_lin(0), I_lin(2), 0, 0, 2 * I_lin(1), 0,
426 0, 0, volume, 0, I_lin(1), I_lin(0), 0, 0, 2 * I_lin(2),
427 I_lin(1), I_lin(0), 0, I_sqr(0) + I_sqr(1), I_mix(2), I_mix(1), 2 * I_mix(0), 2 * I_mix(0), 0,
428 0, I_lin(2), I_lin(1), I_mix(2), I_sqr(1) + I_sqr(2), I_mix(0), 0, 2 * I_mix(1), 2 * I_mix(1),
429 I_lin(2), 0, I_lin(0), I_mix(1), I_mix(0), I_sqr(2) + I_sqr(0), 2 * I_mix(2), 0, 2 * I_mix(2),
430 2 * I_lin(0), 0, 0, 2 * I_mix(0), 0, 2 * I_mix(2), 4 * I_sqr(0), 0, 0,
431 0, 2 * I_lin(1), 0, 2 * I_mix(0), 2 * I_mix(1), 0, 0, 4 * I_sqr(1), 0,
432 0, 0, 2 * I_lin(2), 0, 2 * I_mix(1), 2 * I_mix(2), 0, 0, 4 * I_sqr(2);
433 Eigen::Matrix<double, 1, 9> M_rhs;
434 M_rhs.segment<3>(0) = I_lin;
435 M_rhs.segment<3>(3) = I_mix;
436 M_rhs.segment<3>(6) = I_sqr;
437 // M_rhs << I_lin, I_mix, I_sqr;
438 M.bottomRows(dim).rowwise() += 2.0 * M_rhs;
439 //TODO
440 // assert(false);
441
442 show_matrix_stats(M);
443
444 // Compute L
445 C.resize(9, num_kernels + 1 + dim + dim * (dim + 1) / 2);
446 C.setZero();
447 C.block(0, 0, dim, num_kernels) = K_lin.transpose();
448 C.block(dim, 0, dim, num_kernels) = K_mix.transpose();
449 C.block(dim + dim, 0, dim, num_kernels) = 2.0 * (K_sqr.colwise() + K_cst).transpose();
450 C.block(dim + dim, num_kernels, dim, 1).setConstant(2.0 * volume);
451 C.bottomRightCorner(9, 9) = M;
452 // std::cout << C.bottomRightCorner(9, 12) << std::endl;
453}

457void RBFWithQuadraticLagrange::compute_weights(const AssemblerUtils &assembler, const std::string &assembler_name, const Eigen::MatrixXd &samples,
458 const Eigen::MatrixXd &local_basis_integral, const Quadrature &quadr,
459 Eigen::MatrixXd &b, bool with_constraints)
460{
461 logger().trace("#kernel centers: {}", centers_.rows());
462 logger().trace("#collocation points: {}", samples.rows());
463 logger().trace("#quadrature points: {}", quadr.weights.size());
464 logger().trace("#non-vanishing bases: {}", b.cols());
465 logger().trace("#constraints: {}", b.cols());
466
467 // Compute A
468 Eigen::MatrixXd A;
469 compute_kernels_matrix(samples, A);
470
471 if (!with_constraints)
472 {
473 // Solve the system
474 const int num_kernels = centers_.rows();
475 logger().trace("-- Solving system of size {}x{}", num_kernels, num_kernels);
476 weights_ = (A.transpose() * A).ldlt().solve(A.transpose() * b);
477 logger().trace("-- Solved!");
478
479 return;
480 }
481
482 const int num_bases = b.cols();
483
484 const Eigen::MatrixXd At = A.transpose();
485
486 // Compute C
487 Eigen::MatrixXd C;
488 if (is_volume())
489 {
490 compute_constraints_matrix_3d(num_bases, quadr, C);
491 }
492 else
493 {
494 compute_constraints_matrix_2d(assembler, assembler_name, num_bases, quadr, C);
495 }
496
497 const int dim = centers_.cols();
498 assert(centers_.rows() + 1 + dim + dim * (dim + 1) / 2 > C.rows());
499 logger().trace("#constraints: {}", C.rows());
500
501 // Compute rhs = [ A^T b; d ]
502 assert(local_basis_integral.cols() == C.rows());
503 assert(local_basis_integral.rows() == b.cols());
504 assert(A.rows() == b.rows());
505 Eigen::MatrixXd rhs(A.cols() + local_basis_integral.cols(), b.cols());
506 rhs.topRows(A.cols()) = At * b;
507 rhs.bottomRows(local_basis_integral.cols()) = local_basis_integral.transpose();
508
509 // Compute M = [ A^T A, C^T; C, 0]
510 assert(C.cols() == A.cols());
511 assert(A.rows() == b.rows());
512 Eigen::MatrixXd M(A.cols() + C.rows(), A.cols() + C.rows());
513 M.topLeftCorner(A.cols(), A.cols()) = At * A;
514 M.topRightCorner(A.cols(), C.rows()) = C.transpose();
515 M.bottomLeftCorner(C.rows(), A.cols()) = C;
516 M.bottomRightCorner(C.rows(), C.rows()).setZero();
517
518 // std::cout << M.bottomRightCorner(10, 10) << std::endl;
519
520 // Solve the system
521 logger().trace("-- Solving system of size {}x{}", M.rows(), M.cols());
522 auto ldlt = M.ldlt();
523 if (ldlt.info() == Eigen::NumericalIssue)
524 {
525 logger().error("-- WARNING: Numerical issues when solving the harmonic least square.");
526 }
527 const auto tmp = ldlt.solve(rhs);
528 weights_ = tmp.topRows(A.cols());
529 logger().trace("-- Solved!");
530 logger().trace("-- Mean residual: {}", (A * weights_ - b).array().abs().colwise().maxCoeff().mean());
531 logger().trace("-- Max constraints error: {}", (C * weights_ - local_basis_integral.transpose()).array().abs().maxCoeff());
