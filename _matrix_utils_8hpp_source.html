16 T determinant(const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> &mat)
17 {
18 assert(mat.rows() == mat.cols());
19
20 if (mat.rows() == 1)
21 return mat(0);
22 else if (mat.rows() == 2)
23 return mat(0, 0) * mat(1, 1) - mat(0, 1) * mat(1, 0);
24 else if (mat.rows() == 3)
