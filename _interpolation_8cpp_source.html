124 double PiecewiseInterpolation::extend(const double t) const
125 {
126 assert(points_.size() == values_.size());
127 const double t0 = points_.front(), tn = points_.back();
128 assert(t < t0 || t >= tn);
129 const double y0 = values_.front(), yn = values_.back();
130
131 double offset = 0;
132 switch (extend_)
133 {
134 case Extend::CONSTANT:
135 return (t < t0) ? y0 : yn;
136
137 case Extend::EXTRAPOLATE:
138 {
139 if (t < t0)
140 return dy_dt(t0) * (t - t0) + y0;
141 else
142 return dy_dt(tn) * (t - tn) + yn;
143 }
144
145 case Extend::REPEAT_OFFSET:
146 offset = floor((t - t0) / (tn - t0)) * (yn - y0);
147 // NOTE: fallthrough
148 [[fallthrough]]; // suppress warning
149
150 case Extend::REPEAT:
151 {
152 double t_mod = std::fmod(t - t0, tn - t0);
153 t_mod += (t_mod < 0) ? tn : t0;
154 return eval(t_mod) + offset;
155 }
156
157 default:
158 assert(false);
159 return 0;
160 }
