441 void State::init_time()
442 {
443 if (!is_param_valid(args, "time"))
444 return;
445
446 const double t0 = Units::convert(args["time"]["t0"], units.time());
447 double tend, dt;
448 int time_steps;
449
450 // from "tend", "dt", "time_steps" only two can be used at a time
451 const int num_valid = is_param_valid(args["time"], "tend")
452 + is_param_valid(args["time"], "dt")
453 + is_param_valid(args["time"], "time_steps");
454 if (num_valid < 2)
455 {
456 log_and_throw_error("Exactly two of (tend, dt, time_steps) must be specified");
457 }
458 else if (num_valid == 2)
459 {
460 if (is_param_valid(args["time"], "tend"))
461 {
462 tend = Units::convert(args["time"]["tend"], units.time());
463 assert(tend > t0);
464 if (is_param_valid(args["time"], "dt"))
465 {
466 dt = Units::convert(args["time"]["dt"], units.time());
467 assert(dt > 0);
468 time_steps = int(ceil((tend - t0) / dt));
469 assert(time_steps > 0);
470 }
471 else if (is_param_valid(args["time"], "time_steps"))
472 {
473 time_steps = args["time"]["time_steps"];
474 assert(time_steps > 0);
475 dt = (tend - t0) / time_steps;
476 assert(dt > 0);
477 }
478 else
479 {
480 throw std::runtime_error("This code should be unreachable!");
481 }
482 }
483 else if (is_param_valid(args["time"], "dt"))
484 {
485 // tend is already confirmed to be invalid, so time_steps must be valid
486 assert(is_param_valid(args["time"], "time_steps"));
487
488 dt = Units::convert(args["time"]["dt"], units.time());
489 assert(dt > 0);
490
491 time_steps = args["time"]["time_steps"];
492 assert(time_steps > 0);
493
494 tend = t0 + time_steps * dt;
495 }
496 else
497 {
498 // tend and dt are already confirmed to be invalid
499 throw std::runtime_error("This code should be unreachable!");
500 }
501 }
502 else if (num_valid == 3)
503 {
504 tend = Units::convert(args["time"]["tend"], units.time());
505 dt = Units::convert(args["time"]["dt"], units.time());
506 time_steps = args["time"]["time_steps"];
507
508 // Check that all parameters agree
509 if (abs(t0 + dt * time_steps - tend) > 1e-12)
510 {
511 log_and_throw_error("Exactly two of (tend, dt, time_steps) must be specified");
512 }
513 }
514
515 // Store these for use later
516 args["time"]["tend"] = tend;
517 args["time"]["dt"] = dt;
518 args["time"]["time_steps"] = time_steps;
519
520 logger().info("t0={}, dt={}, tend={}", t0, dt, tend);
521 }

523 void State::set_materials(std::vector<std::shared_ptr<assembler::Assembler>> &assemblers) const
524 {
525 const int size = (assembler->is_tensor() || assembler->is_fluid()) ? mesh->dimension() : 1;
526 for (auto &a : assemblers)
527 a->set_size(size);
528
529 if (!utils::is_param_valid(args, "materials"))
530 return;
531
532 if (!args["materials"].is_array() && args["materials"]["type"] == "AMIPS")
533 {
534 json transform_params = {};
535 transform_params["canonical_transformation"] = json::array();
536 if (!mesh->is_volume())
537 {
538 Eigen::MatrixXd regular_tri(3, 3);
539 regular_tri << 0, 0, 1,
540 1, 0, 1,
541 1. / 2., std::sqrt(3) / 2., 1;
542 regular_tri.transposeInPlace();
543 Eigen::MatrixXd regular_tri_inv = regular_tri.inverse();
544
545 const auto &mesh2d = *dynamic_cast<mesh::Mesh2D *>(mesh.get());
546 for (int e = 0; e < mesh->n_elements(); e++)
547 {
548 Eigen::MatrixXd transform;
549 mesh2d.compute_face_jacobian(e, regular_tri_inv, transform);
550 transform_params["canonical_transformation"].push_back(json({
551 {
552 transform(0, 0),
553 transform(0, 1),
554 },
555 {
556 transform(1, 0),
557 transform(1, 1),
558 },
559 }));
560 }
561 }
562 else
563 {
564 Eigen::MatrixXd regular_tet(4, 4);
565 regular_tet << 0, 0, 0, 1,
566 1, 0, 0, 1,
567 1. / 2., std::sqrt(3) / 2., 0, 1,
568 1. / 2., 1. / 2. / std::sqrt(3), std::sqrt(3) / 2., 1;
569 regular_tet.transposeInPlace();
570 Eigen::MatrixXd regular_tet_inv = regular_tet.inverse();
571
572 const auto &mesh3d = *dynamic_cast<mesh::Mesh3D *>(mesh.get());
573 for (int e = 0; e < mesh->n_elements(); e++)
574 {
575 Eigen::MatrixXd transform;
576 mesh3d.compute_cell_jacobian(e, regular_tet_inv, transform);
577 transform_params["canonical_transformation"].push_back(json({
578 {
579 transform(0, 0),
580 transform(0, 1),
581 transform(0, 2),
582 },
583 {
584 transform(1, 0),
585 transform(1, 1),
586 transform(1, 2),
587 },
588 {
589 transform(2, 0),
590 transform(2, 1),
591 transform(2, 2),
592 },
593 }));
594 }
595 }
596 transform_params["solve_displacement"] = true;
597 assembler->set_materials({}, transform_params, units);
598
599 return;
600 }
601
602 std::vector<int> body_ids(mesh->n_elements());
603 for (int i = 0; i < mesh->n_elements(); ++i)
604 body_ids[i] = mesh->get_body_id(i);
605
606 for (auto &a : assemblers)
607 a->set_materials(body_ids, args["materials"], units);
608 }
